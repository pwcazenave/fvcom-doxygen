175  SUBROUTINE wet_judge
176 
177 !------------------------------------------------------------------------------|
178 ! DETERMINE IF NODES/ELEMENTS ARE WET OR DRY |
179 !------------------------------------------------------------------------------|
180 
181  USE mod_prec
182  USE all_vars
183  USE mod_par
184  IMPLICIT NONE
185  REAL(SP) :: DTMP
186  INTEGER :: ITA_TEMP
187  INTEGER :: I,IL,IA,IB,K1,K2,K3,K4,K5,K6
188 
189  integer :: KT
190 
191  IF (dbg_set(dbg_sbr)) WRITE(ipt,*) "START: WET_JUDGE"
192 !
193 !--Determine If Node Points Are Wet/Dry Based on Depth Threshold---------------!
194 !
195  iswetn = 1
196  DO i = 1, m
197  dtmp = h(i) + elf(i)
198  IF((dtmp - min_depth) < 1.0e-5_sp) iswetn(i) = 0
199  END DO
200 
201 !
202 !--Determine if Cells are Wet/Dry Based on Depth Threshold---------------------!
203 !
204  iswetc = 1
205  DO i = 1, n
206  dtmp = max(elf(nv(i,1)),elf(nv(i,2)),elf(nv(i,3))) + &
207  min( h(nv(i,1)), h(nv(i,2)), h(nv(i,3)))
208  IF((dtmp - min_depth) < 1.0e-5_sp) iswetc(i) = 0
209  END DO
210 
211 !
212 !--A Secondary Condition for Nodal Dryness-(All Elements Around Node Are Dry)--!
213 !
214  DO i = 1, m
215  IF(sum(iswetc(nbve(i,1:ntve(i)))) == 0) iswetn(i) = 0
216  END DO
217 
218 !
219 !--Adjust Water Surface So It Does Not Go Below Minimum Depth------------------!
220 !
221  elf = max(elf,-h + min_depth)
222 
223 !
224 !--Recompute Element Based Depths----------------------------------------------!
225 !
226  DO i = 1, n
227  elf1(i) = one_third*(elf(nv(i,1))+elf(nv(i,2))+elf(nv(i,3)))
228  END DO
229 
230 !
231 !--Extend Element/Node Based Wet/Dry Flags to Domain Halo----------------------!
232 !
233 # if defined (MULTIPROCESSOR)
234  IF(par)THEN
235 
236  CALL aexchange(ec,myid,nprocs,iswetc)
237  CALL aexchange(nc,myid,nprocs,iswetn)
238 
239  END IF
240 # endif
241 
242  IF (dbg_set(dbg_sbr)) WRITE(ipt,*) "END: WET_JUDGE"
243  RETURN
244  END SUBROUTINE wet_judge
