114  ! Calculate Potential Density Based on Potential Temp and Salinity |
115  ! Pressure effects are incorported (Can Model Fresh Water < 4 Deg C) |
116  ! Ref: algorithms for computation of fundamental properties of |
117  ! seawater , Fofonoff and Millard. |

122  SUBROUTINE fofonoff_millard_2d(MYS,MYT,MYP,PREF, MYRHO)
123 
124  !------------------------------------------------------------------------------|
125  IMPLICIT NONE
126  REAL(SP), INTENT(IN), DIMENSION(:,:) :: MYS,MYT,MYP
127  REAL(SP), INTENT(IN) :: PREF
128  REAL(SP), INTENT(INOUT), DIMENSION(:,:):: MYRHO
129  INTEGER :: I,K,ub1,lb1,ub2,lb2
130  REAL(SP) ::PT,SVA,SIGMA
131  !==============================================================================|
132 
133  ! SET DIMS USING MY S
134  lb1 = lbound(mys,1)
135  ub1 = ubound(mys,1)
136 
137  lb2 = lbound(mys,2)
138  ub2 = ubound(mys,2)
139 
140  ! CHECK MY T
141  if(lb1 /= lbound(myt,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
142  if(ub1 /= ubound(myt,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
143 
144  if(lb2 /= lbound(myt,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
145  if(ub2 /= ubound(myt,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
146 
147  ! CHECK MY P
148  if(lb1 /= lbound(myp,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
149  if(ub1 /= ubound(myp,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
150 
151  if(lb2 /= lbound(myp,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
152  if(ub2 /= ubound(myp,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
153 
154  ! CHECK MY RHO
155  if(lb1 /= lbound(myrho,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
156  if(ub1 /= ubound(myrho,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
157 
158  if(lb2 /= lbound(myrho,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
159  if(ub2 /= ubound(myrho,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
160 
161 
162 
163  DO i=lb1,ub1
164  DO k=lb2,ub2
165 
166  pt = theta(mys(i,k),myt(i,k),myp(i,k),pref)
167  sva = svan(mys(i,k),pt,myp(i,k),sigma)
168  myrho(i,k) = sigma*1.e-3_sp
169  END DO
170  END DO
171 
172  !----------------TRANSFORM TO FACE CENTER--------------------------------------
173 
174  END SUBROUTINE fofonoff_millard_2d

254  SUBROUTINE dens2_2d(MYS,MYT,MYRHO)
255 
256  !==============================================================================|
257  IMPLICIT NONE
258  REAL(SP), INTENT(IN), DIMENSION(:,:) :: MYS,MYT
259  REAL(SP), INTENT(INOUT), DIMENSION(:,:):: MYRHO
260 
261 
262  INTEGER :: I,K,ub1,lb1,ub2,lb2
263  !==============================================================================|
264 
265  ! SET DIMS USING MY S
266  lb1 = lbound(mys,1)
267  ub1 = ubound(mys,1)
268 
269  lb2 = lbound(mys,2)
270  ub2 = ubound(mys,2)
271 
272  ! CHECK MY T
273  if(lb1 /= lbound(myt,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
274  if(ub1 /= ubound(myt,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
275 
276  if(lb2 /= lbound(myt,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
277  if(ub2 /= ubound(myt,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
278 
279  ! CHECK MY RHO
280  if(lb1 /= lbound(myrho,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
281  if(ub1 /= ubound(myrho,1)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
282 
283  if(lb2 /= lbound(myrho,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
284  if(ub2 /= ubound(myrho,2)) CALL fatal_error("EQS_OF_STATE:: Dimension mismatch!")
285 
286 
287  !
288  ! CALCULATE DENSITY FROM EQUATION OF STATE
289  !
290  myrho =mys*mys*mys*6.76786136e-6_sp &
291  & - mys*mys*4.8249614e-4_sp &
292  & + mys*8.14876577e-1_sp &
293  & - 0.22584586e0_sp
294 
295  myrho = myrho * &
296  & ( myt*myt*myt*1.667e-8_sp &
297  & - myt*myt*8.164e-7_sp &
298  & + myt*1.803e-5_sp &
299  & )
300 
301  myrho = myrho+1.0_sp &
302  & - myt*myt*myt*1.0843e-6_sp &
303  & + myt*myt*9.8185e-5_sp &
304  & - myt*4.786e-3_sp
305 
306  myrho = myrho* &
307  & ( mys*mys*mys*6.76786136e-6_sp &
308  & - mys*mys*4.8249614e-4_sp &
309  & + mys*8.14876577e-1_sp &
310  & +3.895414e-2_sp &
311  & )
312 
313  myrho = myrho &
314  & - (myt-3.98_sp) *(myt-3.98_sp) * (myt+283.0_sp) / (503.57_sp*(myt+67.26_sp))
315 
316  !
317  ! CALCULATE RHO
318  !
319  myrho = myrho*1.e-3_sp
320 
321  END SUBROUTINE dens2_2d

713  ! salinity s4 (ipss-78) |
714  ! spec. vol. ana. svan m**3/kg *1.0e-8 |
715  ! density ana. sigma kg/m**3 |
716  ! |
717  ! check value: svan=981.3021 e-8 m**3/kg. for s = 40 (ipss-78), |
718  ! t = 40 deg c, p0= 10000 decibars. |

739  ! convert pressure to bars and take square root salinity.
740

780  ! |

834  dvan=sva/(v350p*(v350p+sva))
835  sigma=dr350+dr35p-dvan
836
