934  do i=1,nsta
935  temp(i,:) = 0.0_sp
936  do j =1, ntvegl(node_sta(i))
937  temp(i,1:k1m1) = temp(i,1:k1m1) + var(nbvegl(node_sta(i),j),1:k1m1)
