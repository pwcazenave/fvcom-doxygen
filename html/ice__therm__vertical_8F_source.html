446 ! !ROUTINE: frzmlt_bottom_lateral - bottom and lateral heat fluxes
447 !
448 ! !DESCRIPTION:
449 !
450 ! Adjust frzmlt to account for changes in fhnet since from\_coupler.
451 ! Compute heat flux to bottom surface.
452 ! Compute fraction of ice that melts laterally.
453 !
454 ! !REVISION HISTORY:
455 !
456 ! authors C. M. Bitz, UW
457 ! William H. Lipscomb, LANL
458 ! Elizabeth C. Hunke, LANL
459 !
460 ! !INTERFACE:
461 !
462  subroutine frzmlt_bottom_lateral (Tbot, fbot, rside)
463 !
464 ! !USES:
465 !
466 ! !INPUT/OUTPUT PARAMETERS:
467 !
468  real (kind=dbl_kind), dimension(ilo:ihi,jlo:jhi), intent(out) :: &
469  Tbot & ! ice bottom surface temperature (deg C)
470  , fbot & ! heat flux to ice bottom (W/m^2)
471  , rside ! fraction of ice that melts laterally
472 !
473 !EOP
474 !
475  integer (kind=int_kind) :: &
476  i, j & ! horizontal indices
477  , ni & ! thickness category index
478  , k & ! layer index
479  , ij & ! horizontal index, combines i and j loops
480  , icells ! number of cells with ice melting
481 
482  integer (kind=int_kind), dimension(1:(ihi-ilo+1)*(jhi-jlo+1)) :: &
483  indxi, indxj ! compressed indices for cells with ice melting
484 
485  real (kind=dbl_kind), dimension(ilo:ihi,jlo:jhi) :: &
486  etot & ! total energy in column
487  , fside ! lateral heat flux (W/m^2)
488 
489  real (kind=dbl_kind) :: &
490  deltaT &! SST - Tbot >= 0
491  , ustar &! skin friction velocity for fbot (m/s)
492  , wlat &! lateral melt rate (m/s)
493  , xtmp ! temporarty variable
494 
495  ! Parameters for bottom melting
496 
497  ! 0.006 = unitless param for basal heat flx ala McPhee and Maykut
498 
499  real (kind=dbl_kind), parameter :: &
500  cpchr = -cp_ocn*rhow*0.006_dbl_kind &
501  , ustar_min = 5.e-3_dbl_kind
502 
503  ! Parameters for lateral melting
504 
505  real (kind=dbl_kind), parameter :: &
506  floediam = 300.0_dbl_kind & ! effective floe diameter (m)
507  , alpha = 0.66_dbl_kind & ! constant from Steele (unitless)
508  , m1 = 1.6e-6_dbl_kind & ! constant from Maykut & Perovich
509  ! (m/s/deg^(-m2))
510  , m2 = 1.36_dbl_kind ! constant from Maykut & Perovich
511  ! (unitless)
512 
513  !-----------------------------------------------------------------
514  ! Set bottom ice surface temperature and initialize fluxes.
515  !-----------------------------------------------------------------
516 
517 
518  do j = jlo, jhi
519  do i = ilo, ihi
520  tbot(i,j) = tf(i,j)
521  fbot(i,j) = c0i
522  rside(i,j) = c0i
523  fside(i,j) = c0i
524  etot(i,j) = c0i
525  enddo ! i
526  enddo ! j
527 
528  !-----------------------------------------------------------------
529  ! Identify grid cells where ice can melt.
530  !-----------------------------------------------------------------
531 
532  icells = 0
533  do j = jlo, jhi
534  do i = ilo, ihi
535  if (aice(i,j) > puny .and. frzmlt(i,j) < c0i) then ! ice can melt
536  icells = icells + 1
537  indxi(icells) = i
538  indxj(icells) = j
539  endif
540  enddo ! i
541  enddo ! j
542 
543  do ij = 1, icells ! cells where ice can melt
544  i = indxi(ij)
545  j = indxj(ij)
546 
547  !-----------------------------------------------------------------
548  ! Use boundary layer theory for fbot.
549  ! See Maykut and McPhee (1995): JGR, 100, 24,691-24,703.
550  !-----------------------------------------------------------------
551 
552  deltat = max((sst(i,j)-tbot(i,j)),c0i)
553 
554  ! strocnx has units N/m^2 so strocnx/rho has units m^2/s^2
555  ustar = sqrt(sqrt(strocnxt(i,j)**2+strocnyt(i,j)**2)/rhow)
556  ustar = max(ustar,ustar_min*ustar_scale)
557 
558  fbot(i,j) = cpchr * deltat * ustar ! < 0
559  fbot(i,j) = max(fbot(i,j), frzmlt(i,j)) ! frzmlt < fbot < 0
560 
561 !!! uncomment to use all frzmlt for standalone runs
562 ! fbot(i,j) = min (c0i, frzmlt(i,j))
563 
564  !-----------------------------------------------------------------
565  ! Compute rside. See these references:
566  ! Maykut and Perovich (1987): JGR, 92, 7032-7044
567  ! Steele (1992): JGR, 97, 17,729-17,738
568  !-----------------------------------------------------------------
569 
570  wlat = m1 * deltat**m2 ! Maykut & Perovich
571 ! rside(i,j) = wlat*dt*pi/(alpha*floediam) ! Steele
572  rside(i,j) = wlat*dtice*pi/(alpha*floediam) ! Steele
573  rside(i,j) = max(c0i,min(rside(i,j),c1i))
574 
575  enddo ! ij
576 
577  !-----------------------------------------------------------------
578  ! Compute heat flux associated with this value of rside.
579  !-----------------------------------------------------------------
580 
581  do ni = 1, ncat
582 
583  ! melting energy/unit area in each column, etot < 0
584  do ij = 1, icells
585  i = indxi(ij)
586  j = indxj(ij)
587  etot(i,j) = esnon(i,j,ni)
588  enddo ! ij
589 
590  do k = 1, nilyr
591 !DIR$ CONCURRENT !Cray
592 !cdir nodep !NEC
593 !ocl novrec !Fujitsu
594  do ij = 1, icells
595  i = indxi(ij)
596  j = indxj(ij)
597  etot(i,j) = etot(i,j) + eicen(i,j,ilyr1(ni)+k-1)
598  enddo ! ij
599  enddo ! k
600 
601 !DIR$ CONCURRENT !Cray
602 !cdir nodep !NEC
603 !ocl novrec !Fujitsu
604  do ij = 1, icells
605  i = indxi(ij)
606  j = indxj(ij)
607  ! lateral heat flux
608 ! fside(i,j) = fside(i,j) + rside(i,j)*etot(i,j)/dt ! fside < 0
609  fside(i,j) = fside(i,j) + rside(i,j)*etot(i,j)/dtice ! fside < 0
610  enddo ! ij
611 
612  enddo ! n
613 
614  !-----------------------------------------------------------------
615  ! Limit bottom and lateral heat fluxes if necessary.
616  !-----------------------------------------------------------------
617 
618 !DIR$ CONCURRENT !Cray
619 !cdir nodep !NEC
620 !ocl novrec !Fujitsu
621 
622 
623  do ij = 1, icells
624  i = indxi(ij)
625  j = indxj(ij)
626 
627  xtmp = frzmlt(i,j)/(fbot(i,j) + fside(i,j) + puny)
628  xtmp = min(xtmp, c1i)
629  fbot(i,j) = fbot(i,j) * xtmp
630  rside(i,j) = rside(i,j) * xtmp
631  enddo ! ij
632 
633  end subroutine frzmlt_bottom_lateral

2476  !--------------------------------------------------------------
2477 
2478  if (hsn(i,j) > puny) then ! add snow with enthalpy qsn
2479  dhs = econ / (qsn(i,j) - rhos*lvap) ! econ < 0, dhs > 0
2480  hsn(i,j) = hsn(i,j) + dhs
2481  mvap(i,j) = mvap(i,j) + dhs*rhos
2482  else ! add ice with enthalpy qin(i,j,1)
2483  dhi = econ / (qin(i,j,1) - rhoi*lvap) ! econ < 0, dhi > 0

2485  mvap(i,j) = mvap(i,j) + dhi*rhoi

2490  !--------------------------------------------------------------

2494  esub(i,j) = esub(i,j) - dhs*qsub
2495  esub(i,j) = max(esub(i,j), c0i) ! in case of roundoff error
2496  mvap(i,j) = mvap(i,j) + dhs*rhos

2498 
2499  !--------------------------------------------------------------
2500  ! Sublimation of ice (mvap < 0)
2501  !--------------------------------------------------------------
2502

2504 !DIR$ CONCURRENT !Cray
2505 !cdir nodep !NEC
2506 !ocl novrec !Fujitsu

2508  i = indxi(ij)
2509  j = indxj(ij)
2510  qsub = qin(i,j,k) - rhoi*lvap ! qsub < 0

2512  hnew(i,j,k) = hnew(i,j,k) + dhi
2513  esub(i,j) = esub(i,j) - dhi*qsub
2514  esub(i,j) = max(esub(i,j), c0i)
2515  mvap(i,j) = mvap(i,j) + dhi*rhoi
2516  enddo ! ij
2517  enddo ! k
2518

2520  ! Top melt
2521  ! (There is no top growth because there is no liquid water to
2522  ! freeze at the top.)
2523  !-----------------------------------------------------------------
2524

2526  ! Melt snow
2527  !--------------------------------------------------------------
2528 !DIR$ CONCURRENT !Cray
2529 !cdir nodep !NEC
2530 !ocl novrec !Fujitsu
2531  do ij = 1, icells
2532  i = indxi(ij)
2533  j = indxj(ij)
2534

2536  w1 = (fsurf(i,j) - fcondtop(i,j)) * dtice
2537  etop_mlt(i,j) = max(w1, c0i) ! etop_mlt > 0
2538  dhs = max(-hsn(i,j), etop_mlt(i,j)/qsn(i,j)) ! qsn < 0, dhs < 0
2539  hsn(i,j) = hsn(i,j) + dhs
2540  etop_mlt(i,j) = etop_mlt(i,j) - dhs*qsn(i,j)
2541  etop_mlt(i,j) = max(etop_mlt(i,j), c0i) ! in case of roundoff error
2542 
2543  ! history diagnostics
2544  if (dhs < -puny .and. mlt_onset(i,j) < puny) &
2545  mlt_onset(i,j) = yday
2546 
2547  enddo ! ij
2548

2555 !cdir nodep !NEC
2556 !ocl novrec !Fujitsu
2557  do ij = 1, icells
2558  i = indxi(ij)
2559  j = indxj(ij)
2560

2562  hnew(i,j,k) = hnew(i,j,k) + dhi ! qin < 0, dhi < 0
2563  etop_mlt(i,j) = etop_mlt(i,j) - dhi*qin(i,j,k)
2564  etop_mlt(i,j) = max(etop_mlt(i,j), c0i)

2569  meltt(i,j) = meltt(i,j) - dhi*aicen(i,j,ni)
2570 
2571  enddo ! ij
2572  enddo ! k

2575  ! Bottom growth and melt
2576  !----------------------------------------------------------------
2577 
2578 !DIR$ CONCURRENT !Cray
2579 !cdir nodep !NEC
2580 !ocl novrec !Fujitsu
2581  do ij = 1, icells
2582  i = indxi(ij)
2583  j = indxj(ij)
2584 
2585 ! w1 = (fcondbot(i,j) - fbot(i,j)) * dt
2586  w1 = (fcondbot(i,j) - fbot(i,j)) * dtice
2587  ebot_mlt(i,j) = max(w1, c0i) ! ebot_mlt > 0
2588  ebot_gro = min(w1, c0i) ! ebot_gro < 0

2596  + lfresh * (c1i-tmlt(nilyr+1)/tbot(i,j)) &
2597  - cp_ocn * tmlt(nilyr+1))

2600

2606

2608  congel(i,j) = congel(i,j) + dhi*aicen(i,j,ni)

2625  j = indxj(ij)

2636  enddo ! k

2640  !--------------------------------------------------------------

2642 !DIR$ CONCURRENT !Cray
2643 !cdir nodep !NEC
2644 !ocl novrec !Fujitsu
2645  do ij = 1, icells
2646  i = indxi(ij)
2647  j = indxj(ij)
2648 
2649  dhs = max(-hsn(i,j), ebot_mlt(i,j)/qsn(i,j))
2650  hsn(i,j) = hsn(i,j) + dhs ! qsn < 0, dhs < 0
2651  ebot_mlt(i,j) = ebot_mlt(i,j) - dhs*qsn(i,j)
2652  ebot_mlt(i,j) = max(ebot_mlt(i,j), c0i)
2653 
2654  ! snow-to-ice conversion (very rare)
2655  if (hsn(i,j) > puny .and. ebot_mlt(i,j) > puny*lfresh) then
2656  hnew(i,j,1) = hsn(i,j) * rhos/rhoi ! reduce thickness
2657  qin(i,j,1) = qsn(i,j) * rhoi/rhos ! increase q to conserve energy
2658  hsn(i,j) = c0i
2659  endif
2660 
2661  enddo ! ij
2662

2664  ! Give the ocean any energy left over after melting
2665  !-----------------------------------------------------------------
2666

2669 !ocl novrec !Fujitsu

2672  j = indxj(ij)
2673

2675  + (esub(i,j) + etop_mlt(i,j) + ebot_mlt(i,j))/dtice
2676 ! + (esub(i,j) + etop_mlt(i,j) + ebot_mlt(i,j))/dt
2677 
2678  enddo ! ij

2682 !---!-------------------------------------------------------------------
2683

2685  ! Initialize the new ice thickness.

2688  !-----------------------------------------------------------------

2690  do ij = 1, icells
2691  i = indxi(ij)
2692  j = indxj(ij)
2693  hin(i,j) = c0i
2694  efinal(i,j) = hsn(i,j)*qsn(i,j) - mvap(i,j)*lvap
2695  enddo
2696 
2697  !-----------------------------------------------------------------
2698  ! Compute the new ice thickness.
2699  ! Initialize h*q for new layers.
2700  !-----------------------------------------------------------------
2701 
2702  do k = 1, nilyr
2703 !DIR$ CONCURRENT !Cray
2704 !cdir nodep !NEC
2705 !ocl novrec !Fujitsu
2706  do ij = 1, icells
2707  i = indxi(ij)
2708  j = indxj(ij)
2709  hin(i,j) = hin(i,j) + hnew(i,j,k)
2710  hq(i,j,k) = c0i
2711  enddo ! ij
2712  enddo ! k
2713

2716  ! Compute depths zold of old layers (unequal thicknesses).
2717  ! Compute depths znew of new layers (all the same thickness).
2718  !-----------------------------------------------------------------
2719 
2720 !DIR$ CONCURRENT !Cray
2721 !cdir nodep !NEC
2722 !ocl novrec !Fujitsu
2723  do ij = 1, icells
2724  i = indxi(ij)
2725  j = indxj(ij)
2726

2728  hlyr(i,j) = hin(i,j) / real(nilyr,kind=dbl_kind)
2729  rhlyr(i,j) = c1i / hlyr(i,j)
2730  else
2731  hin(i,j) = c0i
2732  hlyr(i,j) = c0i
2733  rhlyr(i,j) = c0i

2737  znew(i,j,0) = c0i

2740  znew(i,j,nilyr) = hin(i,j)

2743

2747  j = indxj(ij)

2752  end do ! ij
2753  enddo ! k
2754

2758 
2759  do k = 1, nilyr
2760  do kold = 1, nilyr
2761 !DIR$ CONCURRENT !Cray
2762 !cdir nodep !NEC
2763 !ocl novrec !Fujitsu
2764  do ij = 1, icells
2765  i = indxi(ij)
2766  j = indxj(ij)
2767 
2768  hovlp = min(zold(i,j,kold), znew(i,j,k)) &
2769  - max(zold(i,j,kold-1), znew(i,j,k-1))
2770  hovlp = max(hovlp, c0i)

2773 
2774  enddo ! ij
2775  enddo ! kold
2776  enddo ! k
2777 
2778  !-----------------------------------------------------------------
2779  ! Compute new values of qin and increment ice-snow energy.
2780  ! Note: Have to finish previous loop before updating qin
2781  ! in this loop.
2782  !-----------------------------------------------------------------
2783 
2784  do k = 1, nilyr
2785 !DIR$ CONCURRENT !Cray
2786 !cdir nodep !NEC
2787 !ocl novrec !Fujitsu
2788  do ij = 1, icells
2789  i = indxi(ij)
2790  j = indxj(ij)
2791 
2792  qin(i,j,k) = hq(i,j,k) * rhlyr(i,j)
2793  efinal(i,j) = efinal(i,j) + hlyr(i,j)*qin(i,j,k)
2794 
2795  enddo ! ij
2796  enddo !
2797 
2798  end subroutine thickness_changes
