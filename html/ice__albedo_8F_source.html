93 ! !IROUTINE: albedos - compute snow/ice albedos and aggregate
94 !
95 ! !INTERFACE:
96 !
97  subroutine albedos
98 !
99 ! !DESCRIPTION:
100 !
101 ! Compute albedos and aggregate them \\
102 ! note: ice albedo is zero if no ice present
103 !
104 ! !REVISION HISTORY:
105 !
106 ! authors: Bruce P. Briegleb, NCAR
107 ! Elizabeth C. Hunke, LANL
108 !
109 ! !USES:
110 !
111  use ice_constants
112  use ice_grid
113  use ice_state
114 !
115 ! !INPUT/OUTPUT PARAMETERS:
116 !
117 !EOP
118 !
119  real (kind=dbl_kind), parameter :: &
120  ahmax = 0.5_dbl_kind &! thickness above which ice albedo constant (m)
121  , dt_mlt = 1._dbl_kind &! change in temp to give dalb_mlt albedo change
122  , dalb_mlt = -0.075_dbl_kind &! albedo change per dT_mlt change
123  ! in temp for ice
124  , dalb_mltv = -0.100_dbl_kind &! albedo vis change per dT_mlt change
125  ! in temp for snow
126  , dalb_mlti = -0.150_dbl_kind ! albedo nir change per dT_mlt change
127  ! in temp for snow
128 
129  integer (kind=int_kind) :: i, j, ni
130 
131  real (kind=dbl_kind) :: &
132  hi &! ice thickness (m)
133  , hs &! snow thickness (m)
134  , albo &! effective ocean albedo, function of ice thickness
135  , asnow &! snow-covered area fraction
136  , asnwv &! snow albedo, visible
137  , asnwi &! snow albedo, near IR
138  , fh &! piecewise linear function of thickness
139  , fT &! piecewise linear function of surface temperature
140  , dTs &! difference of Tsfc and Timelt
141  , fhtan ! factor used in albedo dependence on ice thickness
142 
143  integer (kind=int_kind) :: &
144  icells &! number of ice/ocean grid cells
145  , ij ! horizontal index, combines i and j loops
146 
147  integer (kind=int_kind), dimension (1:(ihi-ilo+1)*(jhi-jlo+1)) :: &
148  indxi &! compressed indices for ice/ocean cells
149  , indxj
150 
151  fhtan = atan(ahmax*c4i)
152 
153  icells = 0
154  do j = jlo, jhi
155  do i = ilo, ihi
156  if (tmask(i,j)) then
157  icells = icells + 1
158  indxi(icells) = i
159  indxj(icells) = j
160  endif ! tmask
161  enddo
162  enddo
163 
164  !-----------------------------------------------------------------
165  ! albedo for each thickness category
166  !-----------------------------------------------------------------
167 
168  do ni = 1, ncat
169 !cdir$ ivdep !Cray
170 !cdir nodep !NEC
171 !ocl novrec !Fujitsu
172  do ij = 1, icells
173  i = indxi(ij)
174  j = indxj(ij)
175 
176  if (aicen(i,j,ni) > puny) then
177  hi = vicen(i,j,ni) / aicen(i,j,ni)
178  hs = vsnon(i,j,ni) / aicen(i,j,ni)
179 
180  ! bare ice, thickness dependence
181  fh = min(atan(hi*c4i)/fhtan,c1i)
182  albo = albocn*(c1i-fh)
183  alvdfn(i,j,ni) = albicev*fh + albo
184  alidfn(i,j,ni) = albicei*fh + albo
185 
186  ! bare ice, temperature dependence
187  dts = timelt - tsfcn(i,j,ni)
188  ft = min(dts/dt_mlt-c1i,c0i)
189  alvdfn(i,j,ni) = alvdfn(i,j,ni) - dalb_mlt*ft
190  alidfn(i,j,ni) = alidfn(i,j,ni) - dalb_mlt*ft
191 
192  ! avoid negative albedos for thin, bare, melting ice
193  alvdfn(i,j,ni) = max(alvdfn(i,j,ni), albocn)
194  alidfn(i,j,ni) = max(alidfn(i,j,ni), albocn)
195 
196  if( hs > puny ) then
197 
198  ! fractional area of snow on ice (thickness dependent)
199  asnow = hs / ( hs + snowpatch )
200  asnwv = albsnowv
201  asnwi = albsnowi
202 
203  ! snow on ice, temperature dependence
204  asnwv = asnwv - dalb_mltv*ft
205  asnwi = asnwi - dalb_mlti*ft
206 
207  ! combine ice and snow albedos
208  alvdfn(i,j,ni) = alvdfn(i,j,ni)*(c1i-asnow) + &
209  asnwv*asnow
210  alidfn(i,j,ni) = alidfn(i,j,ni)*(c1i-asnow) + &
211  asnwi*asnow
212  endif ! hs > puny
213 
214  alvdrn(i,j,ni) = alvdfn(i,j,ni)
215  alidrn(i,j,ni) = alidfn(i,j,ni)
216 
217  else ! no ice
218  alvdfn(i,j,ni) = albocn
219  alidfn(i,j,ni) = albocn
220  alvdrn(i,j,ni) = albocn
221  alidrn(i,j,ni) = albocn
222  endif ! aicen > puny
223  enddo ! ij
224  enddo ! ncat
225 
226  !-----------------------------------------------------------------
227  ! aggregate
228  !-----------------------------------------------------------------
229 
230  do j = jlo, jhi
231  do i = ilo, ihi
232  alvdf(i,j) = c0i
233  alidf(i,j) = c0i
234  alvdr(i,j) = c0i
235  alidr(i,j) = c0i
236  enddo
237  enddo
238 
239  do ni = 1, ncat
240 !cdir$ ivdep !Cray
241 !cdir nodep !NEC
242 !ocl novrec !Fujitsu
243  do ij = 1, icells
244  i = indxi(ij)
245  j = indxj(ij)
246  if (aice(i,j) > puny) then
247  alvdf(i,j) = alvdf(i,j) + alvdfn(i,j,ni)*aicen(i,j,ni)
248  alidf(i,j) = alidf(i,j) + alidfn(i,j,ni)*aicen(i,j,ni)
249  alvdr(i,j) = alvdr(i,j) + alvdrn(i,j,ni)*aicen(i,j,ni)
250  alidr(i,j) = alidr(i,j) + alidrn(i,j,ni)*aicen(i,j,ni)
251  endif
252  enddo ! ij
253  enddo ! ncat
254 
255 !cdir$ ivdep !Cray
256 !cdir nodep !NEC
257 !ocl novrec !Fujitsu
258  do ij = 1, icells
259  i = indxi(ij)
260  j = indxj(ij)
261  if (aice(i,j) > puny) then
262  alvdf(i,j) = alvdf(i,j) / aice(i,j)
263  alidf(i,j) = alidf(i,j) / aice(i,j)
264  alvdr(i,j) = alvdr(i,j) / aice(i,j)
265  alidr(i,j) = alidr(i,j) / aice(i,j)
266  endif ! aicen > puny
267  enddo ! ij
268 
269  end subroutine albedos
