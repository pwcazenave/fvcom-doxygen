30  SUBROUTINE advection_edge_gcn(XFLUX,YFLUX)
31 !==============================================================================|
32 ! Calculate the Advection and Diffusion Terms of 3D Velocity Field |
33 ! These Terms will be vertically integrated to form the Mean Terms in |
34 ! the Gx and Gy Terms of the External Mode Equation |
35 !==============================================================================|
36 
37  USE all_vars
38  USE mod_utils
39  USE bcs
40  USE mod_spherical
41  USE mod_northpole
42  USE mod_wd
43 
44 # if defined (MEAN_FLOW)
45  USE mod_meanflow
46  USE mod_obcs2
47  USE mod_obcs3
48 # endif
49 
50 # if defined (THIN_DAM)
51  USE mod_dam !Jadon
52 # endif
53 
54  IMPLICIT NONE
55  REAL(SP), INTENT(OUT), DIMENSION(0:NT,KB) :: XFLUX,YFLUX
56  REAL(SP) :: DIJ
57  REAL(SP) :: COFA1,COFA2,COFA3,COFA4,COFA5,COFA6,COFA7,COFA8
58  REAL(SP) :: XADV,YADV,TXXIJ,TYYIJ,TXYIJ,UN_TMP
59  REAL(SP) :: VISCOF,VISCOF1,VISCOF2,TEMP,TPA,TPB
60  REAL(SP) :: XIJA,YIJA,XIJB,YIJB,UIJ,VIJ
61  REAL(SP) :: FACT,FM1
62  INTEGER :: I,IA,IB,J1,J2,K1,K2,K3,K4,K5,K6,K,II,J,I1,I2
63  REAL(SP) :: ISWETTMP
64 
65 !# if defined (THIN_DAM)
66  REAL(SP) :: A1UIA1,A1UIA2,A1UIA3,A1UIA4,A2UIA1,A2UIA2,A2UIA3,A2UIA4
67  REAL(SP) :: A1UIB1,A1UIB2,A1UIB3,A1UIB4,A2UIB1,A2UIB2,A2UIB3,A2UIB4
68  INTEGER :: J11,J12,J21,J22,E1,E2,ISBCE1,ISBC_TMP,IB_TMP
69  LOGICAL :: ISMATCH
70 !# endif

72 # if defined (LIMITED_NO)
73  REAL(SP) :: UIJ1,VIJ1,UIJ2,VIJ2,FXX,FYY

85  SELECT CASE(horizontal_mixing_type)

88  fm1 = 0.0_sp
89  CASE('constant')

91  fm1 = 1.0_sp
92  CASE DEFAULT

94  & trim(horizontal_mixing_type) )

97 
98 !
99 !--Initialize Variables--------------------------------------------------------|
100 !
101  xflux = 0.0_sp
102  yflux = 0.0_sp
103 
104 !
105 !--Loop Over Edges and Accumulate Fluxes-For Each Element----------------------|
106 !
107 # if !defined (LIMITED_NO)
108  ALLOCATE(uij1(ne),vij1(ne),uij2(ne),vij2(ne),stat=error)
109  IF(error /= 0) &
110  & CALL fatal_error("The arrays UIJ1,VIJ1,UIJ2 and VIJ2 can not be allocated.")
111 
112  ALLOCATE(ualfa(0:nt),valfa(0:nt),stat=error)
113  IF(error /= 0) &
114  & CALL fatal_error("The arrays UALFA,VALFA can not be allocated.")

118  & CALL fatal_error("The arrays FXX,FYY can not be allocated.")

120  DO k=1,kbm1
121  uij1=0.0_sp; vij1=0.0_sp; uij2=0.0_sp; vij2=0.0_sp
122  ualfa=1.0_sp; valfa=1.0_sp
123  fxx=0.0_sp; fyy=0.0_sp
124 
125  DO i=1,ne
126  ia=iec(i,1)
127  ib=iec(i,2)
128 
129 # if defined (WET_DRY)
130 # if !defined (SEMI_IMPLICIT)
131  IF(iswetct(ia)*iswetc(ia) == 1 .OR. iswetct(ib)*iswetc(ib) == 1)THEN
132 # else
133  IF(iswetct(ia) == 1 .OR. iswetct(ib) == 1)THEN
134 # endif
135 # endif
136  j1=ienode(i,1)
137  j2=ienode(i,2)
138 
139  k1=nbe(ia,1)
140  k2=nbe(ia,2)
141  k3=nbe(ia,3)
142  k4=nbe(ib,1)
143  k5=nbe(ib,2)
144  k6=nbe(ib,3)
145 # if defined (SPHERICAL)
146  xija=dltxne(i,1)
147  yija=dltyne(i,1)
148  xijb=dltxne(i,2)
149  yijb=dltyne(i,2)
150 # if defined (THIN_DAM)
151  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k<=kdam1(ia))xijb=dltxne_dam_match(i)
152  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k<=kdam1(ia))yijb=dltyne_dam_match(i)
153 # endif
154 # else
155  xija=xijc(i)-xc(ia)
156  yija=yijc(i)-yc(ia)
157  xijb=xijc(i)-xc(ib)
158  yijb=yijc(i)-yc(ib)
159 # if defined (THIN_DAM)
160  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k<=kdam1(ia))xijb=xijc(i)-xc(e_dam_match(ia))
161  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k<=kdam1(ia))yijb=yijc(i)-yc(e_dam_match(ia))
162 # endif
163 # endif
164 
165  dij= 0.5_sp*(dt(j1)*dz(j1,k)+dt(j2)*dz(j2,k))
166  ib_tmp = ib
167 
168 !--------------Used for Dam Model by Jadon----------------------
169 # if defined (THIN_DAM)
170  a1uia1 = a1u(ia,1)
171  a1uia2 = a1u(ia,2)
172  a1uia3 = a1u(ia,3)
173  a1uia4 = a1u(ia,4)
174  a2uia1 = a2u(ia,1)
175  a2uia2 = a2u(ia,2)
176  a2uia3 = a2u(ia,3)
177  a2uia4 = a2u(ia,4)
178 
179  a1uib1 = a1u(ib,1)
180  a1uib2 = a1u(ib,2)
181  a1uib3 = a1u(ib,3)
182  a1uib4 = a1u(ib,4)
183  a2uib1 = a2u(ib,1)
184  a2uib2 = a2u(ib,2)
185  a2uib3 = a2u(ib,3)
186  a2uib4 = a2u(ib,4)
187 
188  IF(isbce(ia) == 1 .AND. k <= kdam1(ia))THEN
189  a1uia1 = a1u_dam(ia,1)
190  a1uia2 = a1u_dam(ia,2)
191  a1uia3 = a1u_dam(ia,3)
192  a1uia4 = a1u_dam(ia,4)
193  a2uia1 = a2u_dam(ia,1)
194  a2uia2 = a2u_dam(ia,2)
195  a2uia3 = a2u_dam(ia,3)
196  a2uia4 = a2u_dam(ia,4)
197  IF(k1 == 0)k1 = nbe_dam(ia)
198  IF(k2 == 0)k2 = nbe_dam(ia)
199  IF(k3 == 0)k3 = nbe_dam(ia)
200  END IF
201 
202  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k<=kdam1(ia))ib_tmp=e_dam_match(ia)
203  IF(isbce(ib_tmp) == 1 .AND. k <= kdam1(ib_tmp))THEN
204  k4=nbe(ib_tmp,1)
205  k5=nbe(ib_tmp,2)
206  k6=nbe(ib_tmp,3)
207  a1uib1 = a1u_dam(ib_tmp,1)
208  a1uib2 = a1u_dam(ib_tmp,2)
209  a1uib3 = a1u_dam(ib_tmp,3)
210  a1uib4 = a1u_dam(ib_tmp,4)
211  a2uib1 = a2u_dam(ib_tmp,1)
212  a2uib2 = a2u_dam(ib_tmp,2)
213  a2uib3 = a2u_dam(ib_tmp,3)
214  a2uib4 = a2u_dam(ib_tmp,4)
215  IF(k4 == 0)k4 = nbe_dam(ib_tmp)
216  IF(k5 == 0)k5 = nbe_dam(ib_tmp)
217  IF(k6 == 0)k6 = nbe_dam(ib_tmp)
218  END IF
219 # else
220  a1uia1 = a1u(ia,1)
221  a1uia2 = a1u(ia,2)
222  a1uia3 = a1u(ia,3)
223  a1uia4 = a1u(ia,4)
224  a2uia1 = a2u(ia,1)
225  a2uia2 = a2u(ia,2)
226  a2uia3 = a2u(ia,3)
227  a2uia4 = a2u(ia,4)
228 
229  a1uib1 = a1u(ib_tmp,1)
230  a1uib2 = a1u(ib_tmp,2)
231  a1uib3 = a1u(ib_tmp,3)
232  a1uib4 = a1u(ib_tmp,4)
233  a2uib1 = a2u(ib_tmp,1)
234  a2uib2 = a2u(ib_tmp,2)
235  a2uib3 = a2u(ib_tmp,3)
236  a2uib4 = a2u(ib_tmp,4)
237 # endif
238 !---------------------------------------------------------------
239  !!FORM THE LEFT FLUX
240  cofa1=a1uia1*u(ia,k)+a1uia2*u(k1,k)+a1uia3*u(k2,k)+a1uia4*u(k3,k)
241  cofa2=a2uia1*u(ia,k)+a2uia2*u(k1,k)+a2uia3*u(k2,k)+a2uia4*u(k3,k)
242  cofa5=a1uia1*v(ia,k)+a1uia2*v(k1,k)+a1uia3*v(k2,k)+a1uia4*v(k3,k)
243  cofa6=a2uia1*v(ia,k)+a2uia2*v(k1,k)+a2uia3*v(k2,k)+a2uia4*v(k3,k)
244 ! COFA1=A1U(IA,1)*U(IA,K)+A1U(IA,2)*U(K1,K)+A1U(IA,3)*U(K2,K)+A1U(IA,4)*U(K3,K)
245 ! COFA2=A2U(IA,1)*U(IA,K)+A2U(IA,2)*U(K1,K)+A2U(IA,3)*U(K2,K)+A2U(IA,4)*U(K3,K)
246 ! COFA5=A1U(IA,1)*V(IA,K)+A1U(IA,2)*V(K1,K)+A1U(IA,3)*V(K2,K)+A1U(IA,4)*V(K3,K)
247 ! COFA6=A2U(IA,1)*V(IA,K)+A2U(IA,2)*V(K1,K)+A2U(IA,3)*V(K2,K)+A2U(IA,4)*V(K3,K)
248 
249  uij1(i)=cofa1*xija+cofa2*yija
250  vij1(i)=cofa5*xija+cofa6*yija
251  ualfa_tmp=abs(u(ia,k)-u(ib_tmp,k))/abs(uij1(i)+epsilon(eps))
252  valfa_tmp=abs(v(ia,k)-v(ib_tmp,k))/abs(vij1(i)+epsilon(eps))
253  IF(ualfa_tmp > 1)ualfa_tmp = 1.0_sp
254  IF(valfa_tmp > 1)valfa_tmp = 1.0_sp
255  ualfa(ia)=min(ualfa(ia),ualfa_tmp)
256  valfa(ia)=min(valfa(ia),valfa_tmp)
257 
258  !!FORM THE RIGHT FLUX
259  cofa3=a1uib1*u(ib_tmp,k)+a1uib2*u(k4,k)+a1uib3*u(k5,k)+a1uib4*u(k6,k)
260  cofa4=a2uib1*u(ib_tmp,k)+a2uib2*u(k4,k)+a2uib3*u(k5,k)+a2uib4*u(k6,k)
261  cofa7=a1uib1*v(ib_tmp,k)+a1uib2*v(k4,k)+a1uib3*v(k5,k)+a1uib4*v(k6,k)
262  cofa8=a2uib1*v(ib_tmp,k)+a2uib2*v(k4,k)+a2uib3*v(k5,k)+a2uib4*v(k6,k)
263 ! COFA3=A1U(IB,1)*U(IB,K)+A1U(IB,2)*U(K4,K)+A1U(IB,3)*U(K5,K)+A1U(IB,4)*U(K6,K)
264 ! COFA4=A2U(IB,1)*U(IB,K)+A2U(IB,2)*U(K4,K)+A2U(IB,3)*U(K5,K)+A2U(IB,4)*U(K6,K)
265 ! COFA7=A1U(IB,1)*V(IB,K)+A1U(IB,2)*V(K4,K)+A1U(IB,3)*V(K5,K)+A1U(IB,4)*V(K6,K)
266 ! COFA8=A2U(IB,1)*V(IB,K)+A2U(IB,2)*V(K4,K)+A2U(IB,3)*V(K5,K)+A2U(IB,4)*V(K6,K)
267 
268  uij2(i)=cofa3*xijb+cofa4*yijb
269  vij2(i)=cofa7*xijb+cofa8*yijb
270  ualfa_tmp=abs(u(ia,k)-u(ib_tmp,k))/abs(uij2(i)+epsilon(eps))
271  valfa_tmp=abs(v(ia,k)-v(ib_tmp,k))/abs(vij2(i)+epsilon(eps))
272  IF(ualfa_tmp > 1)ualfa_tmp = 1.0_sp
273  IF(valfa_tmp > 1)valfa_tmp = 1.0_sp
274  ualfa(ib_tmp)=min(ualfa(ib_tmp),ualfa_tmp)
275  valfa(ib_tmp)=min(valfa(ib_tmp),valfa_tmp)
276 
277  viscof1=art(ia)*sqrt(cofa1**2+cofa6**2+0.5_sp*(cofa2+cofa5)**2)
278  viscof2=art(ib_tmp)*sqrt(cofa3**2+cofa8**2+0.5_sp*(cofa4+cofa7)**2)
279 
280 ! VISCOF = HORCON*(FACT*0.5_SP*(VISCOF1+VISCOF2)/HPRNU + FM1)
281 ! VISCOF = HORCON*(FACT*0.5_SP*(VISCOF1+VISCOF2) + FM1)/HPRNU
282  ! David moved HPRNU and added HVC
283  viscof=(fact*0.5_sp*(viscof1*cc_hvc(ia)+viscof2*cc_hvc(ib_tmp)) + fm1*0.5_sp*(cc_hvc(ia)+cc_hvc(ib_tmp)))/hprnu
284 
285  txxij=(cofa1+cofa3)*viscof
286  tyyij=(cofa6+cofa8)*viscof
287  txyij=0.5_sp*(cofa2+cofa4+cofa5+cofa7)*viscof
288  fxx(i)=dij*(txxij*dltyc(i)-txyij*dltxc(i))
289  fyy(i)=dij*(txyij*dltyc(i)-tyyij*dltxc(i))
290 # if defined (WET_DRY)
291  END IF
292 # endif
293  END DO
294 
295  DO i=1,ne
296  ia=iec(i,1)
297  ib=iec(i,2)
298 
299 # if defined (WET_DRY)
300 # if !defined (SEMI_IMPLICIT)
301  IF(iswetct(ia)*iswetc(ia) == 1 .OR. iswetct(ib)*iswetc(ib) == 1)THEN
302 # else
303  IF(iswetct(ia) == 1 .OR. iswetct(ib) == 1)THEN
304 # endif
305 # endif
306  j1=ienode(i,1)
307  j2=ienode(i,2)
308 
309  ib_tmp = ib
310 
311 # if defined (THIN_DAM)
312  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k<=kdam1(ia))ib_tmp=e_dam_match(ia)
313 # endif
314 
315  dij= 0.5_sp*(dt(j1)*dz(j1,k)+dt(j2)*dz(j2,k))
316  uij1(i)=u(ia,k)+ualfa(ia)*uij1(i)
317  vij1(i)=v(ia,k)+valfa(ia)*vij1(i)
318  uij2(i)=u(ib_tmp,k)+ualfa(ib_tmp)*uij2(i)
319  vij2(i)=v(ib_tmp,k)+valfa(ib_tmp)*vij2(i)
320 
321 # if defined (LIMITED_1)
322  IF(uij1(i) > max(u(ia,k),u(ib_tmp,k)) .OR. uij1(i) < min(u(ia,k),u(ib_tmp,k)) .OR. &
323  uij2(i) > max(u(ia,k),u(ib_tmp,k)) .OR. uij2(i) < min(u(ia,k),u(ib_tmp,k)))THEN
324  uij1(i)=u(ia,k)
325  uij2(i)=u(ib_tmp,k)
326  END IF
327 
328  IF(vij1(i) > max(v(ia,k),v(ib_tmp,k)) .OR. vij1(i) < min(v(ia,k),v(ib_tmp,k)) .OR. &
329  vij2(i) > max(v(ia,k),v(ib_tmp,k)) .OR. vij2(i) < min(v(ia,k),v(ib_tmp,k)))THEN
330  vij1(i)=v(ia,k)
331  vij2(i)=v(ib_tmp,k)
332  END IF
333 # endif
334 
335  !!COMPUTE THE NORMAL VELOCITY ACROSS THE EDGE
336  uij=0.5_sp*(uij1(i)+uij2(i))
337  vij=0.5_sp*(vij1(i)+vij2(i))
338  un_tmp=vij*dltxc(i) - uij*dltyc(i)
339 
340  !!UPWIND THE ADVECTIVE FLUX
341  xadv=dij*un_tmp*((1.0_sp-sign(1.0_sp,un_tmp))*uij2(i)+(1.0_sp+sign(1.0_sp,un_tmp))*uij1(i))*0.5_sp
342  yadv=dij*un_tmp*((1.0_sp-sign(1.0_sp,un_tmp))*vij2(i)+(1.0_sp+sign(1.0_sp,un_tmp))*vij1(i))*0.5_sp
343 
344 
345  !!COMPUTE BOUNDARY FLUX AUGMENTERS
346 # if !defined (MEAN_FLOW)
347 
348 # if defined (THIN_DAM)
349  IF(ib==0.AND.e_dam_match(ia)/=0.AND.k <= kdam1(ia))THEN
350  isbc_tmp = 0
351  ELSE
352  isbc_tmp = isbc(i)
353  ENDIF
354 # else
355  isbc_tmp = isbc(i)
356 # endif
357 
358  tpa = float(1-isbc_tmp)*epor(ia)
359  tpb = float(1-isbc_tmp)*epor(ib_tmp)
360  !!ACCUMULATE THE FLUX
361 ! XFLUX(IA,K)=XFLUX(IA,K)+XADV*TPA+FXX*TPA
362 ! YFLUX(IA,K)=YFLUX(IA,K)+YADV*TPA+FYY*TPA
363 ! XFLUX(IB,K)=XFLUX(IB,K)-XADV*TPB-FXX*TPB
364 ! YFLUX(IB,K)=YFLUX(IB,K)-YADV*TPB-FYY*TPB
365 
366  xflux(ia,k)=xflux(ia,k)+xadv*tpa+(fxx(i)+3.0_sp*fxx(i)*float(isbc_tmp))*epor(ia)
367  yflux(ia,k)=yflux(ia,k)+yadv*tpa+(fyy(i)+3.0_sp*fyy(i)*float(isbc_tmp))*epor(ia)
368  xflux(ib,k)=xflux(ib,k)-xadv*tpb-(fxx(i)+3.0_sp*fxx(i)*float(isbc_tmp))*epor(ib)
369  yflux(ib,k)=yflux(ib,k)-yadv*tpb-(fyy(i)+3.0_sp*fyy(i)*float(isbc_tmp))*epor(ib)
370 # else
371  tpa = float(1-isbc(i))
372  tpb = float(1-isbc(i))
373  xflux(ia,k)=xflux(ia,k)+(xadv*tpa+(fxx(i)+3.0_sp*fxx(i)*float(isbc(i))))*iucp(ia)
374  yflux(ia,k)=yflux(ia,k)+(yadv*tpa+(fyy(i)+3.0_sp*fyy(i)*float(isbc(i))))*iucp(ia)
375  xflux(ib,k)=xflux(ib,k)-(xadv*tpb+(fxx(i)+3.0_sp*fxx(i)*float(isbc(i))))*iucp(ib)
376  yflux(ib,k)=yflux(ib,k)-(yadv*tpb+(fyy(i)+3.0_sp*fyy(i)*float(isbc(i))))*iucp(ib)
377 # endif
378 
379 # if defined (WET_DRY)
380  END IF
381 # endif
382  END DO
383  END DO
384 
385  DEALLOCATE(uij1,vij1,uij2,vij2,stat=error)
386  IF(error /= 0) &
387  & CALL fatal_error("Unexpected deallocation error for UIJ1,VIJ1,UIJ2 and VIJ2.")
388  DEALLOCATE(ualfa,valfa,stat=error)
389  IF(error /= 0) &
390  & CALL fatal_error("Unexpected deallocation error for UALFA,VALFA.")
391  DEALLOCATE(fxx,fyy,stat=error)
392  IF(error /= 0) &
393  & CALL fatal_error("Unexpected deallocation error for FXX,FYY.")
394 
395 # else
396 
397  DO i=1,ne
398  ia=iec(i,1)
399  ib=iec(i,2)
400 
401 # if defined (WET_DRY)
402 # if !defined (SEMI_IMPLICIT)
403  IF(iswetct(ia)*iswetc(ia) == 1 .OR. iswetct(ib)*iswetc(ib) == 1)THEN
404 # else
405  IF(iswetct(ia) == 1 .OR. iswetct(ib) == 1)THEN
406 # endif
407 # endif
408  j1=ienode(i,1)
409  j2=ienode(i,2)
410 
411  k1=nbe(ia,1)
412  k2=nbe(ia,2)
413  k3=nbe(ia,3)
414  k4=nbe(ib,1)
415  k5=nbe(ib,2)
416  k6=nbe(ib,3)
