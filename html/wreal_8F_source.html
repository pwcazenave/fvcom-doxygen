51 !------------------------------------------------------------------------------!
52 
53  wtts = wts
54

56 ! CALCULATE A NEW OMEGA VELOCITY !
57 !------------------------------------------------------------------------------!
58 
59  DO i=1,n
60 # if defined (WET_DRY)
61  IF(iswetc(i) == 1)THEN
62 # endif
63  j1=nv(i,1)
64  j2=nv(i,2)
65  j3=nv(i,3)
66  dddx=awx(i,1) * d(j1)+awx(i,2) * d(j2)+awx(i,3)*d(j3)
67  dddy=awy(i,1) * d(j1)+awy(i,2) * d(j2)+awy(i,3)*d(j3)
68  dedx=awx(i,1)*elf(j1)+awx(i,2)*elf(j2)+awx(i,3)*elf(j3)
69  dedy=awy(i,1)*elf(j1)+awy(i,2)*elf(j2)+awy(i,3)*elf(j3)
70  etf1aa=one_third*(el(nv(i,1))+el(nv(i,2))+el(nv(i,3)))
71  DO k=1,kbm1
72  ww1=0.5_sp*(w(i,k)+w(i,k+1))+u(i,k)*(zz1(i,k)*dddx+dedx)+ &
73  v(i,k)*(zz1(i,k)*dddy+dedy)
74  ww2=(zz1(i,k)+1.)*(etf1aa-et1(i))/dti
75  ww(i,k)=ww1+ww2
76  END DO
77 # if defined (WET_DRY)
78  ELSE
79  DO k=1,kbm1
80  ww(i,k)=0.0_sp
81  END DO
82  END IF
83 # endif
84  END DO
85 
86 # if defined(MULTIPROCESSOR)
87  IF(par)CALL aexchange(ec,myid,nprocs,ww)
88 # endif
89
