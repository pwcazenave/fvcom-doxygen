85  ramp=tanh(tmp/real(iramp,sp))

99 # if defined (ICE)
100  ! ggao 01042008 for ice ocean couple
101 
102  DO i=1,nt
103  IF(isicec(i)==1) THEN
104  !! (magnitude of relative ocean current)*rhow*drag*aice
105  ! DUVI = DRAGW*SQRT((U(I,1)-UICE2(I))**2+(V(I,1)-VICE2(I))**2) ! m/s
106  duvi = dragw*min(sqrt((u(i,1)-uice2(i))**2+(v(i,1)-vice2(i))**2),1.0_sp) ! m/s
107  !! ice/ocean stress
108 ! ice_ocnDX = DUVI*((U(I,1)*COSW-V(I,1)*SINW)-(uice2(i)*cosw-vice2(i)*sinw))
109 ! ice_ocnDY = DUVI*((V(I,1)*COSW+U(I,1)*SINW)-(vice2(i)*cosw+uice2(i)*sinw))
110 
111  ice_ocndx = duvi*max(min(((u(i,1)*cosw-v(i,1)*sinw)-(uice2(i)*cosw-vice2(i)*sinw)),1.0_sp),-1.0_sp)
112  ice_ocndy = duvi*max(min(((v(i,1)*cosw+u(i,1)*sinw)-(vice2(i)*cosw+uice2(i)*sinw)),1.0_sp),-1.0_sp)
113 
114 
115  wusurf2(i)=wusurf2(i)*(1.0_sp-aiu(i))-ice_ocndx*aiu(i)*1.0e-3 !*RAMP
116  wvsurf2(i)=wvsurf2(i)*(1.0_sp-aiu(i))-ice_ocndy*aiu(i)*1.0e-3 !*RAMP
117  ENDIF
118  END DO
119 # if defined (MULTIPROCESSOR)
120  IF(par)CALL aexchange(ec,myid,nprocs,wusurf2,wvsurf2)
121 # endif
122 
123 #endif
124 
125 
126 !
127 !------SAVE VALUES FROM CURRENT TIME STEP--------------------------------------!
128 !
129  elrk1 = el1
130  elrk = el
131  uark = ua
132  vark = va
133 
134 # if defined (EQUI_TIDE)
135  elrk_eqi = el_eqi
136 # endif
137 # if defined (ATMO_TIDE)
138  elrk_atmo = el_atmo
139 # endif
140 # if defined (AIR_PRESSURE)
141  elrk_air = el_air
142 # endif
143 
144 ! New Open Boundary Condition ----5
145 # if defined (MEAN_FLOW)
146  elrkt = elt
147  elrkp = elp
148  uarknt = uant !change BKI
149  varknt = vant !change BKI
150  uarkn = uan !change BKI
151  varkn = van !change BKI
152 # endif
153 
154 
155 !
156 !------BEGIN MAIN LOOP OVER EXTERNAL MODEL 4 STAGE RUNGE-KUTTA INTEGRATION-----!
157 !
158 
159  DO k=1,4
160 
161  rktime = exttime + imdte * (alpha_rk(k) - 1.0_dp)
162 
163 ! CALL PRINT_REAL_TIME(RKTIME,IPT,"RUNGE-KUTTA")
164 
165 
166 ! New Open Boundary Condition ----6
167 # if defined (MEAN_FLOW)
168  CALL bcond_tide_2d
169  ! CALL BCOND_NG_2D ! change BKI
170  CALL flux_obn2d(k)
171  CALL flux_obc2d
172 # endif
173 
174 
175 !FREE SURFACE AMPLITUDE UPDATE --> ELF
176  CALL extel_edge(k)
177 # if defined (MULTIPROCESSOR)
178  IF(par) CALL aexchange(nc,myid,nprocs,elf)
179 # endif
180 
181 
182 # if defined (EQUI_TIDE)
183  CALL elevation_equi
184  elf_eqi = elrk_eqi +alpha_rk(k)*(elf_eqi-elrk_eqi)
185 # endif
186 # if defined (ATMO_TIDE)
187  CALL elevation_atmo
188  elf_atmo = elrk_atmo +alpha_rk(k)*(elf_atmo-elrk_atmo)
189 # endif
190 
191 # if defined (AIR_PRESSURE)
192  CALL bcond_pa_air
193  elf_air = elrk_air +alpha_rk(k)*(elf_air-elrk_air)
194 # endif
195 
196 
197  ! New Open Boundary Condition ----7
198 # if defined (MEAN_FLOW)
199  IF (ntidenode > 0) THEN
200  DO i = 1, ntidenode ! need to calculate NEXT_OBC column of ELPF
201  j = i_tidenode_n(i)
202  elpf(i) = elf(j) - eltf(i)
203  END DO
204  END IF
205 !------------------------------Jianzhong------------------------------!
206 ! This code is for two OBCs along the open boundary
207 ! One is meanflow method combined transport with tidal forcing
208 ! One is only tidal focring as ordinary
209 # if defined (GCN)
210  CALL bcond_gcn(1)
211 # else
212  CALL bcond_gcy(1)
213 # endif
214 !---------------------------------------------------------------------!
215  IF(nmfcell_i>0)THEN
216  CALL extelpf_edge(k)
217  ENDIF
218 !------------------------------Jianzhong------------------------------!
219 ! IF (IOBCN > 0) THEN
220 ! DO I = 1, IOBCN
221 ! J = I_OBC_N(I)
222 ! J1= I_OBC_NODE(J)
223 ! ELF(J) = ELTF(J1) + ELPF(J1)
224 ! END DO
225 ! END IF
226 
227  IF (ibcn(2) > 0) THEN
228  DO i = 1, ibcn(2)
229  jn= obc_lst(2,i)
230  j = i_obc_n(jn)
231  j1= i_obc_node(j)
232  elf(j) = eltf(j1) + elpf(j1)
233 ! WRITE(IPT_P,*)'TEST3.2:',ELF(J)
234  END DO
235  END IF
236 
237  IF(ibcn(1)>0)THEN
238  DO i=1,ibcn(1)
239  jn = obc_lst(1,i)
240  j=i_obc_n(jn)
241  elf(j)=elrk(j)+alpha_rk(k)*(elf(j)-elrk(j))
242  END DO
243  ENDIF
244 
245 !---------------------------------------------------------------------!
246 
247 
248 # else
249 
250  ! VALUES FOR THE OPEN BOUNDARY ARE ONLY UPDATED IN THE LOCAL DOMAIN
251  ! THE HALO IS NOT SET HERE
252 # if defined (GCN)
253  CALL bcond_gcn(1)
254 # else
255  CALL bcond_gcy(1)
256 # endif
257 
258  IF(nesting_on )THEN
259  CALL set_var(exttime,el=elf)
260  END IF
261 
262  DO i=1,ibcn(1)
263  jn = obc_lst(1,i)
264  j=i_obc_n(jn)
265  elf(j)=elrk(j)+alpha_rk(k)*(elf(j)-elrk(j))
266  END DO
267 # endif
268 
269 
270  ! DAVID ADDED THIS EXCHANGE CALL:
271  ! IT SEEMS LIKELY THAT THE HALO VALUES OF ELF WILL BE USED
272  ! BEFORE THEY ARE SET CORRECTLY OTHERWISE
273 # if defined (MULTIPROCESSOR)
274  IF(par) CALL aexchange(nc,myid,nprocs,elf)
275 # endif
276 
277 !---------------For Dam Model-----------------------
278 ! Jadon
279 
280 # if defined (THIN_DAM)
281  IF(node_dam1_n > 0)CALL ele_dam1
282  IF(node_dam2_n > 0)CALL ele_dam2
283  IF(node_dam3_n > 0)CALL ele_dam3
284  IF(par)CALL aexchange(nc,myid,nprocs,elf) ! GWC and Jadon
285 # endif
286 
287  CALL n2e2d(elf,elf1)
288 
289  IF(wetting_drying_on)CALL wet_judge
290 
291  CALL flux_obn(k)
292 
293  !CALCULATE ADVECTIVE, DIFFUSIVE, AND BAROCLINIC MODES --> UAF ,VAF
294 # if defined (GCN)
295  CALL advave_edge_gcn(advua,advva) !Compute Ext Mode Adv/Diff
296 # else
297  CALL advave_edge_gcy(advua,advva) !Compute Ext Mode Adv/Diff
298 # endif
299 
300 
301  CALL extuv_edge(k)
302 
303 # if defined (THIN_DAM)
304  CALL get_kdam !Jadon
305  CALL get_kdam_internal
306 # endif
307 
308 # if defined (GCN)
309  CALL bcond_gcn(2)
310 # else
311  CALL bcond_gcy(2)
312 # endif
313 
314 # if defined (PLBC)
315  CALL replace_vel_2d
316 # endif
317 
318  IF(nesting_on )THEN
319  CALL set_var(exttime,ua=uaf)
320  CALL set_var(exttime,va=vaf)
321  END IF
322 
323 # if defined (MULTIPROCESSOR)
324  IF(par)CALL node_match(1,nbn,bn_mlt,bn_loc,bnc,mt,1,myid,nprocs,elf)
325 # endif
326 
327 
328 # if defined (PLBC)
329  CALL replace_ele
330 # endif
331 
332  !UPDATE WATER SURFACE ELEVATION
333  CALL assign_elm1_to_elm2
334 
335  el = elf
336  el1 = elf1
337 
338 # if defined (EQUI_TIDE)
339  el_eqi = elf_eqi
340 # endif
341 
342 # if defined (ATMO_TIDE)
343  el_atmo = elf_atmo
344 # endif
345 # if defined (AIR_PRESSURE)
346  el_air = elf_air
347 # endif
348 
349  !!INTERPOLATE DEPTH FROM NODE-BASED TO ELEMENT-BASED VALUES
350  CALL n2e2d(el,el1)
351 
352  !UPDATE DEPTH AND VERTICALLY AVERAGED VELOCITY FIELD
353  d = h + el
354  d1 = h1 + el1
355  ua = uaf
356  va = vaf
357  dtfa = d
358 
359 
360  ! New Open Boundary Condition ----8
361 # if defined (MEAN_FLOW)
362  elt = eltf
363  uat = uatf
364  vat = vatf
365  ! The part below is equivalent to do both NODE MATCH and EXCHANGE for ELPF
366 # if defined (MULTIPROCESSOR)
367  IF (ntidenode > 0) THEN
368  DO i = 1, ntidenode
369  j = i_tidenode_n(i)
370  elpf(i) = elf(j) - eltf(i)
371  END DO
372  END IF
373 # endif
374  elp = elpf
375  IF(k == 4 .and. nmfcell > 0)THEN
376  DO i = 1, nmfcell
377  j = i_mfcell_n(i)
378  obc2d_x_tide(i) = obc2d_x_tide(i) + uant(i) * d1(j)
379  obc2d_y_tide(i) = obc2d_y_tide(i) + vant(i) * d1(j)
380  ENDDO
381  END IF
382  CALL bcond_bki_2d(k) ! change BKI
383 # endif
384 
385 
386  !!ENSURE ALL CELLS ARE WET IN NO FLOOD/DRY CASE
387 # if !defined (WET_DRY)
388  CALL depth_check
389 # endif
390 
391  !EXCHANGE ELEMENT-BASED VALUES ACROSS THE INTERFACE
392 # if defined (MULTIPROCESSOR)
393  IF(par)CALL aexchange(ec,myid,nprocs,ua,va,d1)
394 !!# if defined (WET_DRY)
395 !! IF(PAR .AND. K==3)CALL AEXCHANGE(EC,MYID,NPROCS,UAS,VAS)
396 !!# endif
397 # endif
398 
399 # if !defined (TWO_D_MODEL)
400  !SAVE VALUES FOR 3D MOMENTUM CORRECTION AND UPDATE
401  IF(k == 3)THEN
402  uard = uard + ua*d1
403  vard = vard + va*d1
404  egf = egf + el/isplit
405 
406 # if defined (EQUI_TIDE)
407  egf_eqi = egf_eqi + el_eqi/isplit
408 # endif
409 
410 # if defined (ATMO_TIDE)
411  egf_atmo = egf_atmo + el_atmo/isplit
412 # endif
413 # if defined (AIR_PRESSURE)
414  egf_air = egf_air + el_air/isplit
415 # endif
416 
417 !!# if defined (WET_DRY)
418 !! UARDS = UARDS + UAS*D1
419 !! VARDS = VARDS + VAS*D1
420 !!# endif
421  END IF
422 
423  !CALCULATE VALUES USED FOR SALINITY/TEMP BOUNDARY CONDITIONS
424  IF(k == 4.AND.iobcn > 0) THEN
425  DO i=1,iobcn
426  j=i_obc_n(i)
427  tmp=-(elf(j)-elrk(j))*art1(j)/dte-xflux_obcn(i)
428  uard_obcn(i)=uard_obcn(i)+tmp/float(isplit)
429  END DO
430  END IF
431 # endif
432 !end !defined (TWO_D_MODEL)
433 
434 
435  !UPDATE WET/DRY FACTORS
436  IF(wetting_drying_on)CALL wd_update(1)
437 
438  END DO !! END RUNGE-KUTTA LOOP
439 
440 
441  if(dbg_set(dbg_sbr)) write(ipt,*) "End: external_step"
442 
443 END SUBROUTINE external_step
