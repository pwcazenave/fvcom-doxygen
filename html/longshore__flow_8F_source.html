29 subroutine longshore_flow
30 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
31 ! SETUP THE METRICS FOR LONGSHORE WIND DRIVEN FLOW
32 ! CALCULATE BOUNDARY ANGLE
33 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
34  USE mod_utils
35  USE all_vars
36  USE mod_par
37  IMPLICIT NONE
38  INTEGER :: I,J,K,nxt, prv, idx, ndx
39  real(sp) :: dx, dy
40 
41  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "START longshore_flow"
42 
43  ! MAKE SURE THERE ARE ATLEAST TWO NODES IN THE LIST
44  IF (nobclsf == 0 .and. par) THEN
45  ! DO NOTHING AND RETURN
46  RETURN
47  ELSE IF (nobclsf < 2 .and. serial) THEN
48  CALL fatal_error&
49  &('There are less than two long shore flow nodes',&
50  &'You must change the _lsf.dat file to contain atleast two nodes!.')
51  END IF
52 
53  !MAKE SURE THE FIRST NODE IN THIS LIST IS NEXT TO A SOLID BOUNDARY
54  ! OR A HALO NODE (IF PAR)
55  idx = ibclsf(1)
56  i = 0
57  DO
58  i = i + 1
59  if(i > ntsn(idx)) CALL fatal_error&
60  &("THE FIRST NODE IN THE LSF LIST MUST BE NEXT TO THE SOLID BOUNDARY",&
61  &"OR (IN THE PARALLEL CASE) NEXT TO THE HALO?")
62 
63  ndx = nbsn(idx,i)
64 
65  if(isonb(ndx) == 1 .or. (ndx > m .and. ndx <= mt)) THEN
66  exit
67  end if
68  END DO
69 
70 
71  ! MAKE SURE ALL THE NODES IN THE LIST ARE ON THE OPEN BOUNDARY
72  DO i = 1, nobclsf
73  j = ibclsf(i)
74  if (isonb(j) /= 2) THEN
75 
76  write(ipt,*) "LONG SHORE FLOW ERROR ON GLOBAL NODE NUMBER: ", ngid_x(j)
77  CALL fatal_error&
78  &("A LONG SHORE FLOW BOUNDARY NODE IS NOT ON THE OPEN BOUNDARY:", &
79  &"CHECK YOUR BOUNDARY FILES!")
80  END if
81 
82  END DO
83 
84  ! MAKE A LIST OF 'NEXT NODES' - used to determine angles and
85  ! gradients allong the lsf boundary
86 
87  ALLOCATE(nbclsf(nobclsf))
88  DO idx = 1, nobclsf-1
89  prv= idx
90  nxt= idx+1
91 
92  if(ibclsf(nxt) == ibclsf(prv)) CALL fatal_error&
93  &("Two long shore flow nodes in the file list are the same!")
94 
95 
96  i = 0
97  DO
98  i = i+ 1
99  if(i > ntsn(ibclsf(prv))) CALL fatal_error&
100  &("Two long shore flow nodes are not next to eachother!")
101 
102  if( ibclsf(nxt) == nbsn(ibclsf(prv),i) ) exit
103  END DO
104 
105  nbclsf(idx)=ibclsf(nxt)
106 
107  END DO
108 
109 
110  ! NOW FIND THE NEXT NODE AFTER THE LAST IN THE LIST - THERE ARE ALWAYS TWO!
111 
112  idx = ibclsf(nobclsf)
113  IF (nobclsf >1) THEN
114  prv = ibclsf(nobclsf - 1)
115  ELSE
116  ! THIS CAN ONLY HAPPEN IN A PARALLEL CASE WHRE THERE IS ONLY ONE
117  ! LOCAL NODE IN THE LSF BOUNDARY DUE TO A NASTY DECOMPOSITION
118 
119  ! SET PRV TO THE NEIGHBORING HALO NODE!
120  i = 0
121  DO
122  i = i + 1
123  if(i > ntsn(idx)) CALL fatal_error&
124  &("CAN'T FIND ANOTHER OPEN BOUNDARY NODE FOR LONG SHORE FLOW?",&
125  &"IF THERE IS ONLY ONE NODE IN THE DOMAIN IT MUST BE NEXT TO THE HALO!")
126 
127  ndx = nbsn(idx,i)
128 
129  if(isonb(ndx) == 2 .and. ndx > m ) THEN
130  prv = ndx
131  exit
132  end if
133  END DO
134  END IF
135 
136  i = 0
137  DO
138  i = i + 1
139  if(i > ntsn(idx)) CALL fatal_error&
140  &("CAN'T FIND ANOTHER OPEN BOUNDARY NODE FOR LONG SHORE FLOW?",&
141  &"The list can't contain the entire open boundary!")
142 
143  ndx = nbsn(idx,i)
144 
145  if(isonb(ndx) == 2 .and. ndx /= prv .and. ndx /= idx ) THEN
146  nbclsf(nobclsf)= ndx
147  exit
148  end if
149  END DO
150 
151  Allocate(wdf_ang(nobclsf)); wdf_ang = 0.0_sp
152  Allocate(wdf_dist(nobclsf)); wdf_dist = 0.0_sp
153 
154  DO i = 1,nobclsf
155  dx = vx(nbclsf(i)) - vx(ibclsf(i))
156  dy = vy(nbclsf(i)) - vy(ibclsf(i))
157 
158  wdf_dist(i)= sqrt(dx**2 + dy**2)
159  wdf_ang(i) = atan2(dy,dx)
160  END DO
161 
162 
163  write(ipt,*) "! LONG SHORE FLOW BOUNDARY CONDITION METRICS"
164  WRITE(ipt,*) "! NodeList, NextNode, Angle, Distance"
165  DO i = 1, nobclsf
166 
167  write(ipt,*)"! ",ngid_x(ibclsf(i)), ngid_x(nbclsf(i)), wdf_ang(i),wdf_dist(i)
168  if (nbclsf(i) > m .and. dbg_set(dbg_log)) write(ipt,*)&
169  & "! The next node value shown above is in the halo and does not appear co&
170  &rrectly in ngid: Don't worry about it ;-)"
171 
172  END DO
173 
174  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "END longshore_flow"
175 
176 end subroutine longshore_flow
