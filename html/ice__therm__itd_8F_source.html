77 !

102  ! Save the ice area passed to the coupler.
103  ! This is needed to make the history fields consistent with
104  ! the coupler fields.
105  !-----------------------------------------------------------------
106  aice_init = aice
107 
108  !-----------------------------------------------------------------
109  ! Initialize ocean fluxes sent to the coupler.

117  do j=jlo,jhi

121  enddo
122  enddo

126  !-----------------------------------------------------------------
127 
128 ! call ice_timer_start(5) ! thermodynamics
129  do ni = 1, ncat
130  do j=jlo,jhi
131  do i=ilo,ihi
132  if (aicen(i,j,ni) > puny) then
133  hicen(i,j,ni) = vicen(i,j,ni) / aicen(i,j,ni)
134  else
135  hicen(i,j,ni) = c0i
136  hicen_old(i,j,ni) = c0i
137  endif
138  enddo ! i
139  enddo ! j
140  enddo ! n
141 ! call ice_timer_stop(5) ! thermodynamics
142 
143  !-----------------------------------------------------------------
144  ! Given thermodynamic growth rates, transport ice between
145  ! thickness categories.
146  !-----------------------------------------------------------------
147 
148 ! call ice_timer_start(7) ! category conversions (transport in h)
149  if (kitd == 1) call linear_itd (hicen_old, hicen)
150 ! call ice_timer_stop(7) ! category conversions
151 
152  !-----------------------------------------------------------------
153  ! Add frazil ice growing in leads.
154  !-----------------------------------------------------------------
155 
156 ! call ice_timer_start(5) ! thermodynamics
157  call add_new_ice
158 
159  !-----------------------------------------------------------------
160  ! Melt ice laterally.
161  !-----------------------------------------------------------------
162  call lateral_melt (rside)
163 
164  !-----------------------------------------------------------------
165  ! Convert snow below freeboard to ice.
166  !-----------------------------------------------------------------
167  call freeboard
168 ! call ice_timer_stop(5) ! thermodynamics
169

171  ! Make sure ice in each category is within its thickness bounds.
172  ! NOTE: The rebin subroutine is needed only in the rare cases
173  ! when the linear_itd subroutine cannot transfer ice
174  ! correctly (e.g., very fast ice growth).
175  !-----------------------------------------------------------------
176 
177 ! call ice_timer_start(7) ! category conversions
178  if (ncat==1) then
179  call reduce_area (hicen_old(:,:,1), hicen(:,:,1))

181  call rebin
182  endif ! ncat = 1
183 ! call ice_timer_stop(7) ! category conversions
184

187  !-----------------------------------------------------------------
188  call zap_small_areas

192  !-----------------------------------------------------------------
193  call aggregate

196  ! Compute thermodynamic area and volume tendencies.
197  !-----------------------------------------------------------------
198 
199  do j=jlo,jhi
200  do i=ilo,ihi
201 ! daidtt(i,j) = (aice(i,j) - daidtt(i,j)) / dt
202 ! dvidtt(i,j) = (vice(i,j) - dvidtt(i,j)) / dt
203  daidtt(i,j) = (aice(i,j) - daidtt(i,j)) / dtice
204  dvidtt(i,j) = (vice(i,j) - dvidtt(i,j)) / dtice

207  dvidtd(i,j) = vice(i,j) ! temporarily used for initial volume
208  enddo ! i
209  enddo ! j

218 ! !ROUTINE: add_new_ice - add frazil ice to ice thickness distribution
219 !
220 ! !DESCRIPTION:
221 !
222 ! Given the volume of new ice grown in open water, compute its area
223 ! and thickness and add it to the appropriate category or categories.
224 !
225 ! NOTE: Usually all the new ice is added to category 1. An exception is
226 ! made if there is no open water or if the new ice is too thick
227 ! for category 1, in which case ice is distributed evenly over the
228 ! entire cell. Subroutine rebin should be called in case the ice
229 ! thickness lies outside category bounds after new ice formation.
230 !
231 ! !REVISION HISTORY:
232 !
233 ! authors William H. Lipscomb, LANL
234 ! Elizabeth C. Hunke, LANL
235 !
236 ! !INTERFACE:
237 !
238  subroutine add_new_ice
239 !
240 ! !USES:
241 !
242 ! !INPUT/OUTPUT PARAMETERS:
243 !
244 !EOP
245 !
246  integer (kind=int_kind) :: &
247  & i, j & ! horizontal indices
248  &, ni & ! ice category index
249  &, k ! ice layer index
250 
251  real (kind=dbl_kind), dimension(ilo:ihi,jlo:jhi) :: &
252  & ai0new & ! area of new ice added to cat 1
253  &, vi0new & ! volume of new ice added to cat 1
254  &, hsurp & ! thickness of new ice added to each cat
255  &, vlyr ! ice layer volume
256 
257  real (kind=dbl_kind), dimension (imt_local,jmt_local) :: &
258  & vice_init, vice_final ! ice volume summed over categories
259 
260  real (kind=dbl_kind) :: &
261  & fnew & ! heat flx to open water for new ice (W/m^2)
262  &, hi0new & ! thickness of new ice
263  &, hi0max & ! max allowed thickness of new ice
264  &, qi0(nilyr) & ! frazil ice enthalpy
265  &, qi0av & ! mean value of qi0 for new ice (J kg-1)
266  &, vsurp & ! volume of new ice added to each cat
267  &, area1 & ! starting fractional area of existing ice
268  &, rnilyr & ! real(nilyr)
269  &, dfresh & ! change in fresh
270  &, dfsalt & ! change in fsalt
271  &, vi_frzmlt & ! ice vol formed by frzmlt acting alone
272  &, vi_diff ! vi0new - vi_frzmlt
273 
274  real (kind=dbl_kind), parameter :: &
275  & hfrazilmin = 0.05_dbl_kind ! min thickness of new frazil ice (m)
276 
277  integer (kind=int_kind) :: &
278  & icells, jcells, kcells &! grid cell counters
279  &, ij ! combined i/j horizontal index
280 
281  integer (kind=int_kind), &
282  & dimension (1:(ihi-ilo+1)*(jhi-jlo+1)) :: &
283  & indxi, indxj & ! compressed i/j indices
284  &, indxi2, indxj2 &
285  &, indxi3, indxj3
286 
287  character (len=char_len) :: &
288  & fieldid ! field identifier
289 
290  if (ncat > 1) then
291  hi0max = hin_max(1)*0.9_dbl_kind ! not too close to boundary
292  else
293  hi0max = 1.e8_dbl_kind ! big number
294  endif
295 
296  ! initial ice volume in each grid cell
297  call column_sum (ncat, vicen, vice_init)
298 
299  !-----------------------------------------------------------------
300  ! Compute average enthalpy of new ice.
301  !
302  ! POP assumes new ice is fresh. Otherwise, it would be better
303  ! to do something like this:
304  ! qi0(i,j,k) = -rhoi * (cp_ice*(Tmlt(k)-Tf(i,j))
305  ! + Lfresh*(1.-Tmlt(k)/Tf(i,j)) - cp_ocn*Tmlt(k))
306  !-----------------------------------------------------------------
307 
308  rnilyr = real(nilyr,kind=dbl_kind)
309  qi0av = c0i
310  do k = 1, nilyr
311  qi0(k) = -rhoi*lfresh ! note sign convention, qi < 0
312  qi0av = qi0av + qi0(k)
313  enddo
314  qi0av = qi0av/rnilyr
315 
316  !-----------------------------------------------------------------
317  ! Identify ice/ocean grid points.
318  !-----------------------------------------------------------------
319  icells = 0
320  do j = jlo, jhi
321  do i = ilo, ihi
322  if (tmask(i,j)) then
323  icells = icells + 1
324  indxi(icells) = i
325  indxj(icells) = j
326  endif
327  enddo ! i
328  enddo ! j
329 
330  !-----------------------------------------------------------------
331  ! Compute the volume, area, and thickness of new ice.
332  !-----------------------------------------------------------------
333 
334 !DIR$ CONCURRENT !Cray
335 !cdir nodep !NEC
336 !ocl novrec !Fujitsu
337  do ij = 1, icells
338  i = indxi(ij)
339  j = indxj(ij)
340 
341  fnew = max(frzmlt(i,j), c0i) ! fnew > 0 iff frzmlt > 0
342  vi0new(i,j) = -fnew*dtice / qi0av ! note sign convention, qi < 0
343 ! vi0new(i,j) = -fnew*dt / qi0av ! note sign convention, qi < 0
344 
345  ! increment ice volume
346  vice_init(i,j) = vice_init(i,j) + vi0new(i,j)
347 
348  ! history diagnostics
349  frazil(i,j) = vi0new(i,j)
350  if (frazil(i,j) > puny .and. frz_onset(i,j) < puny) &
351  & frz_onset(i,j) = yday
352 
353  !-----------------------------------------------------------------
354  ! Update fresh water and salt fluxes.
355  !
356  ! NOTE: POP assumes fresh water and salt flux due to frzmlt > 0
357  ! is NOT included in fluxes fresh and fsalt.
358  !-----------------------------------------------------------------
359 
360 !!! dfresh = -rhoi*vi0new(i,j)/dt ! if POP had not already adjusted
361  ! itself based on frzmlt
362 !!! dfsalt = ice_ref_salinity*p001*dfresh
363 
364 !!! fresh(i,j) = fresh(i,j) + dfresh
365 !!! fresh_hist(i,j) = fresh_hist(i,j) + dfresh
366 !!! fsalt(i,j) = fsalt(i,j) + dfsalt
367 !!! fsalt_hist(i,j) = fsalt_hist(i,j) + dfsalt
368 
369 !! There is no adjust in FVCOM when it is coupled with CICE model
370 !! be careful to adjust the fresh water balance
371  !dfresh = -rhoi*vi0new(i,j)/dt ! if POP had not already adjusted
372  dfresh = -rhoi*vi0new(i,j)/dtice ! if POP had not already adjusted
373  ! itself based on frzmlt
374  dfsalt = ice_ref_salinity*p001*dfresh
375 
376  fresh(i,j) = fresh(i,j) + dfresh
377  fresh_hist(i,j) = fresh_hist(i,j) + dfresh
378  fsalt(i,j) = fsalt(i,j) + dfsalt
379  fsalt_hist(i,j) = fsalt_hist(i,j) + dfsalt
380 
381 
382 
383 
384 
385  !-----------------------------------------------------------------
386  ! Decide how to distribute the new ice.
387  !-----------------------------------------------------------------
388 
389  hsurp(i,j) = c0i
390  ai0new(i,j) = c0i
391 
392  if (vi0new(i,j) > c0i) then
393 
394  ! new ice area and thickness
395  ! hin_max(0) < new ice thickness < hin_max(1)
396  if (aice0(i,j) > puny) then
397  hi0new = max(vi0new(i,j)/aice0(i,j), hfrazilmin)
398  if (hi0new > hi0max .and. aice0(i,j)+puny < c1i) then
399  ! distribute excess volume over all categories (below)
400  hi0new = hi0max
401  ai0new(i,j) = aice0(i,j)
402  vsurp = vi0new(i,j) - ai0new(i,j)*hi0new
403  hsurp(i,j) = vsurp / aice(i,j)
404  vi0new(i,j) = ai0new(i,j)*hi0new
405  else
406  ! put ice in a single category, with hsurp = 0
407  ai0new(i,j) = vi0new(i,j)/hi0new
408  endif
409  else ! aice0 < puny
410  hsurp(i,j) = vi0new(i,j)/aice(i,j) ! new thickness in each cat
411  vi0new(i,j) = c0i
412  endif ! aice0 > puny
413  endif ! vi0new > puny
414 
415  enddo ! ij
416 
417  !-----------------------------------------------------------------
418  ! Identify grid cells receiving new ice.
419  !-----------------------------------------------------------------
420  jcells = 0
421  kcells = 0
422 
423  do ij = 1, icells
424  i = indxi(ij)
425  j = indxj(ij)
426 
427  if (vi0new(i,j) > c0i) then ! add ice to category 1
428  jcells = jcells + 1
429  indxi2(jcells) = i
430  indxj2(jcells) = j
431  endif
432 
433  if (hsurp(i,j) > c0i) then ! add ice to all categories
434  kcells = kcells + 1
435  indxi3(kcells) = i
436  indxj3(kcells) = j
437  endif
438 
439  enddo
440 
441  !-----------------------------------------------------------------
442  ! Distribute excess ice volume among ice categories by increasing
443  ! ice thickness, leaving ice area unchanged.
444  !-----------------------------------------------------------------
445 
446  do ni = 1, ncat
447 
448 !DIR$ CONCURRENT !Cray
449 !cdir nodep !NEC
450 !ocl novrec !Fujitsu
451  do ij = 1, kcells
452  i = indxi3(ij)
453  j = indxj3(ij)
454 
455  vicen(i,j,ni) = vicen(i,j,ni) + aicen(i,j,ni)*hsurp(i,j)
456  vlyr(i,j) = hsurp(i,j)/rnilyr * aicen(i,j,ni)
457  enddo ! ij
458 
459  do k=1,nilyr
460 !DIR$ CONCURRENT !Cray
461 !cdir nodep !NEC
462 !ocl novrec !Fujitsu
463  do ij = 1, kcells
464  i = indxi3(ij)
465  j = indxj3(ij)
466 
467  eicen(i,j,ilyr1(ni)+k-1) = &
468  & eicen(i,j,ilyr1(ni)+k-1) + qi0(k)*vlyr(i,j)
469  enddo ! ij
470  enddo ! k
471 
472  enddo ! n
473 
474  !-----------------------------------------------------------------
475  ! Combine new ice grown in open water with category 1 ice.
476  ! NOTE: vsnon and esnon are unchanged.
477  !-----------------------------------------------------------------
478 !DIR$ CONCURRENT !Cray
479 !cdir nodep !NEC
480 !ocl novrec !Fujitsu
481  do ij = 1, jcells
482  i = indxi2(ij)
483  j = indxj2(ij)
484 
485  area1 = aicen(i,j,1) ! save
486  aicen(i,j,1) = aicen(i,j,1) + ai0new(i,j)
487  aice0(i,j) = aice0(i,j) - ai0new(i,j)
488  vicen(i,j,1) = vicen(i,j,1) + vi0new(i,j)
489  tsfcn(i,j,1) = (tf(i,j)*ai0new(i,j) + tsfcn(i,j,1)*area1) &
490  & / aicen(i,j,1)
491  tsfcn(i,j,1) = min(tsfcn(i,j,1), c0i)
492  vlyr(i,j) = vi0new(i,j)/rnilyr
493  enddo ! ij
494 
495  do k = 1, nilyr
496 !DIR$ CONCURRENT !Cray
497 !cdir nodep !NEC
498 !ocl novrec !Fujitsu
499  do ij = 1, jcells
500  i = indxi2(ij)
501  j = indxj2(ij)
502  eicen(i,j,k) = eicen(i,j,k) + qi0(k)*vlyr(i,j)
503  enddo
504  enddo
505 
506  call column_sum (ncat, vicen, vice_final)
507  fieldid = 'vice, add_new_ice'
508  call column_conservation_check(vice_init, vice_final, &
509  & puny, fieldid)
510 
511  end subroutine add_new_ice

515 !
516 ! !ROUTINE: lateral_melt - melt ice laterally
517 !
518 ! !DESCRIPTION:
519 !
520 ! Given the fraction of ice melting laterally in each grid cell
521 ! (computed in subroutine frzmlt\_bottom\_lateral), melt ice.
522 !
523 ! !REVISION HISTORY:
524 !
525 ! author: C. M. Bitz, UW
526 ! modified by: Elizabeth C. Hunke, LANL
527 ! William H. Lipscomb, LANL
528 !
529 ! !INTERFACE:
530 !
531  subroutine lateral_melt (rside)
532 !
533 ! !USES:
534 !
535 ! !INPUT/OUTPUT PARAMETERS:
536 !
537  real (kind=dbl_kind), dimension(ilo:ihi,jlo:jhi), &
538  & intent(in) :: &
539  & rside ! fraction of ice that melts laterally
540 !
541 !EOP
542 !
543  integer (kind=int_kind) :: &
544  & i, j & ! horizontal indices
545  &, ni & ! thickness category index
546  &, k &! layer index
547  &, ij &! horizontal index, combines i and j loops
548  &, icells ! number of cells with aice > puny
549 
550  integer (kind=int_kind), dimension(1:(ihi-ilo+1)*(jhi-jlo+1)) :: &
551  & indxi, indxj ! compressed indices for cells with aice > puny
552 
553  real (kind=dbl_kind) :: &
554  & dfhnet & ! change in fhnet
555  &, dfresh & ! change in fresh
556  &, dfsalt ! change in fsalt
557 
558  do ni = 1, ncat
559 
560  !-----------------------------------------------------------------
561  ! Identify grid cells with lateral melting.
562  !-----------------------------------------------------------------
563 
564  icells = 0
565  do j = jlo, jhi
566  do i = ilo, ihi
567  if (rside(i,j) > c0i) then
568  icells = icells + 1
569  indxi(icells) = i
570  indxj(icells) = j
571  endif
572  enddo ! i
573  enddo ! j
574 
575  !-----------------------------------------------------------------
576  ! Melt the ice and increment fluxes.
577  !-----------------------------------------------------------------
578 
579 !DIR$ CONCURRENT !Cray
580 !cdir nodep !NEC
581 !ocl novrec !Fujitsu
582  do ij = 1, icells
583  i = indxi(ij)
584  j = indxj(ij)
585 
586  ! fluxes to coupler (except heat flux for ice melt)
587  ! dfhnet < 0, dfresh > 0, dfsalt > 0
588 
589  dfhnet = esnon(i,j,ni)*rside(i,j) / dtice
590 ! dfhnet = esnon(i,j,n)*rside(i,j) / dt
591  dfresh = (rhos*vsnon(i,j,ni) + rhoi*vicen(i,j,ni)) &
592 ! & * rside(i,j) / dt
593  & * rside(i,j) / dtice
594  dfsalt = rhoi*vicen(i,j,ni)*ice_ref_salinity*p001 &
595  & * rside(i,j) / dtice
596 ! & * rside(i,j) / dt
597 
598  fhnet(i,j) = fhnet(i,j) + dfhnet
599  fhnet_hist(i,j) = fhnet_hist(i,j) + dfhnet
600 
601  fresh(i,j) = fresh(i,j) + dfresh
602  fresh_hist(i,j) = fresh_hist(i,j) + dfresh
603 
604  fsalt(i,j) = fsalt(i,j) + dfsalt
605  fsalt_hist(i,j) = fsalt_hist(i,j) + dfsalt
606 
607  ! history diagnostics
608  meltl(i,j) = meltl(i,j) + vicen(i,j,ni)*rside(i,j)
609 
610  ! state variables (except ice energy)
611  aicen(i,j,ni) = aicen(i,j,ni) * (c1i - rside(i,j))
612  vicen(i,j,ni) = vicen(i,j,ni) * (c1i - rside(i,j))
613  vsnon(i,j,ni) = vsnon(i,j,ni) * (c1i - rside(i,j))
614  esnon(i,j,ni) = esnon(i,j,ni) * (c1i - rside(i,j))
615 
616  enddo ! ij
617 
618  do k = 1, nilyr
619 !DIR$ CONCURRENT !Cray
620 !cdir nodep !NEC
621 !ocl novrec !Fujitsu
622  do ij = 1, icells
623  i = indxi(ij)
624  j = indxj(ij)
625 
626  ! heat flux to coupler for ice melt (dfhnet < 0)
627 
628 ! dfhnet = eicen(i,j,ilyr1(ni)+k-1)*rside(i,j) / dt
629  dfhnet = eicen(i,j,ilyr1(ni)+k-1)*rside(i,j) / dtice
630  fhnet(i,j) = fhnet(i,j) + dfhnet
631  fhnet_hist(i,j) = fhnet_hist(i,j) + dfhnet
632 
633  ! ice energy
634  eicen(i,j,ilyr1(ni)+k-1) = eicen(i,j,ilyr1(ni)+k-1) &
635  & * (c1i - rside(i,j))
636  enddo ! ij
637  enddo ! k
638 
639  enddo ! n
640 
641  end subroutine lateral_melt

645 ! !ROUTINE: freeboard - snow-ice conversion
646 !
647 ! !DESCRIPTION:
648 !
649 ! If there is enough snow to lower the ice/snow interface below
650 ! sea level, convert enough snow to ice to bring the interface back
651 ! to sea level.
652 !
653 ! NOTE: Subroutine rebin should be called after freeboard to make sure
654 ! ice thicknesses are within category bounds.
655 !
656 ! !REVISION HISTORY:
657 !
658 ! authors William H. Lipscomb, LANL
659 ! Elizabeth C. Hunke, LANL
660 !
661 ! !INTERFACE:
662 !
663  subroutine freeboard
664 !
665 ! !USES:
666 !
667 ! !INPUT/OUTPUT PARAMETERS:
668 !
669 !EOP
670 !
671  integer (kind=int_kind) :: i, j, ni, k
672 
673  real (kind=dbl_kind) :: &
674  & hi & ! ice thickness (m)
675  &, hs & ! snow thickness (m)
676  &, dhi & ! change in ice thickness (m)
677  &, dhs & ! change in snow thickness (m)
678  &, dz & ! distance freeboard below SL (m)
679  &, fs ! salt flux due to snow-ice conversion
680 
681  real (kind=dbl_kind), dimension(ilo:ihi,jlo:jhi) :: &
682  & de ! energy transferred to each ice layer(J/m^2)
683 
684  integer (kind=int_kind) :: &
685  & icells & ! number of cells with ice
686  &, jcells & ! number of cells with freeboard adjustment
687  &, ij ! combined i/j horizontal index
688 
689  integer (kind=int_kind), &
690  & dimension (1:(ihi-ilo+1)*(jhi-jlo+1)) :: &
691  & indxi & ! compressed i/j indices
692  &, indxj &
693  &, indxi2 &
694  &, indxj2
695 
696  do ni = 1, ncat
697 
698  !-----------------------------------------------------------------
699  ! Identify grid cells with ice.
700  !-----------------------------------------------------------------
701  icells = 0
702  do j = jlo, jhi
703  do i = ilo, ihi
704  if (aicen(i,j,ni) > puny) then
705  icells = icells + 1
706  indxi(icells) = i
707  indxj(icells) = j
708  endif
709  enddo ! i
710  enddo ! j
711 
712  jcells = 0 ! freeboard adjustment counter
713 
714 !DIR$ CONCURRENT !Cray
715 !cdir nodep !NEC
716 !ocl novrec !Fujitsu
717  do ij = 1, icells ! aicen > puny
718  i = indxi(ij)
719  j = indxj(ij)
720 
721  !-----------------------------------------------------------------
722  ! Determine whether snow lies below freeboard.
723  !-----------------------------------------------------------------
724 
725  hi = vicen(i,j,ni) / aicen(i,j,ni)
726  hs = vsnon(i,j,ni) / aicen(i,j,ni)
727 
728  dz = hs - hi*(rhow-rhoi)/rhos
729 
730  if (dz > puny .and. hs > puny) then ! snow below freeboard
731  jcells = jcells + 1
732  indxi2(jcells) = i
733  indxj2(jcells) = j
734 
735  dhs = min(dz*rhoi/rhow, hs) ! snow to remove
736  dhi = dhs*rhos/rhoi ! ice to add
737 
738  !-----------------------------------------------------------------
739  ! Compute energy transferred from snow to ice.
740  ! NOTE: It would be more realistic to transfer energy only to
741  ! the top ice layer, but it is simpler to transfer equal
742  ! energy to all layers.)
743  !-----------------------------------------------------------------
744 
745  de(i,j) = esnon(i,j,ni)*dhs/hs
746  esnon(i,j,ni) = esnon(i,j,ni) - de(i,j)
747  de(i,j) = de(i,j)/real(nilyr,kind=dbl_kind) ! energy to each ice layer
748 
749  !-----------------------------------------------------------------
750  ! Adjust snow and ice volume.
751  !-----------------------------------------------------------------
752 
753  hi = hi + dhi
754  hs = hs - dhs
755  vicen(i,j,ni) = hi * aicen(i,j,ni)
756  vsnon(i,j,ni) = hs * aicen(i,j,ni)
757 
758  !-----------------------------------------------------------------
759  ! Update history and coupler variables.
760  !-----------------------------------------------------------------
761 
762  ! history diagnostic
763  snoice(i,j) = snoice(i,j) + dhi*aicen(i,j,ni)
764 
765  ! Remove salt from the ocean.
766  ! This is not physically realistic but is needed to
767  ! conserve salt, because salt will be returned to the ocean
768  ! when the ice melts.
769 
770 ! fs = -ice_ref_salinity*p001*aicen(i,j,n)*dhi*rhoi/dt
771  fs = -ice_ref_salinity*p001*aicen(i,j,ni)*dhi*rhoi/dtice
772  fsalt(i,j) = fsalt(i,j) + fs
773  fsalt_hist(i,j) = fsalt_hist(i,j) + fs
774 
775  endif ! dz > puny and hs > puny
776  enddo ! ij
777 
778  !-----------------------------------------------------------------
779  ! Adjust ice energy.
780  !-----------------------------------------------------------------
781  do k = 1, nilyr
782 !DIR$ CONCURRENT !Cray
783 !cdir nodep !NEC
784 !ocl novrec !Fujitsu
785  do ij = 1, jcells ! just cells with freeboard adjustment
786  i = indxi2(ij)
787  j = indxj2(ij)
788  eicen(i,j,ilyr1(ni)+k-1) = &
789  & eicen(i,j,ilyr1(ni)+k-1) + de(i,j)
790  enddo ! ij
791  enddo ! k
792 
793  enddo ! n
794 
795  end subroutine freeboard
