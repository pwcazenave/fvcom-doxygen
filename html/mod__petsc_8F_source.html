228  pmsr = msr ! set petsc master (primarily for printing) to FVCOM master

240  !-----> simple partitioning does not give great load balance partition the domain by
241  ! nodes such that each local node in Petsc Partition is also local in Application partition
242  ! But, we do not have overlapping domains What were originally boundary nodes in FVCOM decomp
243  ! and had multiple partition owners are assigned to the adjacent partition of lower number
244  ! petsc partition id for each node in global application order ==> [partid(1:mgl)]
245  ! we should really consider a load balancing procedure where we first assign node that
246  ! arent boundary nodes and then assign boundary nodes to the adjacent partition which is smaller.
247  ! Ultimately, we want N_verts to be as equal as possible in each domain. For right now,
248  ! use this simple partitioning strategy
249 
250 ! allocate(partid(mgl)) ; partid = nprocs+1
251 ! do i=1,ngl
252 ! do l=1,3
253 ! partid(nvg(i,l)) = min( partid(nvg(i,l)) , el_pid(i))
254 ! end do
255 ! end do

257  !-----> more complex partitioning for better load balance partition the domain by nodes
258  ! such that each local node in Petsc Partition is also local in Application partition
259  ! we will do two sweeps. In a first sweep, any node that was not an interprocessor boundary
260  ! node will be assigned to the owner that had the node in the FVCOM partition.
261  ! then we will count nodes in each Petsc Partition then we will loop over boundary nodes
262  ! (which have multiple owners in the FVCOM partition) and decide who gets it based
263  ! on who has the lowest current node count this should result in a nearly optimal loadbalance
264 
265  !--sweep 1, determine owners, owner count, assume max # owners = 10
266  ALLOCATE(owner_count(mgl)) ; owner_count = 0
267  ALLOCATE(owners(mgl,0:10)) ; owners = 0
268  DO i =1, ngl
269  DO l =1, 3
270  !check if owner is already in list --> can use loc!
271  newown = .true.
272  DO j=1,owner_count(nvg(i,l))
273  IF(owners(nvg(i,l),j) == el_pid(i) ) newown = .false.
274  ENDDO
275  IF(newown)THEN
276  owner_count(nvg(i,l)) = owner_count(nvg(i,l)) + 1
277  IF(owner_count(nvg(i,l)) > 10) THEN
278  WRITE(pscrn,*) 'ERROR IN PETSc PARTITIONING'
279  WRITE(pscrn,*) 'INCREASE OWNER COUNT FROM 10'
280  CALL mpi_finalize(ierr)
281  stop
282  ENDIF
283  owners( nvg(i,l),owner_count(nvg(i,l)) ) = el_pid(i)
284  ENDIF
285  ENDDO
286  ENDDO
287 
288  !--now sweep and petsc partition the non-interprocessor boundary nodes
289  ALLOCATE(part_nodes(0:nprocs)) ; part_nodes = 0
290  ALLOCATE(partid(mgl)) ; partid = -1
291  DO i= 1, mgl
292  IF(owner_count(i) == 1) THEN
293  part_nodes(owners(i,1)) = part_nodes(owners(i,1)) + 1
294  partid(i) = owners(i,1)
295  ENDIF
296  ENDDO
297 
298  !--now sweep boundary nodes, assigning a node to owner with least count
299  DO i= 1, mgl
300  IF(owner_count(i) > 1) THEN
301  mindomsize = huge(i)
302  DO j= 1, owner_count(i)
303  !find which adjacent dom has min count
304  IF (part_nodes(owners(i,j)) < mindomsize) THEN
305  mindomsize = part_nodes(owners(i,j) )
306  mindom = owners(i,j)
307  ENDIF
308  ENDDO
309  !assign boundary node to that partition
310  partid(i) = mindom
311  part_nodes(mindom) = part_nodes(mindom) + 1
312  ENDIF
313  ENDDO
314 
315  !--make sure everybody got assigned
316  DO i = 1, mgl
317  IF(partid(i) == -1) THEN
318  WRITE(pscrn,*) 'ERROR IN PETSc PARTITIONING'
319  WRITE(pscrn,*) 'GLOBAL NODE: ',i,' NEVER GOT ASSIGNED A PETSc PARTITION!'
320  WRITE(pscrn,*) 'FIX IT!'
321  CALL mpi_finalize(ierr)
322  stop
323  ENDIF
324  ENDDO
325 
326  !--deallocate the data we used for 'fancy' partition
327  DEALLOCATE(owners)
328  DEALLOCATE(owner_count)
329 
330  ! lets count (or recount) nodes in each Petsc partition ==> [part_nodes(0:nprocs)]
331  ! we need to start from zero so we can access partition (myid-1) later the total will equal MGL,
332  ! unlike the sum of nodes over FVCOM partitions which is > MGL
333  IF(.NOT.ALLOCATED(part_nodes) ) ALLOCATE(part_nodes(0:nprocs))
334  part_nodes = 0
335  DO i= 1, mgl
336  part_nodes( partid(i)) = part_nodes( partid(i) ) + 1
337  ENDDO
338 
339  !count number of nodes in my local Petsc partition ==> [N_verts] halt if partition is empty
340  n_verts = part_nodes(myid)
341 
342  print*,'N_VERTS and M=',n_verts,m,mt,myid
343  IF(n_verts == 0) THEN
344  WRITE(punit,*) 'MY PETSc PARTITION DOSE NOT CONTAIN ANY UNKNOWNS'
345  CALL mpi_finalize(ierr)
346  stop
347  ENDIF
348  IF(pdebug) WRITE(punit,*) 'NODES IN PETSc PARTION: ',n_verts,' APPLICATION',m
349 
350 # if defined (NH)
351  !set number of layers and total problem size ==> [N_lay,Psize]
352  n_lay = kbm1
353  psize = n_lay*n_verts
354  psizegl = mgl*n_lay
355 # endif
356 
357  !report all sizes(master only)
358  IF(pmsr) WRITE(pscrn,*)'PARTITIONING : COMPLETE'
359  IF(pmsr) THEN
360  DO i= 1, nprocs
361  WRITE(pscrn,*)'PETSc PARTITION: ',i,' SIZE ',part_nodes(i)
362  ENDDO
363  ENDIF
364 
365  !make sure total sum of nodes in all petsc partitions is equal to mgl
366  IF(sum(part_nodes) /= mgl) THEN
367  WRITE(pscrn,*)'TOTAL SUM OF NODES IN PETSc PARTIONS DOES NOT EQUAL GLOBAL NUMBER OF NODES'
368  WRITE(pscrn,*)'EXITING!'
369  CALL mpi_finalize(ierr)
370  stop
371  ENDIF
372 
373  !determine which global nodes (in application order) are in my Petsc partition
374  !gives us a mapping (for interior nodes) of PLO to AGO
375  !allocate it to size MGL because we do not know size of petsc partition halo
376  !gives us ==> [plo_2_ago_node(1:N_verts)]
377  ALLOCATE(plo_2_ago_node(mgl)) ; plo_2_ago_node = 0
378  icnt = 0
379  DO i= 1, mgl
380  IF(partid(i) == myid) THEN
381  icnt = icnt + 1
382  plo_2_ago_node(icnt) = i
383  ENDIF
384  ENDDO
385 
386  IF(pdebug)THEN
387  WRITE(punit,*) 'PLO_2_AGO_NODE'
388  DO i= 1, n_verts
389  WRITE(punit,*) i, plo_2_ago_node(i)
390  ENDDO
391  ENDIF
392 
393  !get a mapping from petsc local nodes to fvcom local nodes ==> plo_2_alo_node(1:N_verts)
394  !we already have the global application order number of each node, we will
395  !use nlid to convert this to local application order of each node
396  !this will be used to in our plo_2_alo function which is ultimately used to
397  !load and unload FVCOM vectors (q and source term)
398  !gives us ==> [plo_2_alo_node(N_verts)]
399  ALLOCATE(plo_2_alo_node(n_verts)); plo_2_alo_node = 0
400  IF(nprocs > 1) THEN
401  DO i=1,n_verts
402  plo_2_alo_node(i) = nlid(plo_2_ago_node(i))
403  ENDDO
404  ELSE
405  plo_2_alo_node = plo_2_ago_node
406  ENDIF
407  IF(pdebug)THEN
408  WRITE(punit,*) 'PLO_2_ALO_NODE:'
409  DO i= 1, n_verts
410  WRITE(punit,*) i, plo_2_alo_node(i)
411  ENDDO
412  ENDIF
413 
414  !get a mapping from application local nodes to petsc local nodes
415  !first we map without setting petsc halos
416  !gives us ==> [alo_2_plo_node(1:m)]
417  ALLOCATE(alo_2_plo_node(mt)) ; alo_2_plo_node = 0
418  IF(nprocs > 1)THEN
419  icnt = 0
420  DO i= 1, mgl
421  IF(nlid_x(i) /= 0 .AND. partid(i) == myid) THEN
422  icnt = icnt + 1
423  alo_2_plo_node(nlid_x(i)) = icnt
424  ENDIF
425  ENDDO
426  ELSE
427  alo_2_plo_node = plo_2_alo_node
428  ENDIF
429  IF(pdebug)THEN
430  WRITE(punit,*) 'ALO_2_PLO_NODE:'
431  DO i= 1, n_verts
432  WRITE(punit,*) i, alo_2_plo_node(i)
433  ENDDO
434  ENDIF
435 
436  IF(icnt /= n_verts)THEN
437  WRITE(punit,*) 'PROBLEM IN ALO to PLO MAPPING!'
438  CALL mpi_finalize(ierr)
439  stop
440  ENDIF
441 
442  !we continue mapping ALO to PLO but now we accumulate halo nodes in PLO
443  !we also extend our plo_2_ago map to include a mapping from petsc local
444  !order to application global order in the petsc partition halos
445  !gives us ==> [n_halo]
446  !gives us ==> [alo_2_plo_node(m+1:mt)]
447  !gives us ==> [plo_2_alo_node(n_verts+1:n_verts+n_halo)]
448  n_halo = 0
449  IF(nprocs > 1)THEN
450  DO i= 1, mgl
451  IF(nlid_x(i) /= 0 .AND. partid(i) /= myid) THEN
452  icnt = icnt + 1
453  n_halo = n_halo + 1
454  alo_2_plo_node(nlid_x(i)) = icnt
455  plo_2_ago_node(icnt ) = i
456  ENDIF
457  ENDDO
458  ENDIF
459  IF(pdebug) THEN
460  WRITE(punit,*) 'ALO_2_PLO_NODE WITH HALO:'
461  DO i= 1, mt
462  WRITE(punit,*) i, alo_2_plo_node(i)
463  ENDDO
464  ENDIF
465 
466  !we now have a map from all all local application order nodes, including
467  !halos to petsc local nodes, including halos. We also have a map from all local
468  !petsc partition nodes, including the halo to application global order.
469  !what we need is a map from petsc partition nodes including halo to petsc
470  !global order, so that we can insert matrix values in local order and they will
471  !get mapped accordingly.
472  !we will construct a map of application global order to petsc global order and
473  !use that to map our palo_2_ago into what we really need, a plo_2_pgo
474  !gives us ==> [ago_2_pgo_node(1:MGL)]
475  ALLOCATE(ago_2_pgo_node(mgl)) ; ago_2_pgo_node = 0
476  ALLOCATE(cnt(nprocs)) ; cnt = 0
477  ALLOCATE(accum(nprocs) ) ; accum = 0
478  DO i= 2, nprocs
479  accum(i) = accum(i-1) + part_nodes(i-1)
480  ENDDO
481  DO i= 1, mgl
482  pid = partid(i)
483  cnt(pid) = cnt(pid) + 1
484  ago_2_pgo_node(i) = accum(pid) + cnt(pid)
485  ENDDO
486  IF(pdebug)THEN
487  WRITE(punit,*)'AGO_2_PGO_NODE:'
488  DO i= 1, mgl
489  WRITE(punit,*) i, ago_2_pgo_node(i)
490  ENDDO
491  ENDIF
492 
493  !now, use our ago_2_pgo_node to construct a plo_2_pgo_node
494  !gives us ==> [plo_2_pgo_node(1:n_verts+n_halo)]
495  ALLOCATE(plo_2_pgo_node(n_verts+n_halo)) ; plo_2_pgo_node = 0
496  DO i= 1, n_verts + n_halo
497  plo_2_pgo_node(i) = ago_2_pgo_node( plo_2_ago_node(i) )
498  ENDDO
499  IF(pdebug)THEN
500  WRITE(punit,*) 'COUNT PLO_2_PGO_NODE:'
501  DO i=1, n_verts+n_halo
502  WRITE(punit,*) i, plo_2_pgo_node(i)
503  ENDDO
504  ENDIF

506  !set a total problem size including petsc partition halo nodes
507  ! ==> [Psize_whalo]

509  psize_el_halo = n_verts + n_halo
510 # endif
511 
512 # if defined (NH)
513  psize_whalo = (n_verts + n_halo)*n_lay
514

518  ! ==> [plo_2_pgo(1:Psize_whalo)]
519  ! should be 1 to 1 for single processor case
520  ALLOCATE(plo_2_pgo( psize_whalo )) ; plo_2_pgo = 0
521  icnt = 0
522  DO i= 1, n_verts + n_halo
523  pgl_node = plo_2_pgo_node(i)
524  DO k= 1, n_lay
525  icnt = icnt + 1
526  plo_2_pgo(icnt) = k + (pgl_node-1)*n_lay
527  ENDDO
528  ENDDO
529  IF(pdebug)THEN
530  WRITE(punit,*) 'COUNT PLO_2_PGO'
531  DO i= 1, psize_whalo
532  WRITE(punit,*) i, plo_2_pgo(i)
533  ENDDO
534  ENDIF
535 # endif
536 
537 # if defined (WAVE_CURRENT_INTERACTION) && !defined (EXPLICIT)
538  psize_wave_halo = n_verts + n_halo
539 # endif

547

549  !setup petsc local to global mapping (isl2g) to all for local indexing in assembly of matrix

800  SUBROUTINE petsc_alloca
801 !============================================================================
802 ! PETSc_AllocA
803 ! preallocate matrix A
804 ! assume our nonzero pattern will not change
805 ! this will make the implementation much more efficient
806 ! malloc-ing on the fly can be very slow
807 !
808 ! currently assumes stencil is:
809 ! -node + surrounding nodes
810 ! -node above + surrounding nodes
811 ! -node below + surrounding nodes
812 !
813 ! typical number of nonzero entries: 21
814 !
815 ! dependencies: all internal to this module
816 ! except: m,kbm1
817 !============================================================================
818  USE all_vars , ONLY : m, kbm1, ntsn, nbsn, mgl
819  IMPLICIT NONE
820  INTEGER :: node,lay,petsc_pos,ncol,ncol2,neynum,ney,colcount,i
821  INTEGER :: col2(max_nz_row)
822  INTEGER,ALLOCATABLE :: onnz2(:)
823  INTEGER,ALLOCATABLE :: dnnz2(:)
824  CHARACTER(LEN=20) :: subname = 'PETSc_AllocA'
825 
826 # if defined (NH)
827  INTEGER :: col(max_nz_row)
828  INTEGER,ALLOCATABLE :: onnz(:), dnnz(:)
829 # endif
830 
831  petscint :: ierr
832 
833  IF(pdebug) WRITE(punit,*) 'STARTING: ', trim(subname)
834  !-----------------------------------------------------------------------------
835  ! assemble matrix, use local vars (requires that MatSetLocalToGlobalMapping is setup)
836  ! note: row and column are C order (starts from 0)
837  ! see page 51 of petsc manual, bottom paragraph
838  ! we will set entries row by row
839  !-----------------------------------------------------------------------------
840 # if defined (NH)
841  ALLOCATE(onnz(psize)) ; onnz = 0
842  ALLOCATE(dnnz(psize)) ; dnnz = 0
843 # endif
844  ALLOCATE(onnz2(n_verts)) ; onnz2 = 0
845  ALLOCATE(dnnz2(n_verts)) ; dnnz2 = 0
846 
847  DO node=1, m
848 
849  col2 = 0
850  ncol2 = 0
851  IF( alo_2_plo_node(node) <= n_verts) THEN
852  ncol2 = 1
853  col2(1) = alo_2_plo_node(node)
854 
855  DO neynum =1, ntsn(node)-1
856  ney = nbsn(node,neynum)
857  IF(ney == node) cycle
858  ncol2 = ncol2 + 1
859  !make sure we havent exceeded max_nz_row
860  IF(ncol2 > 10) THEN
861  WRITE(punit,*) 'WE HAVE EXCEEDED MAX_NZ_ROW==10'
862  WRITE(punit,*) 'INCREASE AND RECOMPILE'
863  CALL mpi_finalize(ierr)
864  stop
865  ENDIF
866  col2(ncol2) = alo_2_plo_node(ney)
867  ENDDO
868 
869  DO colcount =1, ncol2
870  IF(col2(colcount) <= n_verts) THEN
871  onnz2(alo_2_plo_node(node)) = onnz2(alo_2_plo_node(node)) + 1
872  ELSE
873  dnnz2(alo_2_plo_node(node)) = dnnz2(alo_2_plo_node(node)) + 1
874  ENDIF
875  ENDDO
876 
877  ENDIF
878 
879 # if defined (NH)
880  DO lay =1, kbm1
881 
882  !get node number in petsc local domain, cycle if not in domain
883  petsc_pos = alo_2_plo(node,lay)
884  IF(petsc_pos > psize) cycle
885 
886  !initialize column numbers and row values
887  col = 0
888  ncol = 1 !diagonal entry
889 
890  !set row number in PLO. If it is not in PLO interior, just cycle
891  !this means it is a boundar node not in the PLO partition and
892  !another partition will be setting that row
893  col(1) = alo_2_plo(node,lay)
894 ! IF(COL(1) > PSIZE) CYCLE
895 
896  !set discretization in current layer"
897  DO neynum =1, ntsn(node)-1
898  ney = nbsn(node,neynum)
899  IF(ney == node) cycle
900  ncol = ncol + 1
901  !make sure we havent exceeded max_nz_row
902  IF(ncol > max_nz_row) THEN
903  WRITE(punit,*) 'WE HAVE EXCEEDED MAX_NZ_ROW'
904  WRITE(punit,*) 'INCREASE AND RECOMPILE'
905  CALL mpi_finalize(ierr)
906  stop
907  ENDIF
908  col(ncol) = alo_2_plo(ney,lay)
909  ENDDO
910 
911  !set discretization in layer above
912  IF(lay > 1) THEN
913  DO neynum = 1, ntsn(node)-1
914  ney = nbsn(node,neynum)
915  IF(ney == node) cycle
916  ncol = ncol + 1
917  !make sure we havent exceeded max_nz_row
918  IF(ncol > max_nz_row) THEN
919  WRITE(punit,*) 'WE HAVE EXCEEDED MAX_NZ_ROW'
920  WRITE(punit,*) 'INCREASE AND RECOMPILE'
921  CALL mpi_finalize(ierr)
922  stop
923  ENDIF
924  col(ncol) = alo_2_plo(ney,lay-1)
925  ENDDO
926  ENDIF
927 
928  !set discretization in layer below
929  IF(lay < n_lay) THEN
930  DO neynum = 1, ntsn(node)-1
931  ney = nbsn(node,neynum)
932  IF (ney == node) cycle
933  ncol = ncol + 1
934  !make sure we havent exceeded max_nz_row
935  IF(ncol > max_nz_row) THEN
936  WRITE(punit,*) 'WE HAVE EXCEEDED MAX_NZ_ROW'
937  WRITE(punit,*) 'INCREASE AND RECOMPILE'
938  CALL mpi_finalize(ierr)
939  stop
940  ENDIF
941  col(ncol) = alo_2_plo(ney,lay+1)
942  ENDDO
943  ENDIF
944 
945  !set vertical "discretization"
946  IF(lay > 1) THEN
947  ncol = ncol + 1
948  !make sure we havent exceeded max_nz_row
949  IF(ncol > max_nz_row) THEN
950  WRITE(punit,*) 'WE HAVE EXCEEDED MAX_NZ_ROW'
951  WRITE(punit,*) 'INCREASE AND RECOMPILE'
952  CALL mpi_finalize(ierr)
953  stop
954  ENDIF
955  col(ncol) = alo_2_plo(node,lay-1)
956  ENDIF
957 
958  IF(lay < kbm1) THEN
959  ncol = ncol + 1
960  !make sure we havent exceeded max_nz_row
961  IF(ncol > max_nz_row) THEN
962  WRITE(punit,*) 'WE HAVE EXCEEDED MAX_NZ_ROW'
963  WRITE(punit,*) 'INCREASE AND RECOMPILE'
964  CALL mpi_finalize(ierr)
965  stop
966  ENDIF
967  col(ncol) = alo_2_plo(node,lay+1)
968  ENDIF
969 
970  !update diagonal block nonzero counter for this row if it is in petsc local domain
971  DO colcount = 1, ncol
972  IF(col(colcount) <= psize) THEN
973  dnnz(petsc_pos) = dnnz(petsc_pos) + 1
974  ELSE
975  onnz(petsc_pos) = onnz(petsc_pos) + 1
976  ENDIF
977  ENDDO
978 
979  ENDDO
980 # endif
981 
982  ENDDO
983 
984 # if defined (NH)
985  !preallocate the matrix
986  IF(pdebug) THEN
987  WRITE(punit,*)'PREALLOCATING MATRIX'
988  WRITE(punit,*)'N_VERTS: ',n_verts,' MGL: ',mgl
989  WRITE(punit,*)' COUNT DNNZ ONNZ'
990  DO i=1, psize
991  WRITE(punit,*) i,dnnz(i),onnz(i)
992  ENDDO
993  ENDIF
994 # endif
995 
996 # if defined (NH)
997  CALL matcreatempiaij(mpi_comm_world,psize,psize,psizegl,psizegl,0,dnnz,0,onnz,a,ierr);chkerrq(ierr)
998 # endif
999 
1000 # if defined (SEMI_IMPLICIT) || (NH)
1001  CALL matcreatempiaij(mpi_comm_world,n_verts,n_verts,mgl,mgl,0,dnnz2,0,onnz2,a_el,ierr);chkerrq(ierr)
1002 # endif
1003 
1004 # if defined (WAVE_CURRENT_INTERACTION) && !defined (EXPLICIT)
1005  CALL matcreatempiaij(mpi_comm_world,n_verts,n_verts,mgl,mgl,0,dnnz2,0,onnz2,a_wave,ierr);chkerrq(ierr)
1006 # endif
1007 
1008  !deallocate non-zero counters
1009 # if defined (NH)
1010  DEALLOCATE(onnz,dnnz)
1011 # endif
1012  DEALLOCATE(onnz2,dnnz2)
1013 
1014  IF(pdebug) WRITE(punit,*)'FINISHING: ',trim(subname)
1015 
1016  RETURN
1017  END SUBROUTINE petsc_alloca

1021  SUBROUTINE petsc_solver
1022  !============================================================================
1023  ! petsc_solve
1024  ! solve Ax=b, report convergence success, iteration count
1025  ! option to initialize b using exact solution x to check solution
1026  !
1027  ! basic function: solve for a new x
1028  !
1029  ! dependencies: pmsr,u,A,b,one,ierr,ksb,x,neg_one,check_exact from this module
1030  ! norm,its from this module
1031  !
1032  !============================================================================
1033  IMPLICIT NONE
1034  CHARACTER(LEN=20) :: subname = 'PETSc_SOLVER'
1035  petscint :: reason, ierr
1036 
1037  IF(pdebug) WRITE(punit,*) 'STARTING: ', trim(subname)
1038  !---------------------------------------------------------
1039  ! option: use an exact solution of u=1 to create a RHS
1040  !---------------------------------------------------------
1041  IF(check_exact) THEN
1042  CALL vecset(u,one,ierr);chkerrq(ierr)
1043  CALL vecset(x,zero,ierr);chkerrq(ierr)
1044  CALL matmult(a,u,b,ierr);chkerrq(ierr)
1045  ENDIF
1046 
1047  !---------------------------------------------------------
1048  ! solve the linear system Ax=b and gather info
1049  ! about convergence rates, iterations, etc.
1050  !---------------------------------------------------------
1051  CALL kspsolve(ksp,b,x,ierr);chkerrq(ierr)
1052 
1053  !check what happened (divergence/convergence) and report
1054  CALL kspgetconvergedreason(ksp,reason,ierr)
1055  IF(reason < 0) THEN
1056  IF(pmsr) THEN
1057  WRITE (pscrn,*) 'PETSc SOLVER HAS DIVERGED: STOPPING...'
1058  ENDIF
1059  CALL mpi_finalize(ierr)
1060  stop
1061  ELSE
1062  CALL kspgetiterationnumber(ksp,its,ierr);chkerrq(ierr)
1063  !qxu IF(PMSR) WRITE(PSCRN,*)'PETSc CONVERGED IN : ',ITS,'AT RATE ',RTOL**(1.0d0/FLOAT(ITS))
1064  ENDIF
1065 
1066  !option to check solution for exact case
1067  IF(check_exact) THEN
1068  CALL vecaxpy(x,neg_one,u,ierr);chkerrq(ierr)
1069  CALL vecnorm(x,norm_2,norm,ierr);chkerrq(ierr)
1070  IF(pmsr) WRITE(punit,*) 'ERROR NORM FROM EXACT TEST: ',norm/float(psizegl)
1071  IF(pmsr) WRITE(pscrn,*) 'ERROR NORM : ',norm/float(psizegl)
1072  ELSE
1073  CALL vecnorm(x,norm_2,norm,ierr);chkerrq(ierr)
1074  IF(pmsr) WRITE(pscrn,*) 'SOLUTION NORM : ',norm
1075  ENDIF
1076 
1077  IF(pdebug) WRITE(punit,*) 'FINISHING: ',trim(subname)
1078 
1079  RETURN
1080  END SUBROUTINE petsc_solver
