108  IF (iobcn > 0) THEN
109  DO i=1,iobcn
110  i1 = i_obc_n(i)
111  i2 = adjn_obc(i,1)
112  DO j = 1, ntve(i1)
113  ic = nbve(i1,j)
114  n1 = nv(ic,1) ; n2 = nv(ic,2) ; n3 = nv(ic,3)
115  IF( n1-i2 == 0 .OR. n2-i2 == 0 .OR. n3-i2 == 0)THEN
116 # if defined (SPHERICAL)
117  x1_dp=vx(i1); y1_dp=vy(i1)
118  x2_dp=vx(i2); y2_dp=vy(i2)
119  CALL arcx(x1_dp,y1_dp,x2_dp,y2_dp,side)
120  dxn = side; dyn = (vy(i2)-vy(i1))*tpi
121 
122  x2_dp=xc(ic); y2_dp=yc(ic)
123  CALL arcx(x1_dp,y1_dp,x2_dp,y2_dp,side)
124  dxc = side; dyc = (yc(ic)-vy(i1))*tpi
125 # else
126  dxn = vx(i2)-vx(i1) ; dyn = vy(i2)-vy(i1)
127  dxc = xc(ic)-vx(i1) ; dyc = yc(ic)-vy(i1)
128 # endif
129  cross = sign(1.0_sp,dxc*dyn - dyc*dxn)
130  cxobc(ic) = cross*dyn/sqrt(dxn**2 +dyn**2)
131  cyobc(ic) = -cross*dxn/sqrt(dxn**2 +dyn**2)
132  END IF
133  END DO
134 
135  IF(nadjn_obc(i) > 1)THEN

137  DO j = 1, ntve(i1)
138  ic = nbve(i1,j)
139  n1 = nv(ic,1) ; n2 = nv(ic,2) ; n3 = nv(ic,3)
140  IF( n1-i2 == 0 .OR. n2-i2 == 0 .OR. n3-i2 == 0)THEN
141 # if defined (SPHERICAL)
142  x1_dp=vx(i1); y1_dp=vy(i1)
143  x2_dp=vx(i2); y2_dp=vy(i2)
144  CALL arcx(x1_dp,y1_dp,x2_dp,y2_dp,side)
145  dxn = side; dyn = (vy(i2)-vy(i1))*tpi
146 
147  x2_dp=xc(ic); y2_dp=yc(ic)
148  CALL arcx(x1_dp,y1_dp,x2_dp,y2_dp,side)
149  dxc = side; dyc = (yc(ic)-vy(i1))*tpi
150 # else
151  dxn = vx(i2)-vx(i1) ; dyn = vy(i2)-vy(i1)
152  dxc = xc(ic)-vx(i1) ; dyc = yc(ic)-vy(i1)
153 # endif
154  cross = sign(1.0_sp,dxc*dyn - dyc*dxn)
155  cxobc(ic) = cross*dyn/sqrt(dxn**2 +dyn**2)

158  END DO
159  END IF
