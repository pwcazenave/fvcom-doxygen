30 ! CALCULATE THE GRADIENT OF THE WATER DEPTH WITH X ANDY |
31 !==============================================================================|
32 
33  SUBROUTINE depth_gradient
34 
35 !==============================================================================|
36  USE all_vars
37  USE mod_par
38 
39  IMPLICIT NONE
40  REAL(SP) :: PHPX,PHPY,F1
41  INTEGER :: I,J,I1,I2
42 # if defined (SPHERICAL)
43  REAL(DP) :: XTMP,XTMP1
44 # endif
45 !==============================================================================|
46 
47 !----------CALCULATE DERIVATIVES OF DEPTH WITH X AND Y AT NODES----------------!
48 
49  DO i=1,m
50  phpx = 0.0_sp ; phpy = 0.0_sp
51  DO j=1,ntsn(i)-1
52  i1 = nbsn(i,j)
53  i2 = nbsn(i,j+1)
54  f1 = 0.50_sp*(h(i1)+h(i2))
55 # if defined (SPHERICAL)
56  phpx=phpx+f1*(vy(i1)-vy(i2))*tpi
57 
58  xtmp = vx(i2)*tpi-vx(i1)*tpi
59  xtmp1 = vx(i2)-vx(i1)
60  IF(xtmp1 > 180.0_sp)THEN
61  xtmp = -360.0_sp*tpi+xtmp
62  ELSE IF(xtmp1 < -180.0_sp)THEN
63  xtmp = 360.0_sp*tpi+xtmp
64  END IF
65  phpy=phpy+f1*xtmp*cos(deg2rad*vy(i))
66 ! PHPY=PHPY+F1*(VX(I2)-VX(I1))*TPI*COS(DEG2RAD*TY)
68 # else
69  phpx = phpx + f1*(vy(i1)-vy(i2))
70  phpy = phpy + f1*(vx(i2)-vx(i1))
71 # endif
72  END DO
73  phpx = phpx/art2(i)
74  phpy = phpy/art2(i)
75 
76  IF(phpx==0.0_sp .AND. phpy ==0.0_sp)THEN
77  sita_gd(i) = 0.0_sp ; phpn(i) = 0.0_sp
78  ELSE
79  sita_gd(i) = atan2(phpy,phpx)
80  phpn(i) = sqrt(phpy*phpy+phpx*phpx)
81  END IF
82 
83  END DO
84 
85 # if defined (NH) && (MULTIPROCESSOR)
86  IF(par) CALL aexchange(nc,myid,nprocs,sita_gd,phpn)
87 # endif
88 
89  RETURN
90  END SUBROUTINE depth_gradient
