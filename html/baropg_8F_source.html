30 ! CALCULATE THE BAROCLINIC PRESSURE GRADIENT IN SIGMA COORDINATES |
31 !==============================================================================|
32 
33  SUBROUTINE baropg
34 
35 !==============================================================================|
36  USE all_vars
37  USE mod_spherical
38  USE mod_northpole
39  USE mod_wd
40 
41  IMPLICIT NONE
42  REAL(SP) :: RIJK(0:n,3,kbm1), DRIJK1(0:n,3,kbm1), DRIJK2(0:n,kbm1)
43  REAL(SP) :: TEMP,DIJ,DRHO1,DRHO2
44  INTEGER :: I,K,J,J1,J2,IJK
45 # if defined (SPHERICAL)
46  REAL(SP) :: XTMP,XTMP1
47 # endif
48 !==============================================================================|
49 
50  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: baropg.F"
51 
52  ! USE RAMP CALCULATED IN 'internal_step.F'
53 
54 !----------SUBTRACT MEAN DENSITY TO MINIMIZE ROUNDOFF ERROR--------------------!
55 
56  rho1(:,1:kbm1) = rho1(:,1:kbm1) - rmean1(:,1:kbm1)
57  rho = rho - rmean
58 
59 !----------INITIALIZE ARRAYS---------------------------------------------------!
60 
61  drhox = 0.0_sp
62  drhoy = 0.0_sp
63  rmean(0,:) = 0.0_sp
64  rho(0,:) = 0.0_sp
65  rijk = 0.0_sp
66  drijk1 = 0.0_sp
67  drijk2 = 0.0_sp
68 
69 !----------CALCULATE AVERAGE DENSITY ON EACH EDGE------------------------------!
70 
71  DO k=1,kbm1
72  DO i=1,n
73  DO j=1,3
74  j1=j+1-int((j+1)/4)*3
75  j2=j+2-int((j+2)/4)*3
76  rijk(i,j,k) = 0.5_sp*(rho1(nv(i,j1),k)+rho1(nv(i,j2),k))
77  END DO
78  END DO
79  END DO
80 
81  DO i=1,n
82  DO j=1,3
83  drijk1(i,j,1)=rijk(i,j,1)*(-zz1(i,1))
84  DO k=2,kbm1
85  drijk1(i,j,k)=0.5_sp*(rijk(i,j,k-1)+rijk(i,j,k))*(zz1(i,k-1)-zz1(i,k))
86  drijk1(i,j,k)=drijk1(i,j,k)+drijk1(i,j,k-1)
87  END DO
88  END DO
89  END DO
90 
91  DO i=1,n
92  drijk2(i,1)=0.0_sp
93  DO k=2,kbm1
94  drijk2(i,k)=0.5_sp*(zz1(i,k-1)+zz1(i,k))*(rho(i,k)-rho(i,k-1))
95  drijk2(i,k)=drijk2(i,k-1)+drijk2(i,k)
96  END DO
97  END DO
98 
99  DO i = 1, n
100 # if defined (WET_DRY)
101  IF(iswetct(i)*iswetc(i) == 1 .AND. &
102  (h(nv(i,1)) > static_ssh_adj .OR. h(nv(i,2)) > static_ssh_adj .OR. h(nv(i,3)) > static_ssh_adj))THEN
103 # endif
104  DO k=1,kbm1
105  DO j = 1, 3
106  j1=j+1-int((j+1)/4)*3
107  j2=j+2-int((j+2)/4)*3
108  ijk=nbe(i,j)
109  dij=0.5_sp*(dt(nv(i,j1))+dt(nv(i,j2)))
110 
111 # if defined (SPHERICAL)
112  drho1=-deltuy(i,j)*drijk1(i,j,k)*dt1(i)
113  drho2=-deltuy(i,j)*dij*drijk2(i,k)
114 # else
115  drho1=(vy(nv(i,j1))-vy(nv(i,j2)))*drijk1(i,j,k)*dt1(i)
116  drho2=(vy(nv(i,j1))-vy(nv(i,j2)))*dij*drijk2(i,k)
117 # endif
118  drhox(i,k)=drhox(i,k)+drho1+drho2
119 
120 # if defined (SPHERICAL)
121  xtmp = vx(nv(i,j2))*tpi-vx(nv(i,j1))*tpi
122  xtmp1 = vx(nv(i,j2))-vx(nv(i,j1))
123  IF(xtmp1 > 180.0_sp)THEN
124  xtmp = -360.0_sp*tpi+xtmp
125  ELSE IF(xtmp1 < -180.0_sp)THEN
126  xtmp = 360.0_sp*tpi+xtmp
127  END IF
128 
129  drho1=xtmp*cos(deg2rad*yc(i))*drijk1(i,j,k)*dt1(i)
130  drho2=xtmp*cos(deg2rad*yc(i))*dij*drijk2(i,k)
131 ! DRHO1=DELTUX(I,J)*DRIJK1(I,J,K)*DT1(I)
132 ! DRHO2=DELTUX(I,J)*DIJ*DRIJK2(I,K)
133 # else
134  drho1=(vx(nv(i,j2))-vx(nv(i,j1)))*drijk1(i,j,k)*dt1(i)
135  drho2=(vx(nv(i,j2))-vx(nv(i,j1)))*dij*drijk2(i,k)
136 # endif
137  drhoy(i,k)=drhoy(i,k)+drho1+drho2
138 
139  END DO
140  END DO
141 # if defined (WET_DRY)
142  END IF
143 # endif
144  END DO
145 
146 # if defined (SPHERICAL)
147  CALL baropg_xy(drijk1,drijk2)
148 # endif
149 
150 
151 !----------MULTIPLY BY GRAVITY AND ELEMENT DEPTH-------------------------------!
152 
153  DO k=1,kbm1
154  drhox(:,k)=drhox(:,k)*dt1(:)*dz1(:,k)*grav_e(:)*ramp
155  drhoy(:,k)=drhoy(:,k)*dt1(:)*dz1(:,k)*grav_e(:)*ramp
156  END DO
157 
158 !----------ADD MEAN DENSITY BACK ON--------------------------------------------!
159 
160  rho1 = rho1 + rmean1
161  rho = rho + rmean
162 
163  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "End: baropg.F"
164 
165  RETURN
166  END SUBROUTINE baropg
