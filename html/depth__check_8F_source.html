35 SUBROUTINE depth_check
36 
37  !==============================================================================|
38  USE all_vars
39  USE mod_utils
40  USE mod_par
41  IMPLICIT NONE
42  INTEGER, DIMENSION(NPROCS) :: SBUF,RBUF

44  REAL(SP) :: DMIN
45  !==============================================================================|
46 
47  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: depth_check"
48 !--Calculate Minimum Depth and Set Global Node Number if Min Depth < MIN_DEPTH
49  sbuf = 0
50  mloc = 0
51  ierr = 0
52  dmin = minval(d(1:m))
53  ! NGID NOW EXISTS FOR BOTH SERIAL AND PARALLEL
54  IF(dmin < min_depth) mloc = ngid(minloc(d(1:m),dim=1))
55 
56 !--Reduce in Master Processor Array and Dump To Screen
57  sbuf(myid) = mloc
58  rbuf = sbuf
59 # if defined (MULTIPROCESSOR)
60  recv=msrid-1
61  IF(par)CALL mpi_reduce(sbuf,rbuf,nprocs,mpi_integer,mpi_max,recv,mpi_fvcom_group,ierr)
62  IF(ierr/=0) CALL fatal_error("COULD NOT MPI_REDUCE IN DEPTH CHECK")
63 # endif
64 
65 
66 !--If Depth Condition is Violated Write Warning and Halt User
67  IF(msr)THEN
68  IF(sum(rbuf) /= 0) THEN
69  DO i=1,nprocs
70  ii = rbuf(i)
71  IF(ii /= 0)THEN
72  ! WRITE(*,*)'DEPTH IN NODE: ',II,' AT ',XG(II)+VXMIN,YG(II)+VYMIN, &
73  ! ' IS LESS THAN MIN_DEPTH'
74 
75  WRITE(*,*)'DEPTH IN NODE: ',ii,'; IS LESS THAN MIN_DEPTH'
76  WRITE(*,*)'ADJUST BATHYMETRY AT THIS (THESE) LOCATION(S) OR'
77  WRITE(*,*)'RECOMPILE FVCOM WITH FLOODING/DRYING FORMULATION'
78  WRITE(*,*)'STOPPING....'
79  CALL pstop
80  END IF
81  END DO
82  END IF
83  END IF
84 
85  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "End: depth_check"
86 
87  END SUBROUTINE depth_check
