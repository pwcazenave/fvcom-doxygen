262  REAL(SP), DIMENSION(3*(NT),KBM1) :: uvn
263  REAL(SP) :: utmp,vtmp,sitai,ffd,ff1 !,X11,Y11,X22,Y22,X33,Y33,TMP1,TMP2,XI,YI
264  REAL(SP) :: dxa,dya,dxb,dyb,fij1,fij2,un
265  REAL(SP) :: txx,tyy,fxx,fyy,viscof,exflux,temp,stpoint
266  REAL(SP) :: fact,fm1
267  REAL(SP) :: s1min, s1max, s2min, s2max,smin,smax,xxxx
268  INTEGER :: i,i1,i2,ia,ib,j,j1,j2,k,jtmp,jj,kk,ip,ii
269 !!$# if defined (SPHERICAL)
270 !!$ REAL(SP) :: ty,txpi,typi
271 !!$ REAL(DP) :: XTMP1,XTMP
272 !!$ REAL(DP) :: X1_DP,Y1_DP,X2_DP,Y2_DP,XII,YII
273 !!$# endif
274 # if defined (WET_DRY)
275  INTEGER :: n_ntve
276 # endif
277 # if defined (SEMI_IMPLICIT)
278  REAL(SP) :: un1
279  REAL(SP), DIMENSION(3*(NT),KBM1) :: uvn1
280  REAL(SP), DIMENSION(3*(NT),KBM1) :: dtij1
281 # endif
282 
283  IF(dbg_set(dbg_sbr)) WRITE(ipt,*)"Start: adv_dye"
284 !------------------------------------------------------------------------------!
285 
286  SELECT CASE(horizontal_mixing_type)
287  CASE ('closure')
288  fact = 1.0_sp
289  fm1 = 0.0_sp
290  CASE('constant')
291  fact = 0.0_sp
292  fm1 = 1.0_sp
293  CASE DEFAULT
294  CALL fatal_error("UNKNOW HORIZONTAL MIXING TYPE:",&
295  & trim(horizontal_mixing_type) )
296  END SELECT
297 
298 !
299 !--Initialize Fluxes and dyef----------------------------------------------------------!
300 !
301  xflux = 0.0_sp
302  xflux_adv = 0.0_sp
303  dyef = 0.0_sp
304 !
305 !--Loop Over Control Volume Sub-Edges And Calculate Normal Velocity------------!
306 !
307 !!# if !defined (WET_DRY)
308  DO i=1,ncv
309  i1=ntrg(i)
310 ! DTIJ(I)=DT1(I1)
311  DO k=1,kbm1
312  dtij(i,k)=dt1(i1)*dz1(i1,k)
313  uvn(i,k) = v(i1,k)*dltxe(i) - u(i1,k)*dltye(i)
314 # if defined (SEMI_IMPLICIT)
315  dtij1(i,k) = d1(i1)*dz1(i1,k)
316  uvn1(i,k) = vf(i1,k)*dltxe(i) - uf(i1,k)*dltye(i)
317 # endif
318  END DO
319  END DO
320 !!# else
321 !! DO I=1,NCV
322 !! I1=NTRG(I)
323 !!! DTIJ(I)=DT1(I1)
324 !! DO K=1,KBM1
325 !! DTIJ(I,K)=DT1(I1)*DZ1(I1,K)
326 !! UVN(I,K) = VS(I1,K)*DLTXE(I) - US(I1,K)*DLTYE(I)
327 !!# if defined (SEMI_IMPLICIT)
328 !! DTIJ1(I,K) = D1(I1)*DZ1(I1,K)
329 !! UVN1(I,K) = VF(I1,K)*DLTXE(I) - UF(I1,K)*DLTYE(I)
330 !!# endif
331 !! END DO
332 !! END DO
333 !!# endif
334 
335 !
336 !--Calculate the Advection and Horizontal Diffusion Terms----------------------!
337 !
338 
339  DO k=1,kbm1
340  pdyepx = 0.0_sp
341  pdyepy = 0.0_sp
342  pdyepxd = 0.0_sp
343  pdyepyd = 0.0_sp
344  DO i=1,m
345  DO j=1,ntsn(i)-1
346  i1=nbsn(i,j)
347  i2=nbsn(i,j+1)
348 
349 # if defined (WET_DRY)
350  IF(iswetn(i1) == 0 .AND. iswetn(i2) == 1)THEN
351  ffd=0.5_sp*(dye(i,k)+dye(i2,k))!-DYEMEAN(I,K)-DYEMEAN(I2,K))
352  ff1=0.5_sp*(dye(i,k)+dye(i2,k))
353  ELSE IF(iswetn(i1) == 1 .AND. iswetn(i2) == 0)THEN
354  ffd=0.5_sp*(dye(i1,k)+dye(i,k))!-DYEMEAN(I1,K)-DYEMEAN(I,K))
355  ff1=0.5_sp*(dye(i1,k)+dye(i,k))
356  ELSE IF(iswetn(i1) == 0 .AND. iswetn(i2) == 0)THEN
357  ffd=0.5_sp*(dye(i,k)+dye(i,k))!-DYEMEAN(I,K)-DYEMEAN(I,K))
358  ff1=0.5_sp*(dye(i,k)+dye(i,k))
359  ELSE
360  ffd=0.5_sp*(dye(i1,k)+dye(i2,k))!-DYEMEAN(I1,K)-DYEMEAN(I2,K))
361  ff1=0.5_sp*(dye(i1,k)+dye(i2,k))
362  END IF
363 # else
364  ffd=0.5_sp*(dye(i1,k)+dye(i2,k))!-DYEMEAN(I1,K)-DYEMEAN(I2,K))
365  ff1=0.5_sp*(dye(i1,k)+dye(i2,k))
366 # endif
367 
368 !!$# if defined (SPHERICAL)
369 !!$ XTMP = VX(I2)*TPI-VX(I1)*TPI
370 !!$ XTMP1 = VX(I2)-VX(I1)
371 !!$ IF(XTMP1 > 180.0_SP)THEN
372 !!$ XTMP = -360.0_SP*TPI+XTMP
373 !!$ ELSE IF(XTMP1 < -180.0_SP)THEN
374 !!$ XTMP = 360.0_SP*TPI+XTMP
375 !!$ END IF
376 !!$ TXPI=XTMP*COS(DEG2RAD*VY(I))
377 !!$ TYPI=(VY(I1)-VY(I2))*TPI
378 !!$
379 !!$ PDYEPX(I)=PDYEPX(I)+FF1*TYPI
380 !!$ PDYEPY(I)=PDYEPY(I)+FF1*TXPI
381 !!$ PDYEPXD(I)=PDYEPXD(I)+FFD*TYPI
382 !!$ PDYEPYD(I)=PDYEPYD(I)+FFD*TXPI
383 !!$# else
384 !!$ PDYEPX(I)=PDYEPX(I)+FF1*(VY(I1)-VY(I2))
385 !!$ PDYEPY(I)=PDYEPY(I)+FF1*(VX(I2)-VX(I1))
386 !!$ PDYEPXD(I)=PDYEPXD(I)+FFD*(VY(I1)-VY(I2))
387 !!$ PDYEPYD(I)=PDYEPYD(I)+FFD*(VX(I2)-VX(I1))
388 !!$# endif
389 
390  pdyepx(i)=pdyepx(i)+ff1*dltytrie(i,j)
391  pdyepy(i)=pdyepy(i)+ff1*dltxtrie(i,j)
392  pdyepxd(i)=pdyepxd(i)+ffd*dltytrie(i,j)
393  pdyepyd(i)=pdyepyd(i)+ffd*dltxtrie(i,j)
394 
395  END DO
396  pdyepx(i)=pdyepx(i)/art2(i)
397  pdyepy(i)=pdyepy(i)/art2(i)
398  pdyepxd(i)=pdyepxd(i)/art2(i)
399  pdyepyd(i)=pdyepyd(i)/art2(i)
400  END DO
401 
402  IF(k == kbm1)THEN
403  DO i=1,m
404  pfpxb(i) = pdyepx(i)
405  pfpyb(i) = pdyepy(i)
406  END DO
407  END IF
408 
409  DO i=1,m
410  viscoff(i)=viscofh(i,k)
411  END DO
412 
413  IF(k == kbm1) THEN
414 ! AH_BOTTOM(1:M) = HORCON*(FACT*VISCOFF(1:M) + FM1)
415  ah_bottom(1:m) = (fact*viscoff(1:m) + fm1)*nn_hvc(1:m)
416  END IF
417 
418 
419  DO i=1,ncv_i
420  ia=niec(i,1)
421  ib=niec(i,2)
422 
423 
424 !!$ XI=0.5_SP*(XIJE(I,1)+XIJE(I,2))
425 !!$ YI=0.5_SP*(YIJE(I,1)+YIJE(I,2))
426 !!$# if defined (SPHERICAL)
427 !!$ X1_DP=XIJE(I,1)
428 !!$ Y1_DP=YIJE(I,1)
429 !!$ X2_DP=XIJE(I,2)
430 !!$ Y2_DP=YIJE(I,2)
431 !!$ CALL ARCC(X2_DP,Y2_DP,X1_DP,Y1_DP,XII,YII)
432 !!$ XI=XII
433 !!$ XTMP = XI*TPI-VX(IA)*TPI
434 !!$ XTMP1 = XI-VX(IA)
435 !!$ IF(XTMP1 > 180.0_SP)THEN
436 !!$ XTMP = -360.0_SP*TPI+XTMP
437 !!$ ELSE IF(XTMP1 < -180.0_SP)THEN
438 !!$ XTMP = 360.0_SP*TPI+XTMP
439 !!$ END IF
440 !!$
441 !!$ DXA=XTMP*COS(DEG2RAD*VY(IA))
442 !!$ DYA=(YI-VY(IA))*TPI
443 !!$ XTMP = XI*TPI-VX(IB)*TPI
444 !!$ XTMP1 = XI-VX(IB)
445 !!$ IF(XTMP1 > 180.0_SP)THEN
446 !!$ XTMP = -360.0_SP*TPI+XTMP
447 !!$ ELSE IF(XTMP1 < -180.0_SP)THEN
448 !!$ XTMP = 360.0_SP*TPI+XTMP
449 !!$ END IF
450 !!$
451 !!$ DXB=XTMP*COS(DEG2RAD*VY(IB))
452 !!$ DYB=(YI-VY(IB))*TPI
453 !!$# else
454 !!$ DXA=XI-VX(IA)
455 !!$ DYA=YI-VY(IA)
456 !!$ DXB=XI-VX(IB)
457 !!$ DYB=YI-VY(IB)
458 !!$# endif
459 
460  fij1=dye(ia,k)+dltxncve(i,1)*pdyepx(ia)+dltyncve(i,1)*pdyepy(ia)
461  fij2=dye(ib,k)+dltxncve(i,2)*pdyepx(ib)+dltyncve(i,2)*pdyepy(ib)
462 
463  s1min=minval(dye(nbsn(ia,1:ntsn(ia)-1),k))
464  s1min=min(s1min, dye(ia,k))
465  s1max=maxval(dye(nbsn(ia,1:ntsn(ia)-1),k))
466  s1max=max(s1max, dye(ia,k))
467  s2min=minval(dye(nbsn(ib,1:ntsn(ib)-1),k))
468  s2min=min(s2min, dye(ib,k))
469  s2max=maxval(dye(nbsn(ib,1:ntsn(ib)-1),k))
470  s2max=max(s2max, dye(ib,k))
471  IF(fij1 < s1min) fij1=s1min
472  IF(fij1 > s1max) fij1=s1max
473  IF(fij2 < s2min) fij2=s2min
474  IF(fij2 > s2max) fij2=s2max
475 
476  un=uvn(i,k)
477 # if defined (SEMI_IMPLICIT)
478  un1=uvn1(i,k)
479 # endif
480 
481  ! VISCOF=HORCON*(FACT*(VISCOFF(IA)+VISCOFF(IB))*0.5_SP + FM1)
482 
483  ! David moved HPRNU and added VHC
484  viscof=(fact*0.5_sp*(viscoff(ia)*nn_hvc(ia)+viscoff(ib)*nn_hvc(ib)) + &
485  fm1*0.5_sp*(nn_hvc(ia)+nn_hvc(ib)))
486 
487  txx=0.5_sp*(pdyepxd(ia)+pdyepxd(ib))*viscof
488  tyy=0.5_sp*(pdyepyd(ia)+pdyepyd(ib))*viscof
489 
490  fxx=-dtij(i,k)*txx*dltye(i)
491  fyy= dtij(i,k)*tyy*dltxe(i)
492 
493 # if !defined (SEMI_IMPLICIT)
494  exflux=-un*dtij(i,k)* &
495  ((1.0_sp+sign(1.0_sp,un))*fij2+(1.0_sp-sign(1.0_sp,un))*fij1)*0.5_sp+fxx+fyy
496 # else
497  exflux=-un*dtij(i,k)* &
498  ((1.0_sp+sign(1.0_sp,un))*fij2+(1.0_sp-sign(1.0_sp,un))*fij1)*0.5_sp
499  exflux=(1.0_sp-ifceta)*exflux+ifceta*(-un1*dtij1(i,k)* &
500  ((1.0_sp+sign(1.0_sp,un1))*fij2+(1.0_sp-sign(1.0_sp,un1))*fij1)*0.5_sp)+fxx+fyy
501 # endif
502 
503  xflux(ia,k)=xflux(ia,k)+exflux
504  xflux(ib,k)=xflux(ib,k)-exflux
