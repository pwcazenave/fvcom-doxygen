7 !****************************************************************
8 !
9 ! Computation of the source terms due to bottom friction
10 !
11 ! Method
12 !
13 ! In SWAN several bottom dissipation models are computed, i.e.:
14 !
15 ! IBOT = 1 Jonswap bottom friction model
16 ! IBOT = 2 Collins bottom friction model
17 ! IBOT = 3 Madsen bottom friction model (see Tolman)
18 !
19 ! Both methods are implemented in SWAN and the user has to make
20 ! a choice in the input file.
21 !
22 ! 1. Jonswap model:
23 ! -----------------
24 !
25 ! The bottom interaction term SEbf(s,d) is supposed to take the
26 ! Jonswap form (Hasselman et al. 1973):
27 ! 2
28 ! sigma E(s,d)
29 ! SEbf = -GAMMA ----------------
30 ! 2 2
31 ! g sinh (kD)
32 ! 2 -3
33 ! where GAMMA is the decay parameter ,(default GAMMA = 0.067 m s ).
34 ! In the Jonswap form the current velocities are not taken into
35 ! account.
36 !
37 ! 2. COLLINS model:
38 ! -----------------
39 !
40 ! The energy dissipation due to bottom friction is modelled
41 ! according the quadratic friction law:
42 ! 2
43 ! SE = Tau * |U|
44 !
45 ! which for a spectrum can be written as:
46 ! 2
47 ! sigma
48 ! SE(s,d)= - ---------------- * (Cfw.Ub + Cfc.Uc) * E(s,d)
49 ! 2
50 ! g sinh (K(s) * D)
51 !
52 ! Ub is the velocity due to the wave at the bottom
53 !
54 ! The current velocity is Uc
55 !
56 ! 2. MADSEN formulation:
57 ! ----------------------
58 !
59 ! The bottom dissipation applying Madsen formulation is as
60 ! follows:
61 !
62 ! fw [n - 1/2] UBR E(s,d)
63 ! [1] Sdsb(s,d) = - ------------------------
64 ! D
65 !
66 ! in which :
67 ! 2
68 ! s * D
69 ! [1a] (n - 1/2) = -------------
70 ! 2
71 ! 2 g sinh (kD)
72 !
73 ! UBOT(IX,IY) is computed in the subroutine SINTGRL. The friction
74 ! factor fw is estimated using the formulation of Jonsson (1963,
75 ! 1966a):
76 !
77 ! 1 1 Ab,r
78 ! [2] -------- + log { ---------- } = mf + log { ----- }
79 ! 4 sqrt(fw) 10 4 sqrt(fw) 10 Kn
80 !
81 ! with:
82 !
83 ! 2 // 1
84 ! [3] Ab,r = 2 * // -------------- E(s,d) ds dd
85 ! // 2
86 ! sinh (kD)
87 !
88 ! with: Ab,r is the representative near bottom excursion
89 ! amplitude
90 ! Kn equivalent roughness
91 ! mf constant ( mf = -0.08) (determined by Jonsson
92 ! and Carlssen 1976 )
93 !
94 ! [2] is only valid for Ab,r/Kn larger than approximately 1.
95 ! For smaller values a constant value of fw is used (fw = 0.3
96 ! for Ab,r/Kn < 1.57 )

213  SUBROUTINE frabre ( HM, ETOT, QBLOC )
214 !
215 !****************************************************************
216 !
217 ! to compute the fraction of breaking waves in point ix,iy
218 ! of the computational grid
219 !
220 ! Method (updated...)
221 !
222 ! The fraction of breaking waves in a point ix,iy is given by
223 ! the implicit relation:
224 !
225 ! 1 - Qb ETOT
226 ! ------ = -8 * -----
227 ! ln Qb HM**2
228 !
229 ! from which Qb can be found by solving the equation:
230 !
231 ! ETOT
232 ! F = 1 - Qb + 8 * ---- * ln(Qb) = 0.
233 ! 2
234 ! HM
235 !
236 ! The following appproximation is applied:
237 !
238 ! 2
239 ! (1)| B = sqrt( 8 ETOT/HM ), i.e. B = Hrms/HM
240 !
241 !
242 ! | Qo = 0. B <= 0.5
243 ! (2)| 2
244 ! | Qo = ( 2B -1 ) 0.5 < B <= 1
245 !
246 !
247 ! applying the Newton-Raphson procedure (for 0.2<B<1.0):
248 !
249 ! | Qb = 0. B <= 0.2
250 ! |
251 ! | 2
252 ! | 2 Qo - exp((Qo-1)/B )
253 ! (3)| Qb = Qo - B ------------------ 0.2 < B < 1.0
254 ! | 2 2
255 ! | B - exp((Qo-1)/B )
256 ! |
257 ! |
258 ! | Qb = 1. B >= 1.0
259 ! |
260 !
261  USE swcomm4
262  USE ocpcomm4
263 !
264  IMPLICIT NONE
265 
266  REAL :: ETOT, HM, QBLOC
267  INTEGER :: IENT
268  REAL :: B, B2, QO, Z
269 
270  IF((hm > 0.) .AND. (etot >= 0.))THEN
271  b = sqrt(8. * etot / (hm*hm) )
272  ELSE
273  b = 0.0
274  END IF
275 !
276  IF(b <= 0.5)THEN
277  qo = 0.
278  ELSE IF(b <= 1.0)THEN
279  qo = (2.*b - 1.)**2
280  END IF
281 !
282  IF(b <= 0.2)THEN
283  qbloc = 0.0
284  ELSE IF(b < 1.0)THEN
285 !
286 ! *** second iteration to find Qb ***
287 !
288  b2 = b*b
289  z = exp((qo-1.)/b2)
290  qbloc = qo - b2 * (qo-z)/(b2-z)
291  ELSE
292  qbloc = 1.0
293  END IF
294 !
295  RETURN
296  END SUBROUTINE frabre

300  SUBROUTINE ssurf (ETOT ,HM ,QB ,SMEBRK , &
301  imatra ,imatda ,idcmin ,idcmax , &
302  plwbrk ,isstop ,ig )
303 !
304 !****************************************************************
305 ! Computation of the source term due to wave breaking.
306 ! White capping is not taken into account
307 !
308 ! Method
309 !
310 ! The source term for surf breaking is implemented following
311 ! the approach of Battjes/Janssen (1978) for the energy dissipation:
312 !
313 ! Alpha - 2 - SMEBRK
314 ! Dtot = ---- Qb * f * Hm with f = ------
315 ! 4 2 * Pi
316 !
317 ! Now the source term is:
318 !
319 ! SIGMA * AC2(ID,IS,IX,IY)
320 ! Sbr = - Dtot * ------------------------ =
321 ! Etot
322 !
323 !
324 ! Alpha * SMEBRK * Qb * Hm * Hm SIGMA * AC2(ID,IS,IX,IY)
325 ! = - ------------------------------ * -------------------------
326 ! 8 * Pi Etot
327 !
328 !
329 ! = WS * SIGMA * AC2(ID,IS,IX,IY) = WS * E
330 !
331 !
332 ! with
333 !
334 ! Alpha = PSURF(1) ;
335 !
336 ! SMEBRK Qb
337 ! WS = Alpha ------ -- ;
338 ! Pi BB
339 ! 2
340 ! BB = 8 Etot / Hm = - (1 - Qb) / ln (Qb) ;
341 !
342 !
343 ! The local maximum wave height Hm and mean frequency SMEBRK are computed
344 ! in subroutine SINTGRL.
345 ! The fraction of breaking waves Qb is calculated in the subroutine FRABRE
346 !
347 ! The new value for the dissipation is computed implicitly using
348 ! the last computed value for the action density Nold (at the spatial
349 ! gridpoint under consideration).
350 !
351 ! Sbr = WS * N
352 !
353 ! = Sbr_new + (d Sbr/d N) (Nnew - Nold)
354 !
355 ! = WS * Nnew + SbrD * (Nnew - Nold)
356 !
357 ! = (WS + SbrD)* Nnew - SbrD * Nold
358 !
359 ! = SURFA1 * Nnew - SURFA0 * Nold
360 !
361 ! In order to do this we need the derivative
362 ! of the source term Sbr to the action density N
363 !
364 ! d Sbr d WS
365 ! SbrD = ----- = ---- * N + WS
366 ! d N d N
367 !
368 ! Since BB and SMEBRK * N are proportional, we have
369 !
370 ! d Sbr d WS SMEBRK (d Qb/ d BB) *BB - Qb
371 ! ----- = ---- * BB + WS = Alpha ------ --------------------- * BB + WS =
372 ! d N d BB Pi sqr(BB)
373 !
374 !
375 ! SMEBRK d Qb
376 ! Alpha ------ ----
377 ! Pi d BB
378 !
379 ! With:
380 !
381 ! d Qb 1
382 ! ---- = ------------- ;
383 ! d BB (d BB / d Qb)
384 !
385 ! 2
386 ! d Qb ln (Qb)
387 ! ---- = ---------------------------
388 ! d BB ln (Qb) + (1 - Qb) (1 / Qb)
389 !
390 ! Qb (1 - Qb)
391 ! = ------------ ;
392 ! BB (BB - Qb)
393 !
394 ! ------------------------------------------------------------
395 ! Get HM, QB and ETOT from the subroutine SINTGRL
396 ! For spectral direction IS and ID do,
397 ! get the mean energy frequency average over the full spectrum
398 ! If ETOT > 0 then
399 ! compute source term for energy dissipation SURFA0 and SURFA1
400 ! Else
401 ! source term for wave breaking is 0.
402 ! End if
403 ! ----------------------------------------------------------
404 ! Compute source terms for energy averaged frequency
405 ! Store results in the arrays IMATDA and IMATRA
406 ! ------------------------------------------------------------
407  USE swcomm3
408  USE swcomm4
409  USE ocpcomm4
410 ! USE ALL_VARS, ONLY : MT,AC2
411  USE vars_wave, ONLY : mt,ac2
412 !
413  IMPLICIT NONE
414 
415  INTEGER :: ISSTOP,IDCMIN(msc),IDCMAX(msc)
416  REAL :: DISSC0(mdc,msc),DISSC1(mdc,msc), &
417  IMATDA(mdc,msc),IMATRA(mdc,msc),PLWBRK(mdc,msc,nptst)
418  REAL :: ETOT,HM,QB,SMEBRK
419  INTEGER :: ID,IDDUM,IENT,IS,IG
420  DOUBLE PRECISION BB,DIS0,SbrD,SURFA0,SURFA1,WS
421 !
422 ! ALFA = PSURF(1) <default = 1.0>
423 !
424  bb = 8. * dble(etot) / ( dble(hm)**2 )
425 ! SURFA0 = 0.
426 ! SURFA1 = 0.
427 ! IF(REAL(BB) > 0. .AND. REAL(ABS(BB - DBLE(QB))) > 0.)THEN
428 ! IF(BB < 1.)THEN
429 ! WS = ( DBLE(PSURF(1)) / DBLE(PI_W)) * DBLE(QB) * DBLE(SMEBRK) / BB
430 ! SbrD = WS * (1. - DBLE(QB)) / (BB - DBLE(QB))
431 ! ELSE
432 ! WS = ( DBLE(PSURF(1)) / DBLE(PI_W)) * DBLE(SMEBRK)
433 ! SbrD = 0.
434 ! END IF
435 ! SURFA0 = SbrD
436 ! SURFA1 = WS + SbrD
437 ! ELSE
438 ! SURFA0 = 0.
439 ! SURFA1 = 0.
440 ! END IF
441  IF(REAL(BB) > 0. .AND. REAL(ABS(BB - DBLE(QB))) > 0.)then
442  ws = ( dble(psurf(1)) / dble(pi_w)) * dble(qb) * dble(smebrk) / bb
443  ELSE
444  ws = 0.
445  ENDIF
446 
447 !
448 ! *** store the results for surf wave breaking ***
449 ! *** in the matrices IMATDA and IMATRA ***
450 !
451  DO is = 1, isstop
452  DO iddum = idcmin(is), idcmax(is)
453  id = mod( iddum - 1 + mdc , mdc ) + 1
454  imatda(id,is) = imatda(id,is) + REAL(ws)
455  dis0 = ws * dble(ac2(id,is,ig))
456  imatra(id,is) = imatra(id,is) - REAL(dis0)
457  END DO
458  END DO
459 
460 
461  RETURN
462  END SUBROUTINE ssurf

467  SUBROUTINE swcap (SPCDIR ,SPCSIG ,KWAVE ,IDCMIN , &
468  idcmax ,isstop ,etot ,imatda , &

471 ! (This subroutine has not been fully tested. Only tested for
472 ! case IWCAP = 2)
473 !
474 !****************************************************************
475 !
476 ! Calculates the dissipation due to whitecapping
477 !
478 ! Method
479 !
480 ! Whitecapping dissipation is formulated as follows:
481 !
482 ! S_wc(sig,th) = - C_wc E(sig,th)
483 !
484 ! where the coefficient C_wc has four basic forms:
485 !
486 ! C_wc1 = C_K sig~ (k/k~): According to Komen (generalised)
487 ! C_wc2 = C_BJ sig~ (k/k~): According to Battjes-Janssen (modified)
488 ! C_wc3 = C_LH : According to Longuet Higgins (1969)
489 ! C_wc4 = C_CSM : According to cumulative steepness
490 ! method (Ris et al. (1999) and Donelan (1999))
491 !
492 ! In these formulations C_K is defined as (Komen; 1994 p. 145):
493 !
494 ! n1 n2
495 ! C_K = C1 [(1-delta) + delta (k/k~) ] (S~/S_PM)
496 !
497 ! where C1, delta, n1 and n2 can be varied
498 !
499 !
500 ! C_BJ is defined as:
501 !
502 ! 2
503 ! alpha Hm Qb
504 ! C_BJ = -----------
505 ! 8 Pi m0
506 !
507 ! where alpha can be varied.
508 !
509 ! for Hrms > Hm the formulation changes in a limit to (Hrms->Hm; Qb->1):
510 !
511 ! alpha
512 ! C_BJ = -----
513 ! Pi
514 !
515 ! and C_LH is defined as (Komen; 1994 p. 151-152):
516 !
517 ! 4 2 2
518 ! C_LH = C3 Sqrt[(m0 sig0 )/g ] exp(A) sig0 (sig/sig0)
519 !
520 ! where
521 ! 2 2 4 2 2
522 ! A = -1/8 [1-eps ] [g /(m0 sig0 )] with [1-eps ] = [m2 ] / [m0 m4]
523 !
524 ! and C3 can be varied
525 !
526 !
527 ! Cumulative steepness method. Basic idea is that dissipation depends
528 ! on the steepness of the wave spectrum at and below a particular
529 ! frequency. Details can be found in "SWAN fysica plus" (2002),
530 ! report H3937/A832, implemented by Delf Hydraulics and Alkyon by
531 ! order of RIKZ/RWS as a part of the project HR-Ontwikkeling.
532 ! 40.41
533 ! Recent modifications are described in: 40.41
534 ! Hurdle, D.P., and G.Ph. van Vledder, 2004: Improved spectral wave 40.41
535 ! modelling of white-capping dissipation in swell sea systems. Proc. 40.41
536 ! Offshore Mechanics, Arctic Engineering Conference, June 2004, 40.41
537 ! Vancouver, Canada, paper OMAE2004-51562. 40.41
538 ! 40.41
539 ! Default values: C_st = 4 40.41
540 ! m = 2 40.41
541 ! 40.41
542 ! C_CSM = A*C_st S(sig,th) 40.41
543 !
544 ! where
545 !
546 ! C_st is a tuneable coefficient and
547 !
548 ! S(sig,th) = integrals from 0 to sig and from 0 to 2pi of
549 !
550 ! k^2 |cos(th-th')|^m E(sig,th) dsig dth'
551 !
552 ! where coefficient m controls the directional dependence in case
553 ! of straining mechanism (if dominant then m = 2 else m > 10) and
554 ! th is the actual direction where straining mechanism takes
555 ! place.
556 ! 40.41
557 ! A is a normalisation coefficient computed 40.41
558 ! 40.41
559 ! 1 GAMMA(0.5*m+1.) 40.41
560 ! A = -------- ---------------- 40.41
561 ! SQRT(PI) GAMMA(0.5*m+0.5) 40.41
562 !
563 ! In these equations the variables have the following meaning:
564 !
565 ! Hm : Maximum wave height
566 ! Hrms : Root mean square of the wave heights
567 ! eps^2: Measure for the spectral bandwidth
568 ! m0 : Total wave energy density (=ETOT)
569 ! m2 : Second moment of the variance spectrum (=ETOT2)
570 ! m4 : Fourth moment of the variance spectrum (=ETOT4)
571 ! k : Wave number (=KWAVE(IS,1))
572 ! k~ : Mean wave number
573 ! Qb : Fraction of breaking waves
574 ! sig : Frequency (=SPCSIG(IS))
575 ! sig0 : Average zero crossing frequency
576 ! sig~ : Mean frequency
577 ! S~ : Overall steepness (STP_OV)
578 ! S_PM : Overall steepness for a Pierson-Moskowitz spectrum
579 ! th : direction theta (=SPCDIR(ID))
580 !
581 ! ------------------------------------------------------------
582 ! Calculate needed parameters
583 ! ------------------------------------------------------------
584 ! If IWCAP = 1, 2, or 5; Calculate C_K
585 ! If IWCAP = 4 or 5; Calcualte C_BJ
586 ! If IWCAP = 3; Calculate C_LH
587 ! If IWCAP = 6; Calculate cumulative steepness spectrum
588 ! IF IWCAP = 7; Calculate with Alves and Banner (2003)

590 ! For frequency dependent part of the spectrum

888 !
889  IMPLICIT NONE
890 !
891 ! 2. Purpose
892 !
893 ! Determine the bottom slope in upwave direction and calculate
894 ! the slope dependent breaking parameter according to Nelson (1987)
895 ! Note that Nelson (1987) is used here since in Nelson (1994a,1994b)
896 ! an error is present in the equation.
897 !
898 ! 3. Method
899 !
900 ! The breaker parameter is given by:
901 !
902 ! Hm / d = 0.55 + 0.88 exp ( -0.012 * cot beta)
903 !
904 ! with beta the angle the bed makes with the horizontal. This
905 ! above equation is only valid for positive slopes (Negative
906 ! slopes were not considered by Nelson. For very steep slopes
907 ! (>0.05 say) a very large breaker parameter is obtained (>>1).
908 !
909 ! To ensure wave breaking in laboratory cases (with very steep
910 ! slopes an upper limit of 0.81 (which corresponds to a bottom
911 ! slope of 0.01) is imposed on the model of Nelson.
912 !
913 ! For negative bottom slopes (not considered by Nelson) a value
914 ! op 0.73 is imposed (which is the average value in Table 2 of
915 ! Battjes and Janssen (1978).
916 !
917  REAL, INTENT(OUT) :: BRCOEF ! variable breaker coefficient
918 
919  REAL, INTENT(IN) :: SPCSIG(msc)
920  REAL, INTENT(IN) :: AC2(mdc,msc,0:mt) ! action densities
921  REAL, INTENT(IN) :: ECOS(mdc), ESIN(mdc) ! Cos and Sin of Theta
922  REAL, INTENT(IN) :: DEP2(mt) ! depths at grid points
923 
924 !!$ REAL, INTENT(IN) :: RDX(10), RDY(10)
925  INTEGER, INTENT(IN) :: IG

927 ! 9. STRUCTURE
928 !
929 ! ------------------------------------------------------------
930 ! Calculate total energy per direction for all frequencies
931 ! determine action density per direction weighted with cos/sin
932 ! determine mean propagation direction of energy
933 ! ------------------------------------------------------------
934 ! calculate the depth derivative in the mean wave direction
935 ! according to dd/ds (see also subroutine SPROSD)
936 ! calculate the slope dependend breaking coefficient
937 ! ------------------------------------------------------------
938 ! End of NELSON
939 ! -------------------------------------------------------------
940 !

942 !
943  INTEGER :: ID ,IS ! counters
944  INTEGER :: J,I1,I2
945  REAL(SP) :: F1
946 # if defined (SPHERICAL)
947  REAL(DP) XTMP,XTMP1

950 !

953 ! *** determine the average wave direction ***
954 !

958  DO id = 1, mdc
959  ead = 0.

961  sigma1 = spcsig(is)

972  sindir = eey / etots
973  ELSE

975  sindir = 0.
976  END IF

978 ! *** Determine bottom slope in average wave propagation direction ***
979 !
980 ! DDDX = RDX(1) * (DEP2(KCGRD(1)) - DEP2(KCGRD(2))) &
981 ! + RDX(2) * (DEP2(KCGRD(1)) - DEP2(KCGRD(3)))
982 ! DDDY = RDY(1) * (DEP2(KCGRD(1)) - DEP2(KCGRD(2))) &

989  i2 = nbsn(ig,j+1)

991 # if defined (SPHERICAL)

995  xtmp1 = vx(i2)-vx(i1)

997  xtmp = -360.0_sp*tpi+xtmp
998  ELSE IF(xtmp1 < -180.0_sp)THEN
999  xtmp = 360.0_sp*tpi+xtmp

1001  dddy=dddy+f1*xtmp*cos(deg2rad*vy(ig))
1002 # else
1003  dddx = dddx + f1*(vy(i1)-vy(i2))
1004  dddy = dddy + f1*(vx(i2)-vx(i1))
1005 # endif
1006  END DO
1007  dddx = dddx/art2(ig)
1008  dddy = dddy/art2(ig)
1009 
1010 !
1011  ddds = -1. * ( dddx * cosdir + dddy * sindir )
1012 !
1013 ! *** calculate breaking coefficient according to Nelson (1987) ***
1014 !
1015  IF(ddds >= 0.)THEN

1021  END IF

1023 ! PSURF(2) = BRKVAR deleted
1024 ! PSURF(2) is no longer used to transmit br. coefficient
1025
