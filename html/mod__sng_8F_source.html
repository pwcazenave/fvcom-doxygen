142  integer function ftn_opt_lng_get(sng)
143  ! Purpose: Return length of option string preceding space or equals character
144  ! Any preceding dashes are NOT counted towards option length
145  ! Fortran intrinsic len(sng) is returned if string is not NUL-terminated
146  ! Prototype:
147  ! integer ftn_opt_lng_get ! [nbr] Length of option
148  ! Usage:
149  ! Call ftn_opt_lng_get with the full option string, e.g.,
150  ! ftn_opt_lng_get('--dbg_lvl=5') and ftn_opt_lng_get will return the
151  ! length of the option string
152  ! opt_sng=arg_val(3:2+ftn_opt_lng_get(arg_val)) ! [sng] Option string
153  ! if (opt_sng == 'dbg') then ...
154  implicit none
155  ! Input
156  character(len=*),intent(in)::sng
157  ! Local
158  integer idx
159  integer len_sng
160  integer::dsh_nbr=0 ! [nbr] Number of dashes
161  ! Main Code
162  ! Length of option string is one less than position of first space or equal sign
163  len_sng=len(sng)
164  if (len_sng >= 1) then
165  if (sng(1:1) == '-') dsh_nbr=1 ! [nbr] Number of dashes
166  endif
167  if (len_sng >= 2) then
168  if (sng(1:2) == '--') dsh_nbr=2 ! [nbr] Number of dashes
169  endif
170  do idx=1,len_sng
171  ! Look for first abnormal character
172  if (sng(idx:idx) == ' ' .or. sng(idx:idx) == '=') goto 100
173  end do ! end loop over characters
174 100 continue
175  ftn_opt_lng_get=idx-1-dsh_nbr
176  ! Strings initialized to '' should have length 0
177  if (ftn_opt_lng_get < 0 .or. sng == '') ftn_opt_lng_get=0
178  return
179  end function ftn_opt_lng_get ! end ftn_opt_lng_get()

253  ! Notes common to all arg_get() routines:
254  ! arg_val is assumed to be present as a command-line line option
255  ! On input, first two characters of arg_val are assumed to be dashes,
256  ! although this would not be too difficult to change.
257  ! Routine accepts options specified in POSIX format: "--opt_sng val" or "--opt_sng=val" (preferred)
258  ! Case 1: Option string and value are separated by space " " conjunction
259  ! On input, arg_val is assumed to hold option string with dashes, e.g., "--opt_sng"
260  ! On output, arg_val holds corresponding argument, e.g., "73"
261  ! Routine uses getarg() to obtain arg_val, then increments arg_idx
262  ! Case 2: Option string and value are separated by equals "=" conjunction
263  ! On input and output, arg_val holds entire option and value, e.g., "--opt_sng=73"
264  ! Routines do not call getarg() to change arg_val and do not increment arg_idx
265  ! Routines set optional logical argument opt_flg to true if called

267  subroutine ftn_arg_get_dbl( & ! [sbr] Process double-valued command-line argument
268  arg_idx, & ! I/O [idx] Argument counter
269  arg_val, & ! I/O [sng] Double to copy into opt_val
270  opt_val, & ! O [frc] Variable to receive copy of arg_val
271  opt_flg) ! O [frc] Variable set by command-line
272  ! Purpose: Copy command-line double argument arg_val into variable opt_val
273  ! Usage:
274  ! call ftn_arg_get_dbl(arg_idx,arg_val,opt_val) ! [sbr] Process double-valued command-line argument
275  use mod_utils ! [mdl] Debugging constants, prg_nm, dbg_lvl
276  implicit none
277  ! Parameters
278  character(len=*),parameter::sbr_nm='ftn_arg_get_dbl' ! [sng] Subroutine name
279  ! Commons
280  ! Input
281  ! Input/Output
282  integer,intent(inout)::arg_idx ! I/O [idx] Argument counter
283  character(len=*),intent(inout)::arg_val ! I/O [sng] Double to copy into opt_val
284  ! Output
285  real(selected_real_kind(p=12)),intent(out)::opt_val ! O [frc] Variable to receive copy of arg_val
286  logical,optional,intent(out)::opt_flg ! O [flg] Variable set by command-line
287  ! Local
288  integer arg_lng ! [nbr] Length of argument
289  integer opt_lng ! [nbr] Length of option
290  integer arg_val_srt_idx ! [idx] Starting position of argument value
291  logical opt_cnj_is_spc ! [flg] Option conjunction is space character
292  ! Main Code
293  ! Print any diagnostics before current value of arg_val is overwritten
294  arg_lng=len_trim(arg_val) ! [nbr] Length of argument
295  opt_lng=ftn_opt_lng_get(arg_val) ! [nbr] Length of option
296  if (dbg_lvl >= dbg_io) then
297  write (6,'(2a,i2,3a,i2)') prg_nm(1:ftn_strlsc(prg_nm)), &
298  ': DEBUG '//sbr_nm//'() reports arg_idx = ',arg_idx, &
299  ', Full option = ''',arg_val(1:arg_lng),''', Length = ',arg_lng
300  endif ! endif dbg
301  ! Determine whether format is --opt_sng=val or --opt_sng val
302  ! Add two to cover preliminary dashes
303  if (opt_lng+2 < arg_lng) then
304  opt_cnj_is_spc=.false. ! [flg] Option conjunction is space character
305  arg_val_srt_idx=3+opt_lng+1 ! [idx] Starting position of argument value
306  if (dbg_lvl >= dbg_io) then
307  write (6,'(8a)') prg_nm(1:ftn_strlsc(prg_nm)), &
308  ': DEBUG '//sbr_nm//'() diassembles argument into Option = ''', &
309  arg_val(3:2+opt_lng),''', conjunction is ''', &
310  arg_val(3+opt_lng:3+opt_lng),''', argument = ''', &
311  arg_val(arg_val_srt_idx:arg_lng),'''' ! Quote syntax is mystical
312  endif ! endif dbg
313  else
314  opt_cnj_is_spc=.true. ! [flg] Option conjunction is space character
315  arg_val_srt_idx=1 ! [idx] Starting position of argument value
316  endif ! endif
317  if (opt_cnj_is_spc) then
318  ! Obtain value of option by getting next command-line argument
319  ! New arguments will alter input values of arg_idx,arg_val
320  call get_command_argument(arg_idx,arg_val)
321  ! Increment counter for next read
322  arg_idx=arg_idx+1
323  ! Sanity checks
324  arg_lng=len_trim(arg_val) ! [nbr] Length of argument
325  if (arg_lng <= 0) stop 'ftn_???_arg_get() reports option lacks argument'
326  endif ! endif opt_cnj_is_spc
327  ! Read argument into double and return
328  read (arg_val(arg_val_srt_idx:arg_lng),*) opt_val ! [frc] Variable to receive copy of arg_val
329  if (dbg_lvl >= dbg_io) then
330  write (6,'(2a,g12.6)') prg_nm(1:ftn_strlsc(prg_nm)), &
331  ': DEBUG '//sbr_nm//'() assigned argument value = ',opt_val
332  endif ! endif
333  if(present(opt_flg)) opt_flg=.true. ! [flg] Variable set by command-line
334  return
335  end subroutine ftn_arg_get_dbl
