52 ! !ROUTINE: mixed_layer - compute SST and freeze/melt potential
53 !
54 ! !DESCRIPTION:
55 !
56 ! Compute the mixed layer heat balance and update the SST.
57 ! Compute the energy available to freeze or melt ice.
58 ! NOTE: SST changes due to fluxes through the ice are computed in
59 ! ice_therm_vertical.
60 !
61 ! !REVISION HISTORY:
62 !
63 ! authors: John Weatherly, CRREL
64 ! C.M. Bitz, UW
65 ! Elizabeth C. Hunke, LANL
66 ! Bruce P. Briegleb, NCAR
67 ! William H. Lipscomb, LANL
68 !
69 ! !INTERFACE:
70 !
71  subroutine mixed_layer
72 !
73 ! !USES:
74 !
75  use ice_flux
76  use ice_grid, only: tmask
77  use ice_atmo
78  use ice_state
79  use ice_albedo
80 !
81 ! !INPUT/OUTPUT PARAMETERS:
82 !
83 !EOP
84 !
85  integer (kind=int_kind) :: &
86  i, j ! horizontal indices
87 
88  real (kind=dbl_kind), dimension(ilo:ihi,jlo:jhi) ::&
89  delt & ! potential temperature difference (K)
90  , delq & ! specific humidity difference (kg/kg)
91  , dummy1, dummy2, dummy3, dummy4 ! dummy arrays
92 
93  real (kind=dbl_kind) :: &
94  TsfK &! surface temperature (K)
95  , fsh &! sensible heat flux (W/m^2)
96  , flh &! latent heat flux (W/m^2)
97  , swabs &! surface absorbed shortwave heat flux (W/m^2)
98  , flwup &! long-wave upward heat flux (W/m^2)
99  , ft &! fraction reduction of positive qdp
100  , dtcprho ! dt/cp_ocn*rhow
101 
102  dtcprho = dtice/(cp_ocn*rhow)
103 
104  call atmo_boundary_layer (1, 'ocn', sst, &
105  dummy1, dummy2, dummy3, dummy4, delt, delq)
106 
107  do j = jlo,jhi
108  do i = ilo,ihi
109  if (tmask(i,j)) then
110 ! if (hmix(i,j) > puny) then
111 
112  ! ocean surface temperature in Kelvin
113  tsfk = sst(i,j) + tffresh
114 
115  ! shortwave radiative flux
116  swabs = (c1i - albocn) * fsw(i,j)
117 
118  ! longwave radiative flux
119  flwup = -stefan_boltzmann * tsfk**4
120 
121  ! downward latent and sensible heat fluxes
122  flh = lhcoef(i,j) * delq(i,j)
123  fsh = shcoef(i,j) * delt(i,j)
124 
125  ! compute sst change due to exchange with atm/ice above
126  ! Note: fhnet, fswthru are added in ice_therm_vertical.F
127  sst(i,j) = sst(i,j) + (c1i-aice(i,j))*dtcprho/hmix(i,j) &
128  *(fsh + flh + flwup + flw(i,j) + swabs)
129 
130  ! adjust qdp if cooling of mixed layer would occur when sst le Tf
131  if( sst(i,j) <= tf(i,j) .and. qdp(i,j) > c0i ) qdp(i,j) = c0i
132 
133  ! computed T change due to exchange with deep layers:
134  sst(i,j) = sst(i,j) - qdp(i,j)*dtcprho/hmix(i,j)
135 
136  ! compute potential to freeze or melt ice
137  frzmlt(i,j) = (tf(i,j)-sst(i,j))/dtcprho*hmix(i,j)
138  frzmlt(i,j) = min(max(frzmlt(i,j),-c1000),c1000)
139 
140  ! if sst is below freezing, reset sst to Tf
141  if (sst(i,j) <= tf(i,j)) sst(i,j) = tf(i,j)
142 
143  else
144 
145  frzmlt(i,j) = c0i
146  sst(i,j) = tf(i,j)
147 
148 ! endif ! hmix > puny
149  endif ! tmask
150  enddo ! i
151  enddo ! j
152 
153  end subroutine mixed_layer
