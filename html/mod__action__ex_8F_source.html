292  SUBROUTINE tridag(A,B,C,R,U,N)
293  IMPLICIT NONE
294  INTEGER :: N,J
295  REAL,DIMENSION(N) :: A,B,C,R,U
296  INTEGER, PARAMETER :: NMAX = 500

298 
299  IF(b(1) == 0.)pause 'TRIDAG: REWRITE EQUATIONS'
300  bet = b(1)
301  u(1) = r(1)/bet
302  DO j=2,n
303  gam(j) = c(j-1)/bet
304  bet = b(j)-a(j)*gam(j)

306  u(j) = (r(j)-a(j)*u(j-1))/bet

309  u(j) = u(j)-gam(j+1)*u(j+1)
310  END DO
311 
312  RETURN
