272  SUBROUTINE swlta ( IG ,DEP2 ,CGO ,SPCSIG , &
273  kwave ,imatra ,imatda ,iddlow , &
274  iddtop ,isstop ,idcmin ,idcmax , &
275  hs ,smebrk ,pltri ,ursell )
276 ! (This subroutine has not been tested yet)
277 !
278 !****************************************************************
279 !
280  USE ocpcomm4
281  USE swcomm3
282  USE swcomm4
283  USE vars_wave, ONLY : mt, ac2
284 !
285  IMPLICIT NONE
286 !
287 ! 2. Purpose
288 !
289 ! In this subroutine the triad-wave interactions are calculated
290 ! with the Lumped Triad Approximation of Eldeberky (1996). His
291 ! expression is based on a parametrization of the biphase (as
292 ! function of the Ursell number), is directionally uncoupled and
293 ! takes into account for the self-self interactions only.
294 !
295 ! For a full description of the equations reference is made
296 ! to PhD thesis of Eldeberky (1996). Here only the main expressions
297 ! are given.
298 !
299 ! 3. Method
300 !
301 ! The parametrized biphase is given by (see eq. 3.19):
302 !
303 ! 0.2
304 ! beta = - pi/2 + pi/2 tanh ( ----- )
305 ! Ur
306 !
307 ! The Ursell number is calculated in routine SINTGRL
308 !
309 ! The source term as function of frequency p is (see eq. 7.25):
310 !
311 ! + -
312 ! S(p) = S(p) + S(p)
313 !
314 ! in which
315 !
316 ! +
317 ! S(p) = alpha Cp Cg,p (R(p/2,p/2))**2 sin (|beta|) ( E(p/2)**2 -2 E(p) E(p/2) )
318 !
319 ! - +
320 ! S(p) = - 2 S(2p)
321 !
322 ! with alpha a tunable coefficient and R(p/2,p/2) is the interaction
323 ! coefficient of which the expression can be found in Eldeberky (1996);
324 ! see eq. 7.26.
325 !
326 ! Note that a slightly adapted formulation of the LTA is used in
327 ! in the SWAN model:
328 !
329 ! - Only positive contributions to higher harmonics are considered
330 ! here (no energy is transferred to lower harmonics).
331 !
332 ! - The mean frequency in the expression of the Ursell number
333 ! is calculated according to the first order moment over the
334 ! zeroth order moment (personal communication, Y.Eldeberky, 1997).
335 !
336 ! - The interactions are calculated up to 2.5 times the mean
337 ! frequency only.
338 !
339 ! - Since the spectral grid is logarithmically distributed in frequency
340 ! space, the interactions between central bin and interacting bin
341 ! are interpolated such that the distance between these bins is
342 ! factor 2 (nearly).
343 !
344 ! - The interactions are calculated in terms of energy density
345 ! instead of action density. So the action density spectrum
346 ! is firstly converted to the energy density grid, then the
347 ! interactions are calculated and then the spectrum is converted
348 ! to the action density spectrum back.
349 !
350 ! - To ensure numerical stability the Patankar rule is used.
351 !
352  INTEGER IDDLOW, IDDTOP, ISSTOP
353  INTEGER IDCMIN(msc), IDCMAX(msc)
354 
355  REAL :: HS, SMEBRK
356 ! REAL :: AC2(MDC,MSC,0:MT)
357 
358  REAL :: CGO(msc,micmax)
359  REAL :: DEP2(mt)
360  REAL :: IMATDA(mdc,msc), IMATRA(mdc,msc)
361  REAL :: SPCSIG(msc)
362  REAL :: KWAVE(msc,micmax)
363  REAL :: PLTRI(mdc,msc,nptst)
364  REAL :: URSELL(mt)
365 
366  INTEGER I1, I2, ID, IG, IDDUM, IENT, II, IS, ISM, ISM1, ISMAX, ISP, ISP1
367  REAL AUX1, AUX2, BIPH, C0, CM, DEP, DEP_2, DEP_3, E0, EM, &
368  FT, RINT, SIGPI, SINBPH, STRI, WISM, WISM1, WISP, WISP1, &
369  W0, WM, WN0, WNM, XIS, XISLN
370  REAL, ALLOCATABLE :: E(:), SA(:,:)
371 
372 ! DEP = DEP2(IGC)
373  dep = dep2(ig)
374  dep_2 = dep**2
375  dep_3 = dep**3
376 !
377 ! --- compute some indices in sigma space
378 !
379  i2 = int(float(msc) / 2.)
380  i1 = i2 - 1
381  xis = spcsig(i2) / spcsig(i1)
382  xisln = log( xis )
383 
384  isp = int( log(2.) / xisln )
385  isp1 = isp + 1
386  wisp = (2. - xis**isp) / (xis**isp1 - xis**isp)
387  wisp1 = 1. - wisp
388 
389  ism = int( log(0.5) / xisln )
390  ism1 = ism - 1
391  wism = (xis**ism -0.5) / (xis**ism - xis**ism1)
392  wism1 = 1. - wism
393 
394  ALLOCATE (e(1:msc))
395  ALLOCATE (sa(1:mdc,1:msc+isp1))
396  e = 0.
397  sa = 0.
398 !
399 ! --- compute maximum frequency for which interactions are calculated
400 !
401  ismax = 1
402  DO is = 1, msc
403  IF(spcsig(is) < ( ptriad(2) * smebrk) )THEN
404  ismax = is
405  ENDIF
406  ENDDO
407  ismax = max( ismax , isp1 )
408 !
409 ! --- compute 3 wave-wave interactions
410 !
411 ! IF( URSELL(IGC) >= PTRIAD(5) )THEN
412  IF( ursell(ig) >= ptriad(5) )THEN
413 !
414 ! --- calculate biphase
415 !
416 ! BIPH = (0.5*PI_W)*(TANH(PTRIAD(4)/URSELL(IGC))-1.)
417  biph = (0.5*pi_w)*(tanh(ptriad(4)/ursell(ig))-1.)
418  sinbph = abs( sin(biph) )
419 !
420  DO ii = iddlow, iddtop
421  id = mod( ii - 1 + mdc , mdc ) + 1
422 !
423 ! --- initialize array with E(f) for the direction considered
424 !
425  DO is = 1, msc
426 ! E(IS) = AC2(ID,IS,IGC) * 2. * PI_W * SPCSIG(IS)
427  e(is) = ac2(id,is,ig) * 2. * pi_w * spcsig(is)
428  END DO
429 !
430  DO is = 1, ismax
431 
432  e0 = e(is)
433  w0 = spcsig(is)
434  wn0 = kwave(is,1)
435  c0 = w0 / wn0
436 
437  IF( is > -ism1 )THEN
438  em = wism * e(is+ism1) + wism1 * e(is+ism)
439  wm = wism * spcsig(is+ism1) + wism1 * spcsig(is+ism)
440  wnm = wism * kwave(is+ism1,1) + wism1 * kwave(is+ism,1)
441  cm = wm / wnm
442  ELSE
443  em = 0.
444  wm = 0.
445  wnm = 0.
446  cm = 0.
447  END IF
448 
449  aux1 = wnm**2 * ( grav_w * dep + 2.*cm**2 )
450  aux2 = wn0 * dep * ( grav_w * dep + &
451  (2./15.) * grav_w * dep_3 * wn0**2 - &
452  (2./ 5.) * w0**2 * dep_2 )
453  rint = aux1 / aux2
454  ft = ptriad(1) * c0 * cgo(is,1) * rint**2 * sinbph
455 
456  sa(id,is) = max(0., ft * ( em * em - 2. * em * e0 ))
457 
458  END DO
459  END DO
460 !
461 ! --- put source term together
462 !
463  DO is = 1, isstop
464  sigpi = spcsig(is) * 2. * pi_w
465  DO iddum = idcmin(is), idcmax(is)
466  id = mod( iddum - 1 + mdc , mdc ) + 1
467 !
468  stri = sa(id,is) - 2.*(wisp * sa(id,is+isp1) + &
469  wisp1 * sa(id,is+isp ))
470 !
471 ! --- store results in rhs and main diagonal according
472 ! to Patankar-rules
473 !
474  IF(testfl) pltri(id,is,iptst) = stri / sigpi
475  IF(stri > 0.)THEN
476  imatra(id,is) = imatra(id,is) + stri / sigpi
477  ELSE
478  imatda(id,is) = imatda(id,is) - stri / &
479 ! MAX(1.E-18,AC2(ID,IS,IGC)*SIGPI)
480  max(1.e-18,ac2(id,is,ig)*sigpi)
481  END IF
482  END DO
483  END DO
484 
485  END IF
486 
487  DEALLOCATE(e,sa)
488 
489  RETURN
490  END SUBROUTINE swlta

651  SUBROUTINE swsnl8 (WWINT ,UE ,SA1 ,SA2 ,SPCSIG , &
652  snlc1 ,dal1 ,dal2 ,dal3 ,sfnl , &

656 !*********************************************************************
657 !
658  USE swcomm3
659  USE swcomm4
660  USE ocpcomm4
661  USE m_snl4
662 ! USE ALL_VARS, ONLY : MT,AC2
663  USE vars_wave, ONLY : mt,ac2
664 !
665  IMPLICIT NONE
666 !
667 ! 2. Purpose
668 !
669 ! Calculate non-linear interaction using the discrete interaction
670 ! approximation (Hasselmann and Hasselmann 1985; WAMDI group 1988)
671 ! for the full circle.
672 !
673 ! 3. Method
674 !
675 ! Discrete interaction approximation. To make interpolation simple,
676 ! the interactions are calculated in a "folded" space.
677 !
678 ! Frequencies -->
679 ! +---+---------------------+---------+- IDHGH
680 ! d | 3 : 2 : 2 |
681 ! i + - + - - - - - - - - - - + - - - - +- MDC
682 ! r | : : |
683 ! e | 3 : original spectrum : 1 |
684 ! c | : : |
685 ! t. + - + - - - - - - - - - - + - - - - +- 1
686 ! | 3 : 2 : 2 |
687 ! +---+---------------------+---------+- IDLOW
688 ! | | | ^ |
689 ! ISLOW 1 MSC | ISHGH
690 ! | |
691 ! ISCLW ISCHG
692 ! lowest discrete highest discrete
693 ! central bin central bin
694 !
695 ! 1 : Extra tail added beyond MSC
696 ! 2 : Spectrum copied outside ID range
697 ! 3 : Empty bins at low frequencies
698 !
699 ! ISLOW = 1 + ISM1
700 ! ISHGH = MSC + ISP1 - ISM1
701 ! ISCLW = 1
702 ! ISCHG = MSC - ISM1
703 ! IDLOW = 1 - MAX(IDM1,IDP1)
704 ! IDHGH = MDC + MAX(IDM1,IDP1)
705 !
706 ! Note: using this subroutine requires an additional array
707 ! with size MXC*MYC*MDC*MSC.
708 !
709  INTEGER WWINT(*)
710 !
711  REAL DAL1, DAL2, DAL3, FACHFR, KMESPC, SNLC1
712 ! REAL AC2(MDC,MSC,0:MT)
713  REAL DEP2(mt)
714  REAL MEMNL4(mdc,msc,mt)
715  REAL SA1(msc4mi:msc4ma,mdc4mi:mdc4ma)
716  REAL SA2(msc4mi:msc4ma,mdc4mi:mdc4ma)
717  REAL SFNL(msc4mi:msc4ma,mdc4mi:mdc4ma)
718  REAL SPCSIG(msc)
719  REAL UE(msc4mi:msc4ma,mdc4mi:mdc4ma)
720 !
721 !*******************************************************************
722 !
723  INTEGER IS ,ID ,ID0 ,I ,J , &
724  ISLOW ,ISHGH ,IDLOW ,IDHGH , &
725  IDPP ,IDMM ,ISPP ,ISMM , &
726  ISCLW ,ISCHG ,IDDUM ,IENT
727 !
728  REAL X ,X2 ,CONS ,FACTOR ,SNLCS1 ,SNLCS2 , &
729  SNLCS3 ,E00 ,EP1 ,EM1 ,EP2 ,EM2 , &
730  SA1A ,SA1B ,SA2A ,SA2B , &
731  JACOBI ,SIGPI
732 !
733 ! SAVE IENT
734 ! DATA IENT/0/
735 ! IF (LTRACE) CALL STRACE (IENT,'SWSNL8')
736 !
737  islow = wwint(9)
738  ishgh = wwint(10)
739  isclw = wwint(11)
740  ischg = wwint(12)
741  idlow = wwint(13)
742  idhgh = wwint(14)
743  idpp = wwint(21)
744  idmm = wwint(22)
745  ispp = wwint(23)
746  ismm = wwint(24)
747 !
748 ! *** Calculate prop. constant. ***
749 ! *** Calculate factor R(X) to calculate the NL wave-wave ***
750 ! *** interaction for shallow water ***
751 ! *** SNLC1 = 1/GRAV**4 ***
752 !
753  snlcs1 = pquad(3)
754  snlcs2 = pquad(4)
755  snlcs3 = pquad(5)
756 ! X = MAX ( 0.75 * DEP2(IGC) * KMESPC , 0.5 )
757  x = max( 0.75 * dep2(kcgrd(1)) * kmespc , 0.5 )
758  x2 = max( -1.e15, snlcs3*x)
759  cons = snlc1 * ( 1. + snlcs1/x * (1.-snlcs2*x) * exp(x2))
760  jacobi = 2. * pi_w
761 !
762 ! *** extend the area with action density at periodic boundaries ***
763 !
764  DO iddum = idlow, idhgh
765  id = mod( iddum - 1 + mdc , mdc ) + 1
766  DO is=1, msc
767 ! UE (IS,IDDUM) = AC2(ID,IS,IGC) * SPCSIG(IS) * JACOBI
768  ue(is,iddum) = ac2(id,is,kcgrd(1)) * spcsig(is) * jacobi
769  ENDDO
770  ENDDO
771 !
772  DO id = idlow, idhgh
773  DO is = msc+1, ishgh
774  ue(is,id) = ue(is-1,id) * fachfr
775  ENDDO
776  ENDDO
777 !
778 ! *** Calculate (unfolded) interactions ***
779 ! *** Energy at interacting bins ***
780 !
781  DO id = 1, mdc
782  DO is = isclw, ischg
783  e00 = ue(is ,id )
784  ep1 = ue(is+ispp,id+idpp)
785  em1 = ue(is+ismm,id-idmm)
786  ep2 = ue(is+ispp,id-idpp)
787  em2 = ue(is+ismm,id+idmm)
788 !
789 ! Contribution to interactions
790 !
791  factor = cons * af11(is) * pquad(2) * e00
792 !
793  sa1a = e00 * ( ep1*dal1 + em1*dal2 )
794  sa1b = sa1a - ep1*em1*dal3
795  sa2a = e00 * ( ep2*dal1 + em2*dal2 )
796  sa2b = sa2a - ep2*em2*dal3
797 !
798  sa1(is,id) = factor * sa1b
799  sa2(is,id) = factor * sa2b
800 !
801  ENDDO
802  ENDDO
803 !
804 ! *** Fold interactions to side angles -> domain in theta is ***
805 ! *** periodic ***
806 !
807  DO id = 1, idhgh - mdc
808  id0 = 1 - id
809  DO is = isclw, ischg
810  sa1(is,mdc+id) = sa1(is, id )
811  sa2(is,mdc+id) = sa2(is, id )
812  sa1(is, id0 ) = sa1(is,mdc+id0)
813  sa2(is, id0 ) = sa2(is,mdc+id0)
814  ENDDO
815  ENDDO
816 !
817 ! *** Put source term together (To save space I=IS and ***
818 ! *** J=MDC is used) ---- ***
819 !
820  DO i = 1, msc
821  sigpi = spcsig(i) * jacobi
822  DO j = 1, mdc
823  sfnl(i,j) = - 2. * ( sa1(i,j) + sa2(i,j) ) &
824  + ( sa1(i-ispp,j-idpp) + sa2(i-ispp,j+idpp) ) &
825  + ( sa1(i-ismm,j+idmm) + sa2(i-ismm,j-idmm) )
826 !
827 ! *** store value in auxiliary array and use values in ***
828 ! *** next four sweeps (see subroutine FILNL3) ***
829 !
830 ! MEMNL4(J,I,IGC) = SFNL(I,J) / SIGPI
831  memnl4(j,i,kcgrd(1)) = sfnl(i,j) / sigpi
832  ENDDO
833  ENDDO
834 !
835 ! *** value source term in every bin ***
836 !
837 ! IF(ITEST >= 150 .AND. TESTFL)THEN
838 ! DO I=1, MSC
839 ! DO J=1, MDC
840 ! WRITE(PRINTF,2006) I,J,MEMNL4(J,I,KCGRD(1)),SFNL(I,J),SPCSIG(I)
841 !2006 FORMAT (' I J MEMNL() SFNL() SPCSIG:',2I4,3E12.4)
842 ! ENDDO
843 ! ENDDO
844 ! END IF
845 !
846  RETURN
847 !
848  END SUBROUTINE swsnl8
