104 SUBROUTINE watch_init(MYWATCH)
105  IMPLICIT NONE
106  TYPE(watch) :: MYWATCH
107 
108  mywatch%COUNT_RATE = 0
109  mywatch%COUNT_MAX = 0
110  mywatch%COUNT_CURRENT = 0
111  mywatch%COUNT_LAST = 0
112  mywatch%COUNT_0 = 0
113  mywatch%Lap_Count = 0
114 
115  CALL system_clock (mywatch%COUNT_CURRENT, mywatch%COUNT_RATE, mywatch%COUNT_MAX)
116 
117  mywatch%COUNT_0 = mywatch%COUNT_CURRENT
118  mywatch%COUNT_LAST = mywatch%COUNT_CURRENT
119 
120  IF(mywatch%COUNT_MAX == 0) CALL fatal_error &
121  & ("MOD_CLOCK: FORTRAN DOES NOT RECOGNIZE YOUR SYSTEM CLOCK!",&
122  & "INTIRINISIC, SYSTEM_CLOCK returned COUNT_MAX==0")
123 
124  IF(mywatch%COUNT_RATE .LE. 0) CALL fatal_error &
125  & ("MOD_CLOCK: FORTRAN DOES NOT RECOGNIZE YOUR SYSTEM CLOCK!",&
126  & .LE."INTIRINISIC, SYSTEM_CLOCK returned COUNT_RATE 0")
127 
128  IF(mywatch%COUNT_0 .LT. 0) CALL fatal_error &
129  & ("MOD_CLOCK: FORTRAN DOES NOT RECOGNIZE YOUR SYSTEM CLOCK!",&
130  & .LT."INTIRINISIC, SYSTEM_CLOCK returned COUNT 0")
131 
132 END SUBROUTINE watch_init
