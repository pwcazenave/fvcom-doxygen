88 
89  qmean1 = 1.e-8
90 
91 # if defined (SEMI_IMPLICIT)
92  ALLOCATE(uq1(0:nt,kb)); uq1 = 0.0_sp
93  ALLOCATE(vq1(0:nt,kb)); vq1 = 0.0_sp
94 # else
95  ALLOCATE(uq1(0,0))
96  ALLOCATE(vq1(0,0))
97 # endif
98 
99  SELECT CASE(horizontal_mixing_type)
100  CASE ('closure')
101  fact = 1.0_sp
102  fm1 = 0.0_sp
103  CASE('constant')
104  fact = 0.0_sp
105  fm1 = 1.0_sp
106  CASE DEFAULT
107  CALL fatal_error("UNKNOW HORIZONTAL MIXING TYPE:",&
108  & trim(horizontal_mixing_type) )
109  END SELECT
110 
111 !
112 !--Initialize Fluxes-----------------------------------------------------------!
113 !
114  qf = 0.0_sp
115  xflux = 0.0_sp
116 
117  uq = 0.0_sp
118  vq = 0.0_sp
119  uvn = 0.0_sp
120 
121 # if defined (SEMI_IMPLICIT)
122  uvn1 = 0.0_sp
123 # endif
124 
125  DO k=2,kbm1
126  DO i=1,nt
127  uq(i,k) = (u(i,k)*dz1(i,k-1)+u(i,k-1)*dz1(i,k))/(dz1(i,k)+dz1(i,k-1))
128  vq(i,k) = (v(i,k)*dz1(i,k-1)+v(i,k-1)*dz1(i,k))/(dz1(i,k)+dz1(i,k-1))
129 # if defined (SEMI_IMPLICIT)
130  uq1(i,k) = (uf(i,k)*dz1(i,k-1)+uf(i,k-1)*dz1(i,k))/(dz1(i,k)+dz1(i,k-1))
131  vq1(i,k) = (vf(i,k)*dz1(i,k-1)+vf(i,k-1)*dz1(i,k))/(dz1(i,k)+dz1(i,k-1))
132 # endif
133  END DO
134  END DO
135 
136 !
137 !--Loop Over Control Volume Sub-Edges And Calculate Normal Velocity------------!
138 !
139  DO i=1,ncv
140  i1=ntrg(i)
141 ! DTIJ(I)=DT1(I1)
142  DO k=2,kbm1
143  dtij(i,k)=dt1(i1)*dzz1(i1,k-1)
144  uvn(i,k) = vq(i1,k)*dltxe(i) - uq(i1,k)*dltye(i)
145 
146 # if defined (PLBC)
147  uvn(i,k) = 0.0_sp*dltxe(i) - uq(i1,k)*dltye(i)
148 # endif
149 
150 # if defined (SEMI_IMPLICIT)
151  dtij1(i,k)=d1(i1)*dzz1(i1,k-1)
152  uvn1(i,k) = vq1(i1,k)*dltxe(i) - uq1(i1,k)*dltye(i)
153 # endif
154  END DO
155  END DO
156 
157 !
158 !--Calculate the Advection and Horizontal Diffusion Terms----------------------!
159 !
160 
161  DO k=2,kbm1
162  pqpx = 0.0_sp
163  pqpy = 0.0_sp
164  pqpxd = 0.0_sp
165  pqpyd = 0.0_sp
166  DO i=1,m
167  DO j=1,ntsn(i)-1
168  i1=nbsn(i,j)
169  i2=nbsn(i,j+1)
170 
171 ! FFD=0.5_SP*(Q(I1,K)+Q(I2,K)-QMEAN1(I1,K)-QMEAN1(I2,K))
172  ffd=0.5_sp*(q(i1,k)+q(i2,k)-qmean1-qmean1)
173  ff1=0.5_sp*(q(i1,k)+q(i2,k))
174 
175 
176 !!$# if defined (SPHERICAL)
177 !!$ XTMP = VX(I2)*TPI-VX(I1)*TPI
178 !!$ XTMP1 = VX(I2)-VX(I1)
179 !!$ IF(XTMP1 > 180.0_SP)THEN
180 !!$ XTMP = -360.0_SP*TPI+XTMP
181 !!$ ELSE IF(XTMP1 < -180.0_SP)THEN
182 !!$ XTMP = 360.0_SP*TPI+XTMP
183 !!$ END IF
184 !!$ TXPI=XTMP*COS(DEG2RAD*VY(I))
185 !!$ TYPI=(VY(I1)-VY(I2))*TPI
186 !!$
187 !!$ ! ERROR HERE
188 !!$ IF(NODE_NORTHAREA(I) == 1)THEN
189 !!$ VX1_TMP = REARTH * COS(VY(I1)*DEG2RAD) * COS(VX(I1)*DEG2RAD) &
190 !!$ * 2._SP /(1._SP+sin(VY(I1)*DEG2RAD))
191 !!$ VY1_TMP = REARTH * COS(VY(I1)*DEG2RAD) * SIN(VX(I1)*DEG2RAD) &
192 !!$ * 2._SP /(1._SP+sin(VY(I1)*DEG2RAD))
193 !!$
194 !!$ VX2_TMP = REARTH * COS(VY(I2)*DEG2RAD) * COS(VX(I2)*DEG2RAD) &
195 !!$ * 2._SP /(1._SP+sin(VY(I2)*DEG2RAD))
196 !!$ VY2_TMP = REARTH * COS(VY(I2)*DEG2RAD) * SIN(VX(I2)*DEG2RAD) &
197 !!$ * 2._SP /(1._SP+sin(VY(I2)*DEG2RAD))
198 !!$
199 !!$ TXPI = (VX2_TMP-VX1_TMP)/(2._SP /(1._SP+sin(VY(I)*DEG2RAD)))
200 !!$ TYPI = (VY1_TMP-VY2_TMP)/(2._SP /(1._SP+sin(VY(I)*DEG2RAD)))
201 !!$ IF(I /= NODE_NORTHPOLE)THEN
202 !!$ TXPI_TMP = TYPI*COS(VX(I)*DEG2RAD)-TXPI*SIN(VX(I)*DEG2RAD)
203 !!$ TYPI_TMP = TXPI*COS(VX(I)*DEG2RAD)+TYPI*SIN(VX(I)*DEG2RAD)
204 !!$ TYPI_TMP = -TYPI_TMP
205 !!$
206 !!$ TXPI = TXPI_TMP
207 !!$ TYPI = TYPI_TMP
208 !!$ END IF
209 !!$ END IF
210 !!$ ! END ERROR
211 !!$
212 !!$
213 !!$ PQPX(I)=PQPX(I)+FF1*TYPI
214 !!$ PQPY(I)=PQPY(I)+FF1*TXPI
215 !!$ PQPXD(I)=PQPXD(I)+FFD*TYPI
216 !!$ PQPYD(I)=PQPYD(I)+FFD*TXPI
217 !!$# else
218 !!$ PQPX(I)=PQPX(I)+FF1*(VY(I1)-VY(I2))
219 !!$ PQPY(I)=PQPY(I)+FF1*(VX(I2)-VX(I1))
220 !!$ PQPXD(I)=PQPXD(I)+FFD*(VY(I1)-VY(I2))
221 !!$ PQPYD(I)=PQPYD(I)+FFD*(VX(I2)-VX(I1))
222 !!$# endif
223 
224  pqpx(i)=pqpx(i)+ff1*dltytrie(i,j)
225  pqpy(i)=pqpy(i)+ff1*dltxtrie(i,j)
226  pqpxd(i)=pqpxd(i)+ffd*dltytrie(i,j)
227  pqpyd(i)=pqpyd(i)+ffd*dltxtrie(i,j)
228 
229 
230  END DO
231  pqpx(i)=pqpx(i)/art2(i)
232  pqpy(i)=pqpy(i)/art2(i)
233  pqpxd(i)=pqpxd(i)/art2(i)
234  pqpyd(i)=pqpyd(i)/art2(i)
235  END DO
236 
237 # if defined (PLBC)
238  CALL replace_node(pqpx)
239  CALL replace_node(pqpxd)
240  CALL replace_node(pqpy)
241  CALL replace_node(pqpyd)
242 # endif
243 
244 
245  DO i=1,m
246  viscoff(i) = (viscofh(i,k)*dz(i,k-1)+viscofh(i,k-1)*dz(i,k))/ &
247  (dz(i,k)+dz(i,k-1))
248  END DO
249 
250  DO i=1,ncv_i
251  ia=niec(i,1)
252  ib=niec(i,2)
253 !!$ XI=0.5_SP*(XIJE(I,1)+XIJE(I,2))
254 !!$ YI=0.5_SP*(YIJE(I,1)+YIJE(I,2))
255 !!$# if defined (SPHERICAL)
256 !!$ X1_DP=XIJE(I,1)
257 !!$ Y1_DP=YIJE(I,1)
258 !!$ X2_DP=XIJE(I,2)
259 !!$ Y2_DP=YIJE(I,2)
260 !!$ CALL ARCC(X2_DP,Y2_DP,X1_DP,Y1_DP,XII,YII)
261 !!$ XI=XII
262 !!$ XTMP = XI*TPI-VX(IA)*TPI
263 !!$ XTMP1 = XI-VX(IA)
264 !!$ IF(XTMP1 > 180.0_SP)THEN
265 !!$ XTMP = -360.0_SP*TPI+XTMP
266 !!$ ELSE IF(XTMP1 < -180.0_SP)THEN
267 !!$ XTMP = 360.0_SP*TPI+XTMP
268 !!$ END IF
269 !!$
270 !!$ DXA=XTMP*COS(DEG2RAD*VY(IA))
271 !!$ DYA=(YI-VY(IA))*TPI
272 !!$ XTMP = XI*TPI-VX(IB)*TPI
273 !!$ XTMP1 = XI-VX(IB)
274 !!$ IF(XTMP1 > 180.0_SP)THEN
275 !!$ XTMP = -360.0_SP*TPI+XTMP
276 !!$ ELSE IF(XTMP1 < -180.0_SP)THEN
277 !!$ XTMP = 360.0_SP*TPI+XTMP
278 !!$ END IF
279 !!$
280 !!$ DXB=XTMP*COS(DEG2RAD*VY(IB))
281 !!$ DYB=(YI-VY(IB))*TPI
282 !!$# else
283 !!$ DXA=XI-VX(IA)
284 !!$ DYA=YI-VY(IA)
285 !!$ DXB=XI-VX(IB)
286 !!$ DYB=YI-VY(IB)
287 !!$# endif
288 !!$
289 !!$ FIJ1=Q(IA,K)+DXA*PQPX(IA)+DYA*PQPY(IA)
290 !!$ FIJ2=Q(IB,K)+DXB*PQPX(IB)+DYB*PQPY(IB)
291 
292  fij1=q(ia,k)+dltxncve(i,1)*pqpx(ia)+dltyncve(i,1)*pqpy(ia)
293  fij2=q(ib,k)+dltxncve(i,2)*pqpx(ib)+dltyncve(i,2)*pqpy(ib)
294 
295 
296  q1min=minval(q(nbsn(ia,1:ntsn(ia)-1),k))
297  q1min=min(q1min, q(ia,k))
298  q1max=maxval(q(nbsn(ia,1:ntsn(ia)-1),k))
299  q1max=max(q1max, q(ia,k))
300  q2min=minval(q(nbsn(ib,1:ntsn(ib)-1),k))
301  q2min=min(q2min, q(ib,k))
302  q2max=maxval(q(nbsn(ib,1:ntsn(ib)-1),k))
303  q2max=max(q2max, q(ib,k))
304  IF(fij1 < q1min) fij1=q1min
305  IF(fij1 > q1max) fij1=q1max
306  IF(fij2 < q2min) fij2=q2min
307  IF(fij2 > q2max) fij2=q2max
308 
309  un=uvn(i,k)
310 # if defined (SEMI_IMPLICIT)
311  un1=uvn1(i,k)
312 # endif
313 
314 ! VISCOF=HORCON*(FACT*0.5_SP*(VISCOFF(IA)+VISCOFF(IB))/HPRNU + FM1)
315 ! VISCOF=HORCON*(FACT*0.5_SP*(VISCOFF(IA)+VISCOFF(IB)) + FM1)
316 
317  ! David moved HPRNU and added HVC
318  viscof=(fact*0.5_sp*(viscoff(ia)*nn_hvc(ia)+viscoff(ib)*nn_hvc(ib)) + fm1*0.5_sp*(nn_hvc(ia)+nn_hvc(ib)))/hprnu
319 
320  txx=0.5_sp*(pqpxd(ia)+pqpxd(ib))*viscof
321  tyy=0.5_sp*(pqpyd(ia)+pqpyd(ib))*viscof
322 
323  fxx=-dtij(i,k)*txx*dltye(i)
324  fyy= dtij(i,k)*tyy*dltxe(i)
325 
326 # if defined (PLBC)
327  fyy= 0.0_sp
328 # endif
329 
330 # if !defined (SEMI_IMPLICIT)
331  exflux=-un*dtij(i,k)* &
332  ((1.0_sp+sign(1.0_sp,un))*fij2+(1.0_sp-sign(1.0_sp,un))*fij1)*0.5_sp+fxx+fyy
333 # else
334  exflux=-un*dtij(i,k)* &
335  ((1.0_sp+sign(1.0_sp,un))*fij2+(1.0_sp-sign(1.0_sp,un))*fij1)*0.5_sp
336  exflux=(1.0_sp-ifceta)*exflux+ifceta*(-un1*dtij1(i,k)*((1.0_sp+sign(1.0_sp,un1))*fij2+(1.0_sp-sign(1.0_sp,un1))*fij1)*0.5_sp)+fxx+fyy
337 # endif
338 
339  xflux(ia,k)=xflux(ia,k)+exflux
340  xflux(ib,k)=xflux(ib,k)-exflux
341 
342  END DO
343 
344 # if defined (SPHERICAL)
345 # if !defined (SEMI_IMPLICIT)
346  CALL adv_q_xy(xflux,pqpx,pqpy,pqpxd,pqpyd,viscoff,q,uq,vq,k,uq1,vq1,0.0_sp)
347 # else
348  CALL adv_q_xy(xflux,pqpx,pqpy,pqpxd,pqpyd,viscoff,q,uq,vq,k,uq1,vq1,ifceta)
349 # endif
350 # endif
351 
352  END DO !!SIGMA LOOP
353 
354 !
355 !-Accumulate Fluxes at Boundary Nodes
356 !
357 # if defined (MULTIPROCESSOR)
358  IF(par)CALL node_match(0,nbn,bn_mlt,bn_loc,bnc,mt,kb,myid,nprocs,xflux)
359 # endif
360 
361 
362 !--------------------------------------------------------------------
363 ! The central difference scheme in vertical advection
364 !--------------------------------------------------------------------
365  DO k=2,kbm1
366  DO i=1,m
367 # if defined (WET_DRY)
368  IF(iswetn(i)*iswetnt(i) == 1) THEN
369 # endif
370  temp=wts(i,k-1)*q(i,k-1)-wts(i,k+1)*q(i,k+1)
371  xflux(i,k)=xflux(i,k)+temp*art1(i)*dzz(i,k-1)/(dz(i,k-1)+dz(i,k))
372 # if defined (WET_DRY)
373  END IF
374 # endif
375  END DO
376  END DO !! SIGMA LOOP
377 
378 !
379 !--Update Q or QL-------------------------------------------------------------!
380 !
381 
382 # if defined (WET_DRY)
383  DO i=1,m
384  IF(iswetn(i)*iswetnt(i) == 1 )THEN
385  DO k=2,kbm1
386 # if !defined (SEMI_IMPLICIT)
387  qf(i,k)=(q(i,k)-xflux(i,k)/art1(i)*(dti/(dt(i)*dzz(i,k-1))))*(dt(i)/d(i))
388 # else
389  qf(i,k)=(te_tmp(i,k)-rk_te(stage)*xflux(i,k)/art1(i)*(dti/(dt(i)*dzz(i,k-1))))*(dt(i)/d(i))
390 # endif
391  END DO
392  ELSE
393  DO k=2,kbm1
394 # if !defined (SEMI_IMPLICIT)
395  qf(i,k)=q(i,k)
396 # else
397  qf(i,k)=te_tmp(i,k)
398 # endif
399  END DO
400  END IF
401  END DO
402 # else
403 
404  DO i=1,m
405  DO k=2,kbm1
406 # if !defined (SEMI_IMPLICIT)
407  qf(i,k)=(q(i,k)-xflux(i,k)/art1(i)*(dti/(dt(i)*dzz(i,k-1))))*(dt(i)/d(i))
408 # else
409  qf(i,k)=(te_tmp(i,k)-rk_te(stage)*xflux(i,k)/art1(i)*(dti/(dt(i)*dzz(i,k-1))))*(dt(i)/d(i))
410 # endif
411  END DO
412  END DO
413 # endif
414 
415  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "End: adv_q"
416 
417  END SUBROUTINE adv_q
