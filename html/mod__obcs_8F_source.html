223  ! DECIDE WHICH TIME YOU WANT TO USE!

225 # else
226  time_elapsed = inttime - spectime
227 # endif
228 
229 !! IN MY JUDGEMENT, A TIMESERIES FORCING DOES NOT MAKE SENSE FOR ATMO/EQUITIDE
230 
231 
232 ! SELECT CASE(TIDE_FORCING_TYPE)
233 ! CASE(TIDE_FORCING_TIMESERIES)
234  !
235  !-Julian: Set Elevation Based on Linear Interpolation Between Two Data Times-|
236  !
237 ! CALL FATAL_ERROR("ELEVATION_EQUI: Not set up for Julian Timeseries forcing yet?")
238 
239 ! CASE(TIDE_FORCING_SPECTRAL)

241  !-Non-Julian: Set Elevation of Equilibrium Tide -----------------------------|
242  !
243 
244  elf_eqi=0.0_sp
245 
246  IF(use_real_world_time) THEN
247 
248  time_adjust%mjd = 678942
249  time_adjust%musod = 0
250 
251  CALL now_2_month_days(inttime,year_cur,month_cur,mdays_cur)
252  IF(month_equi/=month_cur) THEN
253 
254  ic1 = year_cur/100
255  year_cur = year_cur - ic1*100
256  CALL gday1(16,month_cur,year_cur,ic1,kdm)
257 
258  time_16day%MJD = kdm
259  time_16day%MuSOD = 0
260  month_equi = month_cur
261 
262  hourm = dfloat(kdm)*24.0d0
263  equi_d1 = hourm/24.0d0
264  CALL gday1(31,12,99,18,kd0)
265  equi_d1 = equi_d1-dfloat(kd0)-0.5d0
266  call astro(equi_d1,equi_h,equi_pp,equi_s,equi_p,equi_enp,equi_dh,equi_dpp,equi_ds,equi_dp,equi_dnp)
267  intdays=idint(hourm/24.0d0)
268  equi_hh=hourm-dfloat(intdays*24)
269  equi_tau=equi_hh/24.0d0+equi_h-equi_s
270  dtau=365.0d0+equi_dh-equi_ds
271 
272  do j=1, ntidecomps
273  vdbl=equi_astr(equi_id(j))%ii*equi_tau+equi_astr(equi_id(j))%jj*equi_s+equi_astr(equi_id(j))%kk*equi_h &
274  +equi_astr(equi_id(j))%ll*equi_p+equi_astr(equi_id(j))%mm*equi_enp+equi_astr(equi_id(j))%nn*equi_pp &
275  +equi_astr(equi_id(j))%semi

278  equi_vn(j) = vdbl - equi_iv
279 
280  do i=1, mt
281 
282  slat = vy(i)
283  if(abs(slat).lt.5.0) slat=sign(5.0,slat)

285 
286  sumc = 1.0d0
287  sums = 0.0d0
288  do k=1, equi_astr(equi_id(j))%nj
289  equi_rr = equi_astr(equi_id(j))%ee(k)
290  equi_l = equi_astr(equi_id(j))%ir(k)+1
291  goto (901, 902, 903) equi_l
292  902 equi_rr = equi_astr(equi_id(j))%ee(k)*0.36309d0*(1.0d0-5.0d0*slat*slat)/slat
293  goto 901
294  903 equi_rr = equi_astr(equi_id(j))%ee(k)*2.59808d0*slat
295  901 continue
296  equi_uudbl = equi_astr(equi_id(j))%ldel(k)*equi_p+equi_astr(equi_id(j))%mdel(k)*equi_enp+equi_astr(equi_id(j))%ndel(k)*equi_pp+equi_astr(equi_id(j))%ph(k)
297  equi_iuu = int(equi_uudbl)
298  equi_uu = equi_uudbl - equi_iuu
299 
300  sumc = sumc + equi_rr*dcos(equi_uu*pi2)
301  sums = sums + equi_rr*dsin(equi_uu*pi2)
302  enddo
303 
304  equi_f(i,j)=dsqrt(sumc*sumc+sums*sums)
305  equi_u(i,j)=datan2(sums,sumc)/pi2
306  enddo
307 
308  enddo
309 
310  ENDIF
311 
312  DO j = 1,ntidecomps
313 
314  time_flt = seconds( (time_elapsed-(time_16day-spectime-time_adjust)) * pi2/period(j))
315  SELECT CASE(tide_type(j))
316  CASE(semidiurnal)
317 
318  DO i = 1, mt
319  phai_ij = lon(i)*pi2/360.0_sp+(equi_vn(j)+equi_u(i,j))*pi2/2.0_dp
320  elf_eqi(i) = elf_eqi(i) + equi_f(i,j)*beta_eqi(j)*apt_eqi(j)*cos(lat(i)*pi2/360.0_sp)**2 &
321  *dcos(time_flt+2.0_dp*phai_ij)
322  END DO
323  CASE(diurnal)
324 
325  DO i = 1, mt
326  phai_ij = lon(i)*pi2/360.0_sp+(equi_vn(j)+equi_u(i,j))*pi2
327  elf_eqi(i) = elf_eqi(i) + equi_f(i,j)*beta_eqi(j)*apt_eqi(j)*sin(2.0_sp*lat(i)*pi2/360.0_sp) &
328  *dcos(time_flt+phai_ij)
329  END DO
330 
331  CASE DEFAULT
332  CALL fatal_error("ELEVATION_EQUI: INVALID TIDE_TYPE:",&
333  &"Choices:"//trim(diurnal)//" or "//trim(semidiurnal) )
334  END SELECT
335  END DO
336 
337  ELSE
338 
339  DO j = 1,ntidecomps
340 
341  time_flt = seconds(time_elapsed * pi2/period(j))
342  SELECT CASE(tide_type(j))
343  CASE(semidiurnal)
344 
345  DO i = 1, mt
346  phai_ij = lon(i)*pi2/360.0_sp
347  elf_eqi(i) = elf_eqi(i) + beta_eqi(j)*apt_eqi(j)*cos(lat(i)*pi2/360.0_sp)**2 &
348  *dcos(time_flt+2.0_dp*phai_ij)
349  END DO
350  CASE(diurnal)
351 
352  DO i = 1, mt
353  phai_ij = lon(i)*pi2/360.0_sp
354  elf_eqi(i) = elf_eqi(i) + beta_eqi(j)*apt_eqi(j)*sin(2.0_sp*lat(i)*pi2/360.0_sp) &
355  *dcos(time_flt+phai_ij)

361  END SELECT
362  END DO
363 
364  ENDIF

369 ! CALL FATAL_ERROR("ELEVATION_EQUI: INVALID TIDE FORCING TYPE ?")
