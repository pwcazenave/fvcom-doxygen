30 ! this subroutine uses an implicit method to calculate the vertical !
31 ! diffusion term in the x and y momentum equations !
32 !==============================================================================|

34  SUBROUTINE vdif_uv
35 
36 !------------------------------------------------------------------------------|
37 
38  USE all_vars
39  USE mod_utils
40 # if defined (WET_DRY)
41  USE mod_wd
42 # endif
43 # if defined (SEMI_IMPLICIT)
44  USE mod_semi_implicit, only: stage, kstage_uv, uc, vc
45 # endif
46 # if defined (THIN_DAM)
47  USE mod_dam
48 # endif
49  IMPLICIT NONE
50  INTEGER I,K,ITMP1,ITMP2,ITMP3,KI
51  REAL(SP), DIMENSION(0:NT,KB) :: C,A,VHU,VHPU,VHV,VHPV
52  REAL(SP) :: TMP
53 
54  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: vdif_uv"
55 
56  c = km1
57 
58  DO k = 2, kbm1
59  DO i = 1, n
60 # if !defined (WET_DRY)
61  IF (d1(i) > 0.0_sp) THEN
62 # else
63  IF(iswetc(i) == 1)THEN
64 # endif
65 
66  a(i,k-1) = -dti *(c(i,k)+umol) / (dz1(i,k-1)*dzz1(i,k-1)* d1(i)*d1(i))
67  c(i,k) = -dti *(c(i,k)+umol) / (dz1(i,k)*dzz1(i,k-1)* d1(i)*d1(i))
68  END IF
69  END DO
70  END DO
71 
72  DO i = 1, n
73 # if !defined (WET_DRY)
74  IF (d1(i) > 0.0_sp) THEN
75 # else
76  IF(iswetc(i) == 1)THEN
77 # endif
78  itmp1=isonb(nv(i,1))
79  itmp2=isonb(nv(i,2))
80  itmp3=isonb(nv(i,3))
81  IF(itmp1 == 2 .OR. itmp2 == 2 .OR. itmp3 == 2) THEN
82  tmp=0.0_sp
83  ELSE
84  tmp=1.0_sp
85  END IF
86  vhu(i,1) = a(i,1) / (a(i,1)-1.)
87  vhv(i,1) = a(i,1) / (a(i,1)-1.)
88  vhpu(i,1) = (-dti*wusurf(i)*tmp/(-dz1(i,1)*d1(i))-uf(i,1)) / (a(i,1)-1.)
89  vhpv(i,1) = (-dti*wvsurf(i)*tmp/(-dz1(i,1)*d1(i))-vf(i,1)) / (a(i,1)-1.)
90  END IF
91  END DO
92 
93  DO k = 2, kbm2
94  DO i = 1, n
95 # if !defined (WET_DRY)
96  IF (d1(i) > 0.0_sp) THEN
97 # else
98  IF(iswetc(i) == 1)THEN
99 # endif
100  vhpu(i,k) = 1.0_sp / (a(i,k)+c(i,k)*(1.-vhu(i,k-1))-1.)
101  vhpv(i,k) = 1.0_sp / (a(i,k)+c(i,k)*(1.-vhv(i,k-1))-1.)
102  vhu(i,k) = a(i,k) * vhpu(i,k)
103  vhv(i,k) = a(i,k) * vhpv(i,k)
104  vhpu(i,k) = (c(i,k)*vhpu(i,k-1)-uf(i,k))*vhpu(i,k)
105  vhpv(i,k) = (c(i,k)*vhpv(i,k-1)-vf(i,k))*vhpv(i,k)
106  END IF
107  END DO
108  END DO
109 
110  DO i = 1, n
111 # if !defined (WET_DRY)
112  IF (d1(i) > 0.0_sp) THEN
113 # else
114  IF(iswetc(i) == 1)THEN
115 # endif
116 # if !defined (SEMI_IMPLICIT)
117  tps(i) = cbc(i) * sqrt(u(i,kbm1)**2+v(i,kbm1)**2)
118 # else
119  tps(i) = cbc(i) * sqrt(uc(i,kbm1)**2+vc(i,kbm1)**2)
120 # endif
121  uf(i,kbm1) = (c(i,kbm1)*vhpu(i,kbm2)-uf(i,kbm1))/ &
122  (tps(i)*dti/(-dz1(i,kbm1)*d1(i))-1.-(vhu(i,kbm2)-1.)*c(i,kbm1))
123  vf(i,kbm1) = (c(i,kbm1)*vhpv(i,kbm2)-vf(i,kbm1))/ &
124  (tps(i)*dti/(-dz1(i,kbm1)*d1(i))-1.-(vhv(i,kbm2)-1.)*c(i,kbm1))
125  END IF
126  END DO
127 
128  DO k = 2, kbm1
129  ki = kb - k
130  DO i = 1, n
131 # if !defined (WET_DRY)
132  IF (d1(i) > 0.0_sp) THEN
133 # else
134  IF(iswetc(i) == 1)THEN
135 # endif
136  uf(i,ki) = (vhu(i,ki)*uf(i,ki+1)+vhpu(i,ki))
137  vf(i,ki) = (vhv(i,ki)*vf(i,ki+1)+vhpv(i,ki))
138 # if defined (WET_DRY)
139  ELSE
140  uf(i,ki) = 0.0_sp
141  vf(i,ki) = 0.0_sp
142 # endif
143  END IF
144  END DO
145  END DO
146 
147 !
148 !--Damp Velocity In Sponge Region----------------------------------------------!
149 !
150  DO k = 1, kbm1
151  DO i = 1, n
152 !# if defined (THIN_DAM)
153 ! UF(I,K) = UF(I,K)-DAM_SPONGE(I)*UF(I,K)
154 ! VF(I,K) = VF(I,K)-DAM_SPONGE(I)*VF(I,K)
155 !# endif
156 !old: UF(I,K) = UF(I,K)-CC_SPONGE(I)*UF(I,K)
157 !old: VF(I,K) = VF(I,K)-CC_SPONGE(I)*VF(I,K)
158 ! ---- new: Karsten Lettmann: 2012.06.25 -------
159  uf(i,k) = uf(i,k)/(1.0_sp+cc_sponge(i)*uf(i,k)**2.0_sp)
160  vf(i,k) = vf(i,k)/(1.0_sp+cc_sponge(i)*vf(i,k)**2.0_sp)
161 ! ------- end new -------------------------------
162  END DO
163  END DO
164 
165 # if defined (SEMI_IMPLICIT)
166  IF(stage==kstage_uv) THEN
167 # endif
168  DO i = 1, n
169 # if !defined (WET_DRY)
170  IF (d1(i) > 0.0_sp) THEN
171 # else
172  IF(iswetc(i) == 1)THEN
173 # endif
174  wubot(i) = -tps(i) * u(i,kbm1)
175  wvbot(i) = -tps(i) * v(i,kbm1)
176 # if defined (WET_DRY)
177  ELSE
178  wubot(i) = 0.0_sp
179  wvbot(i) = 0.0_sp
180 # endif
181  END IF
182  END DO
183 # if defined (SEMI_IMPLICIT)
184  ENDIF
185 # endif
186 
187  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: vdif_uv"
188 
189  END SUBROUTINE vdif_uv
