82  SUBROUTINE arc_dbl(XX1,YY1,XX2,YY2,ARCL)
83  !----------------------------------------------------------------------------
84  ! function:
85  ! calculate the arc lenth for given two point on the spherical plane
86  ! input:
87  ! xx1,yy1,xx2,yy2 :are longitude and latitude of two points
88  ! output:
89  ! arcl : arc lenth of two points in spherical plane
90  !-----------------------------------------------------------------------------
91 
92  ! solve the arc length through the earth center
93  IMPLICIT NONE
94  REAL(DP) :: X1,Y1,X2,Y2,XA,YA,ZA,XB,YB,ZB,AB,AOB
95  REAL(DP),INTENT(OUT) :: ARCL
96  REAL(DP),INTENT(IN) :: XX1,YY1,XX2,YY2
97 
98  x1=xx1*deg2rad
99  y1=yy1*deg2rad
100 
101  x2=xx2*deg2rad
102  y2=yy2*deg2rad
103 
104  ! USE DOUBLE PRECISION COS AND SIN
105  xa=dcos(y1)*dcos(x1)
106  ya=dcos(y1)*dsin(x1)
107  za=dsin(y1)
108 
109  xb=dcos(y2)*dcos(x2)
110  yb=dcos(y2)*dsin(x2)
111  zb=dsin(y2)
112 
113  ab=dsqrt((xb-xa)**2+(yb-ya)**2+(zb-za)**2)
114  aob=(2.0_dp -ab*ab)/2.0_dp
115  aob=dacos(aob)
116  arcl=rearth*aob
117 
118  RETURN
119  END SUBROUTINE arc_dbl

131  SUBROUTINE area_dbl(SIDEA,SIDEB,SIDEC,AREA1)

133  ! function:

135  ! input:

137  ! output:
138  ! areal: is area of a triangle on a spherical plane
139  !--------------------------------------------------------------------

142  REAL(DP), INTENT(OUT) :: AREA1

144  REAL(DP) :: PSUM,PM,QMJC
145 
146  side1=sidea/rearth
147  side2=sideb/rearth
148  side3=sidec/rearth
149 
150  ! SLOWER TO CHECK THEN TO CALCULATE
151  ! IF(SIDE1 == 0.0_DP .OR. SIDE2 == 0.0_DP .OR. SIDE3 == 0.0_DP)THEN
152  ! AREA1=0.0_DP
153  ! ELSE
154

162  ! END IF

180  IMPLICIT NONE

182  REAL(DP), INTENT(IN) :: XX1,YY1,XX2,YY2

187 
188  x2=xx2*deg2rad
189  y2=yy2*deg2rad
190 
191  xxc=dcos(y1)*dsin(x1)+dcos(y2)*dsin(x2)
192  ! XXC=XXC/(COS(Y1)*COS(X1)+COS(Y2)*COS(X2))
193  ! XXC=ATAN(XXC)
194  xxc=datan2(xxc,(dcos(y1)*dcos(x1)+dcos(y2)*dcos(x2)))
195  xxc=xxc/deg2rad
196 
197  ! IF(XXC .LT. 0.0) XXC=180.0+XXC
198  IF(xxc < 0.0_dp) xxc=360.0_dp+xxc
199 
200  yyc=dcos(y1)*dcos(y1)+dcos(y2)*dcos(y2)+2.0_dp*dcos(y1)*dcos(y2)*dcos(x1-x2)
201  ! YYC=SQRT(YYC)/(SIN(Y1)+SIN(Y2))
202  yyc=datan2(dsqrt(yyc),(dsin(y1)+dsin(y2)))
203  ! YYC=ATAN(YYC)
