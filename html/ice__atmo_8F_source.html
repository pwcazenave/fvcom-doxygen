48 !
49  subroutine atmo_boundary_layer (ni, sfctype, Tsf, &
50  strx, stry, trf, qrf, delt, delq)
51 !
52 ! !DESCRIPTION:
53 !
54 ! Compute coefficients for atm/ice fluxes, stress, and reference
55 ! temperature. NOTE: \\
56 ! (1) all fluxes are positive downward, \\
57 ! (2) here, tstar = (WT)/U*, and qstar = (WQ)/U*, \\
58 ! (3) wind speeds should all be above a minimum speed (eg. 1.0 m/s). \\
59 !
60 ! ASSUME: \\
61 ! The saturation humidity of air at T(K): qsat(T) (kg/m**3) \\
62 !
63 ! Code originally based on CSM1 \\
64 !
65 ! !REVISION HISTORY:
66 !
67 ! author: Elizabeth C. Hunke, LANL
68 !
69 ! !USES:
70 !
71  use ice_grid ! for tmask
72 !
73 ! !INPUT/OUTPUT PARAMETERS:
74 !
75  integer (kind=int_kind), intent(in) :: &
76  ni ! thickness category index
77 
78  character (len=3), intent(in) ::&
79  sfctype ! ice or ocean
80 
81  real (kind=dbl_kind), dimension (ilo:ihi,jlo:jhi), intent(in) :: &
82  Tsf ! surface temperature of ice or ocean
83 
84  real (kind=dbl_kind), dimension (ilo:ihi,jlo:jhi), intent(out) ::&
85  strx &! x surface stress (N)
86  , stry &! y surface stress (N)
87  , Trf &! reference height temperature (K)
88  , Qrf &! reference height specific humidity (kg/kg)
89  , delt &! potential T difference (K)
90  , delq ! humidity difference (kg/kg)

93 !

95  k & ! iteration index
96  , i, j ! horizontal indices
97 
98  real (kind=dbl_kind) :: &
99  TsfK &! surface temperature in Kelvin (K)
100  , xqq &! temporary variable
101  , psimh &! stability function at zlvl (momentum)
102  , tau &! stress at zlvl
103  , fac &! interpolation factor
104  , al2 &! ln(z10 /zTrf)
105  , psix2 &! stability function at zTrf (heat and water)
106  , psimhs &! stable profile
107  , ssq &! sat surface humidity (kg/kg)
108  , qqq &! for qsat, dqsatdt
109  , TTT &! for qsat, dqsatdt
110  , qsat &! the saturation humidity of air (kg/m^3)
111  , Lheat ! Lvap or Lsub, depending on surface type
112 
113  real (kind=dbl_kind), dimension (1:(ihi-ilo+1)*(jhi-jlo+1)) ::&
114  ustar &! ustar (m/s)
115  , tstar &! tstar
116  , qstar &! qstar
117  , rdn &! sqrt of neutral exchange coefficient (momentum)
118  , rhn &! sqrt of neutral exchange coefficient (heat)
119  , ren &! sqrt of neutral exchange coefficient (water)
120  , rd &! sqrt of exchange coefficient (momentum)
121  , re &! sqrt of exchange coefficient (water)
122  , rh &! sqrt of exchange coefficient (heat)
123  , vmag &! surface wind magnitude (m/s)
124  , alz &! ln(zlvl /z10)
125  , thva &! virtual temperature (K)
126  , cp &! specific heat of moist air
127  , hol &! H (at zlvl ) over L
128  , stable &! stability factor
129  , psixh ! stability function at zlvl (heat and water)
130 
131  integer (kind=int_kind), dimension (1:(ihi-ilo+1)*(jhi-jlo+1)) ::&
132  indxi & ! compressed index in i-direction
133  , indxj ! compressed index in j-direction
134 
135  integer (kind=int_kind) ::&
136  icells & ! number of cells with aicen > puny or tmask true
137  , ij ! combined ij index
138 
139  real (kind=dbl_kind), parameter :: &
140  cpvir = cp_wv/cp_air - c1i &! defined as cp_wv/cp_air - 1.
141  , ztrf = c2i &! reference height for air temp (m)
142  , umin = c1i ! minimum wind speed (m/s)
143 
144  ! local functions
145  real (kind=dbl_kind) :: &
146  xd & ! dummy argument
147  , psimhu & ! unstable part of psimh
148  , psixhu ! unstable part of psimx
149 
150  !------------------------------------------------------------
151  ! Define functions
152  !------------------------------------------------------------
153 
154  psimhu(xd) = log((c1i+xd*(c2i+xd))*(c1i+xd*xd)/c8i) &
155  - c2i*atan(xd) + pi*p5
156 !ech & - c2i*atan(xd) + 1.571_dbl_kind
157 
158  psixhu(xd) = c2i * log((c1i + xd*xd)/c2i)
159 
160  al2 = log(zref/ztrf)
161 
162  !------------------------------------------------------------
163  ! Initialize
164  !------------------------------------------------------------
165 
166  do j = jlo,jhi
167  do i = ilo,ihi
168  lhcoef(i,j) = c0i
169  shcoef(i,j) = c0i
170  strx(i,j) = c0i
171  stry(i,j) = c0i
172  trf(i,j) = c0i
173  qrf(i,j) = c0i
174  delt(i,j) = c0i
175  delq(i,j) = c0i
176  enddo
177  enddo
178 
179  !------------------------------------------------------------
180  ! Compute turbulent flux coefficients, wind stress, and
181  ! reference temperature and humidity.
182  !------------------------------------------------------------
183 
184  if ( sfctype(1:3)=='ice' ) then
185 
186  !------------------------------------------------------------
187  ! identify grid cells with ice
188  !------------------------------------------------------------
189  icells = 0
190  do j = jlo,jhi
191  do i = ilo,ihi
192  if ( aicen(i,j,ni) > puny) then
193  icells = icells + 1
194  indxi(icells) = i
195  indxj(icells) = j
196  endif
197  enddo
198  enddo
199 
200  !------------------------------------------------------------
201  ! define some needed variables
202  !------------------------------------------------------------
203  qqq = qqqice ! for qsat
204  ttt = tttice ! for qsat
205  lheat = lsub ! ice to vapor
206  do ij = 1, icells
207  i = indxi(ij)
208  j = indxj(ij)
209  vmag(ij) = max(umin, wind(i,j))
210  rdn(ij) = vonkar/log(zref/iceruf) ! neutral coefficient
211  enddo ! ij
212 
213  elseif ( sfctype(1:3)=='ocn' ) then
214 
215  !------------------------------------------------------------
216  ! identify ocean cells
217  !------------------------------------------------------------
218  icells = 0
219  do j = jlo,jhi
220  do i = ilo,ihi
221  if ( tmask(i,j) ) then
222  icells = icells + 1
223  indxi(icells) = i
224  indxj(icells) = j
225  endif
226  enddo
227  enddo
228 
229  !------------------------------------------------------------
230  ! define some needed variables
231  !------------------------------------------------------------
232  qqq = qqqocn
233  ttt = tttocn
234  lheat = lvap ! liquid to vapor
235  do ij = 1, icells
236  i = indxi(ij)
237  j = indxj(ij)
238  vmag(ij) = max(umin, wind(i,j))
239  rdn(ij) = sqrt(0.0027_dbl_kind/vmag(ij) &
240  + .000142_dbl_kind + .0000764_dbl_kind*vmag(ij))
241  enddo ! ij
242 
243  endif ! sfctype
244 
245  do ij = 1, icells
246  i = indxi(ij)
247  j = indxj(ij)
248 
249  !------------------------------------------------------------
250  ! define some more needed variables
251  !------------------------------------------------------------
252 
253  tsfk = tsf(i,j) + tffresh ! surface temp (K)
254  qsat = qqq * exp(-ttt/tsfk) ! saturation humidity (kg/m^3)
255  ssq = qsat / rhoa(i,j) ! sat surf hum (kg/kg)
256 
257  thva(ij) = pott(i,j) * (c1i + zvir * qa(i,j)) ! virtual pot temp (K)
258  delt(i,j) = pott(i,j) - tsfk ! pot temp diff (K)
259  delq(i,j) = qa(i,j) - ssq ! spec hum dif (kg/kg)
260  alz(ij) = log(zlvl(i,j)/zref)
261  cp(ij) = cp_air*(c1i + cpvir*ssq)
262 
263  !------------------------------------------------------------
264  ! first estimate of Z/L and ustar, tstar and qstar
265  !------------------------------------------------------------
266 
267  ! neutral coefficients, z/L = 0.0
268  rhn(ij) = rdn(ij)
269  ren(ij) = rdn(ij)
270 
271  ! ustar,tstar,qstar
272  ustar(ij) = rdn(ij) * vmag(ij)
273  tstar(ij) = rhn(ij) * delt(i,j)
274  qstar(ij) = ren(ij) * delq(i,j)
275 
276  enddo ! ij
277 
278  !------------------------------------------------------------
279  ! iterate to converge on Z/L, ustar, tstar and qstar
280  !------------------------------------------------------------
281 
282  do k=1,5
283 
284  do ij = 1, icells
285  i = indxi(ij)
286  j = indxj(ij)
287 
288  ! compute stability & evaluate all stability functions
289  hol(ij) = vonkar * gravit * zlvl(i,j) &
290  * (tstar(ij)/thva(ij) &
291  + qstar(ij)/(c1i/zvir+qa(i,j))) &
292  / ustar(ij)**2
293  hol(ij) = sign( min(abs(hol(ij)),c10i), hol(ij) )
294  stable(ij) = p5 + sign(p5 , hol(ij))
295  xqq = max(sqrt(abs(c1i - c16*hol(ij))) , c1i)
296  xqq = sqrt(xqq)
297 
298  ! Jordan et al 1999
299  psimhs = -(0.7_dbl_kind*hol(ij) &
300  + 0.75_dbl_kind*(hol(ij)-14.3_dbl_kind) &
301  * exp(-0.35_dbl_kind*hol(ij)) + 10.7_dbl_kind)
302  psimh = psimhs*stable(ij) &
303  + (c1i - stable(ij))*psimhu(xqq)
304  psixh(ij) = psimhs*stable(ij) &
305  + (c1i - stable(ij))*psixhu(xqq)
306 
307  ! shift all coeffs to measurement height and stability
308  rd(ij) = rdn(ij) / (c1i+rdn(ij)/vonkar*(alz(ij)-psimh))
309  rh(ij) = rhn(ij) / (c1i+rhn(ij)/vonkar*(alz(ij)-psixh(ij)))
310  re(ij) = ren(ij) / (c1i+ren(ij)/vonkar*(alz(ij)-psixh(ij)))
311 
312  ! update ustar, tstar, qstar using updated, shifted coeffs
313  ustar(ij) = rd(ij) * vmag(ij)
314  tstar(ij) = rh(ij) * delt(i,j)
315  qstar(ij) = re(ij) * delq(i,j)
316 
317  enddo ! ij
318  enddo ! end iteration
319 
320 
321  do ij = 1, icells
322  i = indxi(ij)
323  j = indxj(ij)
324 
325  !------------------------------------------------------------
326  ! coefficients for turbulent flux calculation
327  !------------------------------------------------------------
328 
329 !ccsm shcoef(i,j) = rhoa(i,j) * ustar(ij) * cp(ij) * rh(ij)
330  ! add windless coefficient for sensible heat flux
331  ! as in Jordan et al (JGR, 1999)
332  shcoef(i,j) = rhoa(i,j) * ustar(ij) * cp(ij) * rh(ij) + c1i
333  lhcoef(i,j) = rhoa(i,j) * ustar(ij) * lheat * re(ij)
334  !------------------------------------------------------------
335  ! momentum flux
336  !------------------------------------------------------------
337  ! tau = rhoa(i,j) * ustar * ustar
338  ! strx = tau * uatm(i,j) / vmag
339  ! stry = tau * vatm(i,j) / vmag
340  !------------------------------------------------------------
341 
342  tau = rhoa(i,j) * ustar(ij) * rd(ij) ! not the stress at zlvl(i,j)
343  strx(i,j) = tau * uatm(i,j)
344  stry(i,j) = tau * vatm(i,j)
345 
346  !------------------------------------------------------------
347  ! Compute diagnostics: 2m ref T & Q
348  !------------------------------------------------------------
349  hol(ij) = hol(ij)*ztrf/zlvl(i,j)
350  xqq = max( c1i, sqrt(abs(c1i-c16*hol(ij))) )
351  xqq = sqrt(xqq)
352  psix2 = -c5i*hol(ij)*stable(ij) + (c1i-stable(ij))*psixhu(xqq)
353  fac = (rh(ij)/vonkar) * (alz(ij) + al2 - psixh(ij) + psix2)
354  trf(i,j)= pott(i,j) - delt(i,j)*fac
355  trf(i,j)= trf(i,j) - p01*ztrf ! pot temp to temp correction
356  fac = (re(ij)/vonkar) * (alz(ij) + al2 - psixh(ij) + psix2)
357  qrf(i,j)= qa(i,j) - delq(i,j)*fac
358 
359  enddo ! ij
360 
361  end subroutine atmo_boundary_layer
