497  subroutine interp_coeff_monthly (recslot)
498 !
499 ! !DESCRIPTION:
500 !
501 ! Compute coefficients for interpolating monthly data to current time step.
502 !
503 ! !REVISION HISTORY:
504 !
505 ! authors: same as module
506 !
507 ! !USES:
508 !
509 ! !INPUT/OUTPUT PARAMETERS:
510 !
511  integer (kind=int_kind), intent(in) :: &
512  recslot ! slot (1 or 2) for current record
513 !
514 !EOP
515 !
516  real (kind=dbl_kind) :: &
517  tt & ! seconds elapsed in current year
518  , t1, t2 ! seconds elapsed at month midpoint
519 
520  real (kind=dbl_kind) :: &
521  daymid(0:13) ! month mid-points
522 
523  daymid(1:13) = 14._dbl_kind ! time frame ends 0 sec into day 15
524  daymid(0) = -17._dbl_kind ! Dec 15, 0 sec
525 
526  ! make time cyclic
527  tt = mod(ftime/secday,c365)
528 
529  ! Find neighboring times
530 
531  if (recslot==2) then ! first half of month
532  t2 = daycal(month) + daymid(month) ! midpoint, current month
533  if (month == 1) then
534  t1 = daymid(0) ! Dec 15 (0 sec)
535  else
536  t1 = daycal(month-1) + daymid(month-1) ! midpoint, previous month
537  endif
538  else ! second half of month
539  t1 = daycal(month) + daymid(month) ! midpoint, current month
540  t2 = daycal(month+1) + daymid(month+1)! day 15 of next month (0 sec)
541  endif
542 
543  ! Compute coefficients
544  c1intp = (t2 - tt) / (t2 - t1)
545  c2intp = c1i - c1intp
546 
547  end subroutine interp_coeff_monthly

1094 !
1095 ! !DESCRIPTION:
1096 !
1097 ! Computes downwelling shortwave radiation based on sun declination
1098 ! and cloud fraction, as in AOMIP, following Zillman (1972) and
1099 ! Parkinson and Washington (1979).
1100 !
1101 ! !REVISION HISTORY:
1102 !
1103 ! authors: Elizabeth C. Hunke, LANL
1104 !
1105 ! !USES:
1106 !
1107  use ice_grid
1108  use ice_albedo
1109  use ice_constants
1110 !
1111 ! !INPUT/OUTPUT PARAMETERS:
1112 !
1113 !EOP
1114 !
1115  real (kind=dbl_kind) :: &
1116  hour_angle &
1117  , solar_time &
1118  , declin &
1119  , cosZ &
1120  , e_i, d_i &
1121  , sw0 &
1122  , deg2rad_i
1123 
1124  integer (kind=int_kind) :: i,j
1125 
1126  do j=jlo,jhi
1127  do i=ilo,ihi
1128  deg2rad_i = pi/180._dbl_kind
1129  solar_time = mod(real(sec,kind=dbl_kind),secday)/3600._dbl_kind &
1130  + c12*sin(p5*tlon(i,j))
1131  hour_angle = (c12 - solar_time)*pi/c12
1132  declin = 23.44_dbl_kind*cos((172._dbl_kind-yday) &
1133  * c2i*pi/c365)*deg2rad_i
1134  cosz = sin(tlat(i,j))*sin(declin) &
1135  + cos(tlat(i,j))*cos(declin)*cos(hour_angle)
1136  cosz = max(cosz,c0i)
1137  e_i = 1.e5_dbl_kind*qa(i,j) &
1138  /(0.622_dbl_kind + 0.378_dbl_kind*qa(i,j))
1139  d_i = (cosz+2.7_dbl_kind)*e_i*1.e-5_dbl_kind+1.085_dbl_kind*cosz+p1
1140  sw0 = 1353._dbl_kind*cosz**2/d_i
1141  sw0 = max(sw0,c0i)
1142 
1143  ! total downward shortwave for cice
1144  fsw(i,j) = sw0*(c1i-p6*cldf(i,j)**3)
1145  fsw(i,j) = fsw(i,j)*hm(i,j)
1146  enddo
