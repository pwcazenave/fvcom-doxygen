195  SUBROUTINE set_bndry_meanflow
196 
197 !------------------------------------------------------------------------------!

199  USE bcs

201  USE mod_spherical

203  USE mod_obcs

206  REAL(DP) DX12,DY12,ATMP1,HTMP

214 
215  ALLOCATE(anglemf(nmfcell),mfarea(nmfcell),vlctymf(nmfcell),mfqdis(nmfcell))

217 
218  DO i=1,nmfcell

221  print*, 'NO.',i,'MEAN FLOW CELL'
222  print*, 'IS NOT A OPEN BOUNDARY ONE'
223  CALL pstop
224  END IF

226  DO j=1,3

238  j2=jtmp+2-int((jtmp+2)/4)*3
239  i1=nv(ii,j1)
240  i2=nv(ii,j2)
241 
242  node_mfcell(i,1)=i1
243  node_mfcell(i,2)=i2
244 
245  htmp=0.5_sp*(h(i1)+h(i2)) ! may be a problem here, should be replaced dy D
246  dy12=vy(i1)-vy(i2)
247 # if defined (SPHERICAL)
248  x1_dp = vx(i2)
249  y1_dp = vy(i2)
250  x2_dp = vx(i1)
251  y2_dp = vy(i1)
252  CALL arcx(x1_dp,y1_dp,x2_dp,y2_dp,side)
253  dx12 = side
254 
255  dy12 = tpi*dy12
256 # else
257  dx12=vx(i1)-vx(i2)
258 # endif
259  atmp1=atan2(dy12,dx12)
260  mfarea(i)=sqrt(dx12**2+dy12**2)*htmp ! for spherical coordinates is Phthagolean Theorem still valid?
261  anglemf(i)=atmp1+3.1415927/2.0
262  rdismf(i,1)=art1(i1)/(art1(i1)+art1(i2))
263  rdismf(i,2)=art1(i2)/(art1(i1)+art1(i2))
264  END DO
265  END IF
266 
267  RETURN

296  SUBROUTINE bracket(TMAP,STIME,L1,L2,FACT,BACT,IERR)
297 !==============================================================================|
298 ! DETERMINE DATA INTERVAL IN WHICH CURRENT TIME LIES |
299 ! |
300 ! L1: DATA INTERVAL PROCEEDING TIME |
301 ! L2: DATA INTERVAL AFTER TIME |
302 ! FACT: LINEAR INTERPOLATION COEFFICIENT (0->1) |
303 ! FACT = .5 : STIME LIES EXACTLY BETWEEN TWO DATA TIMES |
304 ! FACT = 1. : STIME OCCURS AT SECOND DATA TIME |
305 ! BACT = 1.-FACT
306 ! IERR: RETURNS INTEGER ERROR |
307 ! IERR = 0 : NO ERROR, TIME IS BRACKETED BY DATA TIMES |
308 ! IERR =-1 : STIME PROCEEDS ALL DATA TIMES |
309 ! IERR = 1 : STIME IS GREATER THAN ALL DATA TIMES |
310 ! |
311 ! IF STIME PROCEEDS DATA, IERR IS SET TO -1, L1 TO 1, AND FACT TO 0. !
312 ! IF STIME SUPERCEEDS DATA, IERR IS SET TO -1, L2 TO LMAX, AND FACT TO 1. !
313 !==============================================================================|
314  USE mod_types
315  IMPLICIT NONE
316 !------------------------------------------------------------------------------!
317  TYPE(bc), INTENT(IN) :: TMAP
318  REAL(SP), INTENT(IN) :: STIME
319  INTEGER, INTENT(OUT) :: L1,L2
320  REAL(SP), INTENT(OUT) :: FACT,BACT
321  INTEGER, INTENT(OUT) :: IERR
322 !------------------------------------------------------------------------------!
323  REAL(SP) T1,T2
324  INTEGER I,NTMAX
325 !==============================================================================|
326 
327  ntmax = tmap%NTIMES
328  IF(stime < tmap%TIMES(1))THEN
329  fact = 0.0_sp
330  bact = 1.0_sp
331  l1 = 1
332  l2 = 1
333  ierr = -1
334  RETURN
335  END IF
336 
337  IF(stime > tmap%TIMES(ntmax))THEN
338  fact = 1.0_sp
339  bact = 0.0_sp
340  l1 = ntmax
341  l2 = ntmax
342  ierr = 1
343  RETURN
344  END IF
345 
346  IF(ntmax == 1)THEN
347  fact = 1.0_sp
348  bact = 0.0_sp
349  l1 = 1
350  l2 = 1
351  ierr = 0
352  RETURN
353  END IF
354 
355 
356  DO i=2,tmap%NTIMES
357  t1 = tmap%TIMES(i-1)
358  t2 = tmap%TIMES(i)
359  IF(stime >= t1 .AND. stime <= t2)THEN
360  l1 = i-1
361  l2 = i
362  ierr = 0
363  fact = (stime-t1)/(t2-t1)
364  bact = 1.0_sp-fact
365  END IF
366  END DO
367 
368 
369  RETURN
370  END SUBROUTINE bracket
