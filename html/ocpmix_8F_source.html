137  iyear = inttim(1)
138  iyrm1 = iyear-1
139  leapyr=(mod(iyear,4) == 0 .AND. mod(iyear,100) /= 0) .OR. &
140  mod(iyear,1000) == 0
141  idnow=0
142  IF(inttim(2) > 12)THEN
143  WRITE (printf, 8) inttim(2), (inttim(ii), ii=1,6)
144  8 FORMAT (' erroneous month ', i2, ' in date/time ', 6i4)
145  ELSE IF(inttim(2) > 1)THEN
146  DO 10 i = 1,inttim(2)-1
147  idnow=idnow+idymon(i)
148  10 CONTINUE
149  ENDIF

152  idnow = idnow + iyear*365 + iyrm1/4 - iyrm1/100 + iyrm1/1000 + 1

154  IF(.NOT.logref)THEN

159  dttime = REAL(IDNOW-REFDAY) * 24.*3600.
160  ENDIF

162  REAL(inttim(6))
163  RETURN
164  END FUNCTION dttime
