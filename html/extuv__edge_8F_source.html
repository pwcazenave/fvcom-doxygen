75 !!# if defined (WET_DRY)

77 
78 !!# if !defined (NH)
79 !! RESX = ADX2D + ADVUA + DRX2D + PSTX - COR*VA*D1*ART &
80 !! -(WUSURF2 + WUBOT)*ART
81 !! RESY = ADY2D + ADVVA + DRY2D + PSTY + COR*UA*D1*ART &
82 !! -(WVSURF2 + WVBOT)*ART
83 !!# else
84 !! RESX = ADX2D + ADVUA + DRX2D + PSTX - COR*VA*D1*ART &
85 !! -(WUSURF2 + WUBOT)*ART + NHQ2DX
86 !! RESY = ADY2D + ADVVA + DRY2D + PSTY + COR*UA*D1*ART &
87 !! -(WVSURF2 + WVBOT)*ART + NHQ2DY
88 !!# endif
89 
90 !!# if defined (SPHERICAL)
91 !! RESX = RESX -UA*VA/REARTH*TAN(DEG2RAD*YC)*D1*ART
92 !! RESY = RESY +UA*UA/REARTH*TAN(DEG2RAD*YC)*D1*ART

95 !!!
96 !!!--UPDATE----------------------------------------------------------------------|
97 !!!
98 
99 !! UAF = (UARK*(H1+ELRK1)-ALPHA_RK(K)*DTE*RESX/ART)/(H1+ELF1)
100 !! VAF = (VARK*(H1+ELRK1)-ALPHA_RK(K)*DTE*RESY/ART)/(H1+ELF1)
101 !! UAS = UAF
102 !! VAS = VAF
103 !! END IF
104 !!# endif
105 
106  DO i=1,nt
107 # if defined (WET_DRY)
108  IF(iswetce(i)*iswetc(i) == 1)THEN
109 # endif
110 
111 # if !defined (NH)
112  resx(i) = adx2d(i)+advua(i)+drx2d(i)+pstx(i)-cor(i)*va(i)*d1(i)*art(i) &
113  -(wusurf2(i)+wubot(i))*art(i)
114  resy(i) = ady2d(i)+advva(i)+dry2d(i)+psty(i)+cor(i)*ua(i)*d1(i)*art(i) &
115  -(wvsurf2(i)+wvbot(i))*art(i)
116 # else
117  resx(i) = adx2d(i)+advua(i)+drx2d(i)+pstx(i)-cor(i)*va(i)*d1(i)*art(i) &
118  -(wusurf2(i)+wubot(i))*art(i)+nhq2dx(i)
119  resy(i) = ady2d(i)+advva(i)+dry2d(i)+psty(i)+cor(i)*ua(i)*d1(i)*art(i) &
120  -(wvsurf2(i)+wvbot(i))*art(i)+nhq2dy(i)
121 # endif
122 
123 # if defined (SPHERICAL)
124  resx(i) = resx(i) &
125  -ua(i)*va(i)/rearth*tan(deg2rad*yc(i))*d1(i)*art(i)
126  resy(i) = resy(i) &
127  +ua(i)*ua(i)/rearth*tan(deg2rad*yc(i))*d1(i)*art(i)

129 
130

133  resy(i) = resy(i) + wavestry_2d(i)
134 # endif

136 !--UPDATE----------------------------------------------------------------------|
137 !
138 
139  uaf(i) = (uark(i)*(h1(i)+elrk1(i))-alpha_rk(k)*dte*resx(i)/art(i))/(h1(i)+elf1(i))
140  vaf(i) = (vark(i)*(h1(i)+elrk1(i))-alpha_rk(k)*dte*resy(i)/art(i))/(h1(i)+elf1(i))
141 # if defined (WET_DRY)
142  ELSE
143  uaf(i) = 0.0_sp
144  vaf(i) = 0.0_sp
145  END IF
146 # endif
147  END DO
148 
149 # if defined (SPHERICAL)
150  CALL extuv_edge_xy(k)
151 # endif
152 
153  vaf(0) = vaft
154  uaf(0) = uaft
155 
156 !
157 !--ADJUST EXTERNAL VELOCITY IN SPONGE REGION-----------------------------------|
158 !
159 !old: UAF = UAF-CC_SPONGE*UAF
160 !old: VAF = VAF-CC_SPONGE*VAF
161 ! ---- new: Karsten Lettmann: 2012.06.25 -------
162  uaf = uaf/(1.0_sp+cc_sponge*uaf**2.0_sp)
163  vaf = vaf/(1.0_sp+cc_sponge*vaf**2.0_sp)
164 ! ------- end new -------------------------------
165 
166 
167 !
168 !--STORE VARIABLES FOR MOMENTUM BALANCE CHECK----------------------------------|
169 !
170 # if defined (BALANCE_2D)
171  IF(k == 4) THEN
172  tmp=art*(h1+elf1)
173  advua2 = advua2 + (advua-adfxa)/tmp/float(isplit) !X- HORIZONTAL ADVECTION (m/s^2)
174  advva2 = advva2 + (advva-adfya)/tmp/float(isplit) !X- HORIZONTAL ADVECTION (m/s^2)
175  adfx2 = adfx2 + adfxa/tmp/float(isplit)
176  adfy2 = adfy2 + adfya/tmp/float(isplit)
177  drx2d2 = drx2d2 + drx2d/tmp/float(isplit) !X- BAROCLINIC PRESURE GRADIENT FORCE
178  dry2d2 = dry2d2 + dry2d/tmp/float(isplit) !Y- BAROCLINIC PRESURE GRADIENT FORCE
179  corx2 = corx2 - cor*va*d1/(h1+elf1)/float(isplit) !X- CORIOLIS FORCE
180  cory2 = cory2 + cor*ua*d1/(h1+elf1)/float(isplit) !Y- CORIOLIS FORCE
181  pstx2 = pstx2 + pstx/tmp/float(isplit) !X- BAROTROPIC PRESURE GRADIENT FORCE
182  psty2 = psty2 + psty/tmp/float(isplit) !Y- BAROTROPIC PRESURE GRADIENT FORCE
183  adx2d2 = adx2d2 + adx2d/tmp/float(isplit) !GX (m/s^2)
184  ady2d2 = ady2d2 + ady2d/tmp/float(isplit) !GY (m/s^2)
185  wusurbf2=wusurbf2-(wusurf2+wubot)/(h1+elf1)/float(isplit) !X-SURFACE & BOTTOM FRICTION
186  wvsurbf2=wvsurbf2-(wvsurf2+wvbot)/(h1+elf1)/float(isplit) !Y-SURFACE & BOTTOM FRICTION
187  dudt2 = dudt2 + (uaf-uark*(h1+elrk1)/(h1+elf1))/dte/float(isplit)
188  dvdt2 = dvdt2 + (vaf-vark*(h1+elrk1)/(h1+elf1))/dte/float(isplit)
189 
190  END IF
191 # endif
192
