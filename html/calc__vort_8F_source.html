30 ! CALCULATE VORTICITY FIELD AT NODES |
31 ! OPTIONAL INPUT: VTYPE |
32 ! 1 CURL OF VERTICALLY-AVERAGED VELOCITY (DEFAULT) |
33 ! 2 CURL OF FLUX (For Streamfunctions) |
34 ! RETURNS: VORT(1:M) |
35 ! |
36 ! CHECK, INITIALIZE WITH ANALYTICAL VELOCITY FIELD (UA = -.5YC, VA = .5XC) !
37 !==============================================================================|
38  SUBROUTINE calc_vort(VTYPE)
39 !==============================================================================|
40  USE all_vars
41  IMPLICIT NONE
42  INTEGER, OPTIONAL, INTENT(IN) :: VTYPE
43  REAL(SP) :: XFLUX(0:m)
44  REAL(SP) :: UIJ,VIJ,EXFLUX,AVE
45  INTEGER :: I,J,IA,IB,I1,CNT,JNODE,MY_VTYPE
46  INTEGER, ALLOCATABLE :: LIST(:)
47 !==============================================================================|
48 
49  !---------------------------------------------
50  !Process input arguments
51  !---------------------------------------------
52  my_vtype = 1
53  IF(present(vtype))Then
54  my_vtype = vtype
55  ENDIF
56 
57  !---------------------------------------------
58  !Initialize Fields
59  !---------------------------------------------
60  ALLOCATE(list(m)) ; list = 0
61  vort = 0.0
62 
63  !===================================================================
64  ! Calculate vorticity at all nodes using c.v. half edges
65  ! Note minus sign (nodes are order clockwise [against convention])
66  !===================================================================
67 
68  SELECT CASE(my_vtype)
69 
70  !----------------------------------------------
71  CASE(1) !Curl of Vert-Avged Velocity
72  !----------------------------------------------
73  DO i=1,ncv
74  i1 = ntrg(i)
75  ia = niec(i,1)
76  ib = niec(i,2)
77  uij = ua(i1)
78  vij = va(i1)
79  exflux = -(uij*dltxe(i) + vij*dltye(i))
80  vort(ia) = vort(ia)-exflux
81  vort(ib) = vort(ib)+exflux
82  END DO
83  !----------------------------------------------
84  CASE(2) !Curl of Flux
85  !----------------------------------------------
86  DO i=1,ncv
87  i1 = ntrg(i)
88  ia = niec(i,1)
89  ib = niec(i,2)
90  uij = ua(i1)
91  vij = va(i1)
92  exflux = -d1(i1)*(uij*dltxe(i) + vij*dltye(i))
93  vort(ia) = vort(ia)-exflux
94  vort(ib) = vort(ib)+exflux
95  END DO
96  !----------------------------------------------
97  CASE DEFAULT
98  !ERROR AND HALT
99  END SELECT
100 
101  vort = vort/art1
102 
103  !===================================================================
104  !Correction at Boundaries (May no longer be necessary for 2.5+)
105  !===================================================================
106 
107  do i=1,m
108  if(isonb(i) > 0)then
109  ave = 0.
110  cnt = 0
111  do j=1,ntsn(i)
112  jnode = nbsn(i,j)
113  if(jnode /= i .and. isonb(jnode) == 0)then
114  ave = ave + vort(jnode)
115  cnt = cnt + 1
116  end if
117  end do
118  vort(i) = ave/float(cnt)
119  if(cnt == 0) list(i) = 1
120  end if
121  end do
122 
123  do i=1,m
124  if(list(i) > 0)then
125  ave = 0.
126  cnt = 0
127  do j=1,ntsn(i)
128  jnode = nbsn(i,j)
129  if(jnode /= i .and. list(jnode) == 0)then
130  ave = ave + vort(jnode)
131  cnt = cnt + 1
132  end if
133  end do
134  vort(i) = ave/float(cnt)
135  end if
136  end do
137 
138 
139  RETURN
140  END SUBROUTINE calc_vort
