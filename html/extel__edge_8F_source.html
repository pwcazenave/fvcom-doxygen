31 !==============================================================================|

58  INTEGER, INTENT(IN) :: K
59  REAL(SP) :: XFLUX(0:mt)
60  REAL(SP) :: DIJ,UIJ,VIJ,DTK,UN,EXFLUX
61  INTEGER :: I,J,I1,IA,IB,JJ,J1,J2
62 
63 # if defined (BALANCE_2D)
64  REAL(SP), DIMENSION(0:MT) :: XFLUXU,XFLUXV
65  REAL(SP), DIMENSION(0:NT) :: XFLUXU1,XFLUXV1

73  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: extel_edge.F",k
74 !----------INITIALIZE FLUX ARRAY ----------------------------------------------!
75 
76  xflux = 0.0_sp
77 
78 # if defined (BALANCE_2D)
79  xfluxu= 0.0_sp
80  xfluxv= 0.0_sp
81  xfluxu1= 0.0_sp
82  xfluxv1= 0.0_sp
83 # endif
84 !---------ACCUMULATE FLUX BY LOOPING OVER CONTROL VOLUME HALF EDGES------------!
85 
86  DO i=1,ncv
87  i1 = ntrg(i)
88  ia = niec(i,1)
89  ib = niec(i,2)
90  dij = d1(i1)
91 
92  uij = ua(i1)
93  vij = va(i1)
94  exflux = dij*(-uij*dltye(i) + vij*dltxe(i))
95 
96 # if defined (PLBC)
97  exflux = dij*(-uij*dltye(i) + 0.0_sp*dltxe(i))
98 # endif
99 
100  xflux(ia) = xflux(ia)-exflux
101  xflux(ib) = xflux(ib)+exflux
102 
103 # if defined (BALANCE_2D)
104  exflux_u = -dij*uij*dltye(i)
105  exflux_v = dij*vij*dltxe(i)
106  xfluxu(ia) = xfluxu(ia)-exflux_u
107  xfluxu(ib) = xfluxu(ib)+exflux_u
108  xfluxv(ia) = xfluxv(ia)-exflux_v
109  xfluxv(ib) = xfluxv(ib)+exflux_v
110 # endif
111 
112  END DO
113 
114 ! write(ipt,*) "after control volume flux",maxval(xflux),minval(xflux)
115 ! write(ipt,*) "after control volume flux",maxloc(xflux),minloc(xflux)
116 
117 # if defined (SPHERICAL)
118  CALL extel_edge_xy(k,xflux)
119 # endif
120 
121 !--ADD EVAPORATION AND PRECIPITATION TERMS-------------------------------------!
122  IF (precipitation_on) THEN
123 # if defined (ICE)
124  WHERE(isicen==1)
125  qprec2=qprec
126  qevap2=qevap
127  END WHERE
128 
129 !!$ DO I=1,M
130 !!$ IF(ISICEN(i)==1) THEN
131 !!$ QPREC2(I)= QPREC3(I)
132 !!$ QEVAP2(I)= QEVAP3(I)
133 !!$ END IF
134 !!$ END DO
135 # endif
136 
137 !qxu XFLUX = XFLUX+(QEVAP2-QPREC2)*ROFVROS*ART1
138 !qxu---the evap is negative for evaporating in ocean
139  xflux = xflux-(qevap2+qprec2)*rofvros*art1
140  END IF
141 
142 !--ADD GROUND WATER TERM-------------------------------------------------------!
143 
144  IF(groundwater_on) THEN
145  xflux = xflux - bfwdis2
146  END IF
147 
148 !--SAVE ACCUMULATED FLUX ON OPEN BOUNDARY NODES AND ZERO OUT OPEN BOUNDARY FLUX!
149 
150  IF(iobcn > 0) THEN
151  DO i=1,iobcn
152  xflux_obcn(i)=xflux(i_obc_n(i))
153  xflux(i_obc_n(i)) = 0.0_sp
154  END DO
155  END IF
156 
157 
158 !---------ADJUST FLUX FOR FRESH WATER DISCHARGE--------------------------------!
159 
160  IF(numqbc >= 1) THEN
161  IF(river_inflow_location == 'node') THEN
162  DO j=1,numqbc
163  jj=inodeq(j)
164  xflux(jj)=xflux(jj)-qdis(j)
165  END DO
166  ELSE IF(river_inflow_location == 'edge') THEN
167  DO j=1,numqbc
168  j1=n_icellq(j,1)
169  j2=n_icellq(j,2)
170  xflux(j1)=xflux(j1)-qdis(j)*rdisq(j,1)
171  xflux(j2)=xflux(j2)-qdis(j)*rdisq(j,2)
172  END DO
173  END IF
174  END IF
175 
176 
177 !----------PERFORM UPDATE ON ELF-----------------------------------------------!
178 
179  dtk = alpha_rk(k)*dte
180  elf = elrk - dtk*xflux/art1
181 !!# if defined (THIN_DAM)
182 !! DO I=1,NODE_DAM1_N
183 !! JN=I_NODE_DAM1_N(I,1)
184 !! ELF(JN)=ELRK(JN)-DTK*(XFLUX(JN)+XFLUX(I_NODE_DAM1_N(I,2)))&
185 !! &/(ART1(JN)+ART1(I_NODE_DAM1_N(I,2)))
186 !! ELF(I_NODE_DAM1_N(I,2))=ELF(JN)
187 !! END DO
188 !! DO I=1,NODE_DAM2_N
189 !! JN=I_NODE_DAM2_N(I,1)
190 !! ELF(JN)=ELRK(JN)-DTK*(XFLUX(JN)+XFLUX(I_NODE_DAM2_N(I,2))&
191 !! &+XFLUX(I_NODE_DAM2_N(I,3)) )&
192 !! &/(ART1(JN)+ART1(I_NODE_DAM2_N(I,2))+ART1(I_NODE_DAM2_N(I,3)))
193 !! ELF(I_NODE_DAM2_N(I,2))=ELF(JN)
194 !! ELF(I_NODE_DAM2_N(I,3))=ELF(JN)
195 !! END DO
196 !! DO I=1,NODE_DAM3_N
197 !! JN=I_NODE_DAM3_N(I,1)
198 !! ELF(JN)=ELRK(JN)-DTK*(XFLUX(JN)+XFLUX(I_NODE_DAM3_N(I,2))&
199 !! &+XFLUX(I_NODE_DAM3_N(I,3))+XFLUX(I_NODE_DAM3_N(I,4)) )&
200 !! &/(ART1(JN)+ART1(I_NODE_DAM3_N(I,2))+ART1(I_NODE_DAM3_N(I&
201 !! &,3))+ART1(I_NODE_DAM3_N(I,4)))
202 !! ELF(I_NODE_DAM3_N(I,2))=ELF(JN)
203 !! ELF(I_NODE_DAM3_N(I,3))=ELF(JN)
204 !! ELF(I_NODE_DAM3_N(I,4))=ELF(JN)
205 !! END DO
206 !!# endif
207 
208 !
209 !--STORE VARIABLES FOR MOMENTUM BALANCE CHECK----------------------------------|
210 !
211 # if defined (BALANCE_2D)
212 
213  DO i=1,n
214  xfluxu1(i)=one_third*(xfluxu(nv(i,1))+ xfluxu(nv(i,2))+ xfluxu(nv(i,3)))
215  xfluxv1(i)=one_third*(xfluxv(nv(i,1))+ xfluxv(nv(i,2))+ xfluxv(nv(i,3)))
216  END DO
217 
218  IF(k == 4) THEN
219  divx2d2 = divx2d2 + xfluxu1/art/float(isplit) !dUD/dx
220  divy2d2 = divy2d2 + xfluxv1/art/float(isplit) !dVD/dy
221  dedt2 = dedt2 + (elf1-elrk1*(h1+elrk1)/(h1+elf1))/dte/float(isplit)
222  END IF
223 # endif
224 
225  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "END: extel_edge.F"
226 
227 # endif
228  RETURN
229  END SUBROUTINE extel_edge
