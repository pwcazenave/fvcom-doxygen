1 SUBROUTINE phytoplankton
2 !==========================================================================
3 ! This routine computes source and sink terms of phytoplankton in 1D ===
4 ! without vertical mixing ===
5 !==========================================================================
6 ! Define global data.
7  USE mod_1d
8  USE mod_nutrient
9  USE mod_phytoplankton
10  USE mod_zooplankton
11  USE mod_detritus
12  USE mod_bacteria
13  USE mod_dom
14 
15  IMPLICIT NONE
16  INTEGER :: I,K,J,I1,I2
17  rEAL(SPP) :: TEMPP, TRANS, TATANUL, IRRAD, IRRAD1
18  REAL(SPP) :: SINKK(10),RATIO(10),RATIOT
19 
20 !=======================================================================
21 ! BEGINNING EXECUTABLE
22 !=======================================================================
23 
24 
25  DO 10 k=1,kbvm1 !vertical
26 !************** PAR-PHYTOPLANKTON PRODUCTION **************************
27  tempp=t_bio(k)
28  tatanul=1.0e-10 !Total attanuation
29  tatanul=tatanul+atanu_w*delta_d(k) !Water attanuation
30  trans=0.0_spp
31  DO j=1,nnp
32  trans=trans+bio_p(k,j)*chl2c(j) !Total chlorophyll
33  END DO
34  IF(trans.GT.0.0) tatanul=tatanul+trans*atanu_c*delta_d(k)
35  trans=0.0_spp
36  DO j=1,nnd
37  trans=trans+bio_d(k,j) !Total detritus
38  END DO
39  IF(trans.GT.0.0) tatanul=tatanul+trans*atanu_d*delta_d(k) ! Detrital attanuation
40  irrad1=irrad0*exp(-tatanul) ! Irradiance at the layer bottom
41  irrad=(irrad0-irrad1)/tatanul ! Average irradiance
42  DO 20 j=1,nnp
43  utr(j)=exp(-a_tp(j)*abs(tempp-t_optp(j))) ! Temperature forcing on phyto
44  SELECT CASE (trim(l_function)) ! Light function choice
45  CASE('EXP_LIGHT') ! Exponential (Frank et al. ,86)
46  ulr(j)=exp(tatanul*depth_z(k))
47  CASE('SL62_LIGHT') ! Exponential (Steele, 62)
48  ulr(j)=irrad/i_opt(j)*exp(1-irrad/i_opt(j))
49  CASE('MM_LIGHT') ! Michaealis (Baly 35)
50  ulr(j)=alphap(j)*irrad/(k_light(j)+alphap(j)*irrad)
51  CASE('LB_LIGHT') ! Michaelis (Bannister, 79)
52  ulr(j)=alphap(j)*irrad/((k_light(j))**n_p(j) &
53  +(alphap(j)*irrad)**betap(j))**(1/n_p(j))
54  CASE('V65_LIGHT') ! Michaelis (Vollenweider 65)
55  ulr(j)=alphap(j)*irrad/(sqrt(i_opt(j)**2+(alphap(j)*irrad)**2) &
56  *(1+(betap(j)*irrad/i_opt(j))**2)**(n_p(j)/2))
57  CASE('PE78_LIGHT') ! Michaelis (Peeters & Eiler, 78)
58  ulr(j)=irrad/i_opt(j)*(2+alphap(j))/(1+alphap(j)*irrad/i_opt(j) &
59  +(irrad/i_opt(j))**2)
60  CASE('WNS74_LIGHT') ! Exponentail (Webb, 1978)
61  ulr(j)=1-exp(-alphap(j)*irrad/umax(j))
62  CASE('PGH80_LIGHT') ! Exponential (Platt et al., 80)
63  ulr(j)=(1-exp(-alphap(j)*irrad/umax(j)))*exp(-betap(j)*irrad/umax(j))
64  CASE('JP76_LIGHT') ! Tangent (Jessy & Platt, 76)
65  ulr(j)=tanh(alphap(j)*irrad/umax(j))
66  CASE('BWDC99_LIGHT') ! Tangent (Bissett et al., 99)
67  ulr(j)=tanh(alphap(j)*(irrad-i_opt(j))/umax(j)) &
68  *exp(betap(j)*(i_opt(j)-irrad))
69  END SELECT
70 !************* NUTRIENT-PHYTOPLANKTON GORWTH ***************
71  unrmin(j)=1.0_spp
72  DO i1=1,nnn
73  unr(j,i1)=fvnn(k,i1)/(fvnn(k,i1)+ksn(j,i1))
74  END DO
75  IF (.NOT. no3_on) THEN
76  DO i1=1,nnn
77  IF(n2cp(i1,j).GT.0.) unrmin(j)=min(unrmin(j),unr(j,i1))
78  END DO
79  ELSE
80  unr(j,2)=fvnn(k,2)/(fvnn(k,2)+ksn(j,2)) & ! NO3 must be nutrient 2
81  *ksn(j,1)/(fvnn(k,1)+ksn(j,1)) ! Parker NH4 inhibition
82  unr(j,nnn+1)=unr(j,1)+unr(j,2) ! Total nitrogen factor
83  DO i1=3,nnn
84  IF(n2cp(i1,j).GT.0.) unrmin(j)=min(unrmin(j),unr(j,i1))
85  END DO
86  unrmin(j)=min(unrmin(j),unr(j,nnn+1)) ! Combined NH4 and NO3
87  END IF
88  u_p(k,j)=umax(j)*utr(j)*(alpha_u(j)*ulr(j)*unrmin(j) &
89  +(1-alpha_u(j))*min(ulr(j),unrmin(j))) &
90  *bio_p(k,j)*chl2c(j) !Chl concentration
91  DO i1=1,nnn
92  uptake_pn(k,i1,j)=u_p(k,j)*n2cp(i1,j) !Nutrient uptake
93  END DO
94  IF (no3_on) THEN
95  uptake_pn(k,1,j)=u_p(k,j)*n2cp(1,j)*unr(j,1)/unr(j,nnn+1) !NH4
96  uptake_pn(k,2,j)=u_p(k,j)*n2cp(1,j)*unr(j,2)/unr(j,nnn+1) !NO3
97  END IF
98 !*************** SINK TERMS ****************
99  p_dom(k,j)=dpdom(j)*fvp(k,j) !DOM EXUDATION
100  p_d(k,j)=mpd(j)*fvp(k,j)**m_p(j)
101  p_n(k,j)=r_p(j)*fvp(k,j)*exp(rp_t*tempp) !Phyto respiration with T forcing
102  ratiot=fvp(k,j)/((p_dom(k,j)+p_d(k,j)+p_n(k,j))*t_step+0.0001)
103  IF(ratiot.LT.1.) THEN
104  p_dom(k,j)=0.0
105  p_d(k,j)=0.0
106  p_n(k,j)=0.0
107  END IF
108  fvp(k,j)=fvp(k,j)-(p_dom(k,j)+p_d(k,j)+p_n(k,j))*t_step
109  20 CONTINUE
110 !************** CHECK NUTRIENT AVAILABILITY ***********
111  ratiot=1.0
112  sinkk=0.
113  ratio=1.0
114  DO i1=1,nnn
115  DO i2=1,nnp
116  sinkk(i1)=sinkk(i1)+uptake_pn(k,i1,i2)
117  END DO
118  sinkk(i1)=sinkk(i1)*t_step
119  END DO
120  DO i1=1,nnn
121  IF (sinkk(i1).GT.fvnn(k,i1).AND.sinkk(i1).GT.0.0) THEN
122  ratio(i1)=fvnn(k,i1)/(sinkk(i1)+0.001)
123  ratiot=min(ratiot,ratio(i1))
124  END IF
125  END DO
126  IF (ratiot .LT. 1.0) THEN
127  DO i2=1,nnp
128  u_p(k,i2)=u_p(k,i2)*ratiot
129  DO i1=1,nnn
130  uptake_pn(k,i1,i2)=uptake_pn(k,i1,i2)*ratiot
131  END DO
132  END DO
133  END IF
134 
135 !**************** LIGHT-CONTROLLED NITRIFICATION *************
136  IF (no3_on) THEN
137  IF (irrad0.GT.0.) THEN
138  trans=l_nh4n-irrad1
139  IF (trans > 0.) THEN
140  nh4_no3(k)=r_an*trans*fvnn(k,1)/l_nh4n
141  ELSE
142  nh4_no3(k)=0.
143  END IF
144  ELSE
145  nh4_no3(k)=r_an*fvnn(k,1)
146  END IF
147  END IF
148  irrad0=irrad1 !For next level
149  10 CONTINUE
150 !******************* CELLS SINKING ***********
151  DO i1=1,nnp
152 ! WSNK_P(1,I1)=(-0.5)*W_P(I1)*(FVP(1,I1)+FVP(2,I1))/DELTA_D(1)
153  wsnk_p(1,i1)=(-1)*w_p(i1)*(fvp(1,i1))/delta_d(1)
154 ! WSNK_P(KBVM1,I1)=W_P(I1)*(FVP(KBV-2,I1)-FVP(KBVM1,I1))/DELTA_D(KBVM1) !Sink out
155  wsnk_p(kbvm1,i1)=w_p(i1)*(fvp(kbv-2,i1))/delta_d(kbvm1) !No sink out
156  END DO
157  DO k=2,kbv-2
158  DO i1=1,nnp
159  wsnk_p(k,i1)=w_p(i1)*0.5*(fvp(k-1,i1)-fvp(k+1,i1))/delta_d(k)
160 ! WSNK_P(K,I1)=W_P(I1)*(FVP(K-1,I1)-FVP(K,I1))/DELTA_D(K)
161  END DO
162  END DO
163 !************** PHYTOPLANKTON SOURCES AND SINKS **************
164  DO k=1,kbvm1
165  DO j=1,nnp
166  bio_p(k,j)=bio_p(k,j)+((1-d_dom(j))*u_p(k,j)+wsnk_p(k,j) & !Growth & Sinking
167  -p_dom(k,j)-p_d(k,j)-p_n(k,j))*t_step !Exudaiton, mortality, respiration
168  DO i2=1,nnz
169  bio_p(k,j)=bio_p(k,j)-g_p(k,j,i2)*t_step !GRAZING LOSSES
170  END DO
171  END DO
172  END DO
173  RETURN
174 END SUBROUTINE phytoplankton
