474 !
475 ! !INTERFACE:

477  subroutine init_state
478 !
479 ! !DESCRIPTION:
480 !
481 ! Initialize state for the itd model
482 !
483 ! !REVISION HISTORY:
484 !
485 ! author C. M. Bitz
486 !
487 ! !USES:
488 !
489  use ice_model_size
490  use ice_constants
491  use ice_flux
492  use ice_therm_vertical
493  use ice_grid
494  use ice_state
495  use ice_itd
496 !
497 ! !INPUT/OUTPUT PARAMETERS:
498 !
499 !EOP
500 !
501  integer (kind=int_kind) :: &
502  i, j & ! horizontal indices
503  , ij & ! horizontal index, combines i and j loops
504  , k & ! ice layer index
505  , ni & ! thickness category index
506  , icells ! number of cells initialized with ice
507 
508  integer (kind=int_kind), dimension(1:(ihi-ilo+1)*(jhi-jlo+1)) :: &
509  indxi, indxj ! compressed indices for cells with aicen > puny
510 
511  real (kind=dbl_kind) :: &
512  slope, Ti, sum, hbar &
513  , ainit(ncat) &
514  , hinit(ncat)
515 
516  real (kind=dbl_kind), parameter :: &
517  hsno_init = 0.20_dbl_kind ! initial snow thickness (m)
518 
519  if (trim(ice_ic) == 'none') then
520  ! Initialize grid with no ice.
521  ! If restarting, these values are overwritten.
522 
523  do ni = 1,ncat
524  do j = 1,jmt_local
525  do i = 1,imt_local
526  aicen(i,j,ni) = c0i
527  vicen(i,j,ni) = c0i
528  vsnon(i,j,ni) = c0i
529  esnon(i,j,ni) = c0i
530  enddo
531  enddo
532  do j = jlo,jhi
533  do i = ilo,ihi
534  tsfcn(i,j,ni) = tf(i,j) ! Tf not defined for ghost cells
535  enddo
536  enddo
537 ! call bound(Tsfcn(:,:,n))
538  enddo
539 
540  do k = 1,ntilay
541  do j = 1,jmt_local
542  do i = 1,imt_local
543  eicen(i,j,k) = c0i
544  enddo
545  enddo
546  enddo
547 
548  else ! ice_ic = 'default'
549 
550  ! initial category areas in cells with ice
551  hbar = c3i ! initial ice thickness with greatest area
552  ! Note: the resulting average ice thickness
553  ! tends to be less than hbar due to the
554  ! nonlinear distribution of ice thicknesses
555  sum = c0i
556  do ni = 1, ncat
557  if (ni < ncat) then
558  hinit(ni) = p5*(hin_max(ni-1) + hin_max(ni)) ! m
559  else ! n=ncat
560  hinit(ni) = (hin_max(ni-1) + c1i) ! m
561  endif
562  ! parabola, max at h=hbar, zero at h=0, 2*hbar
563  ainit(ni) = max(c0i, (c2i*hbar*hinit(ni) - hinit(ni)**2))
564  sum = sum + ainit(ni)
565  enddo
566  do ni = 1, ncat
567  ainit(ni) = ainit(ni) / (sum + puny/ncat) ! normalize
568  enddo
569 
570  ! place ice at high latitudes where ocean sfc is cold
571  icells = 0
572  do j = jlo,jhi
573  do i = ilo,ihi
574  if (tmask(i,j)) then
575  if ((sst(i,j) <= tf(i,j)+p2) .and. &
576  (ulat(i,j) < -64.0_dbl_kind/rad_to_deg .or. &
577 ! ULAT(i,j) > 70.0_dbl_kind/rad_to_deg)) then
578  ulat(i,j) > 65.0_dbl_kind/rad_to_deg)) then
579  icells = icells + 1
580  indxi(icells) = i
581  indxj(icells) = j
582  endif ! cold surface
583  endif ! tmask
