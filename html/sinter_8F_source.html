30 ! |
31 ! this subroutine linearly interpolates and extrapolates an |
32 ! array b. |
33 ! |
34 ! x(m1) must be descending |
35 ! a(x) given function |
36 ! b(y) found by linear interpolation and extrapolation |
37 ! y(n1) the desired depths |
38 ! m1 the number of points in x and a |
39 ! n1 the number of points in y and b |
40 ! |

48  SUBROUTINE sinter_extrp_up(X,A,Y,B,M1,N1)
49 
50 !==============================================================================|
51  USE mod_prec
52  IMPLICIT NONE
53  INTEGER, INTENT(IN) :: M1,N1
54  REAL(SP), INTENT(IN) :: X(m1),A(m1),Y(n1)
55  REAL(SP), INTENT(OUT) :: B(n1)
56  INTEGER I,J,NM
57 !==============================================================================|
58 
59 !
60 ! EXTRAPOLATION
61 !
62  DO i=1,n1
63  IF (y(i) > x(1 )) b(i) = a(1) + ((a(1)-a(2))/(x(1)-x(2))) * (y(i)-x(1))
64  IF (y(i) < x(m1)) b(i) = a(m1)
65  END DO
66 
67 !
68 ! INTERPOLATION
69 !
70  nm = m1 - 1
71  DO i=1,n1
72  DO j=1,nm
73  IF (y(i) <= x(j) .AND. y(i) >= x(j+1)) &
74  b(i) = a(j) - (a(j)- a(j+1)) * (x(j)-y(i)) / (x(j)-x(j+1))
75  END DO
76  END DO
77 
78  RETURN
79  END SUBROUTINE sinter_extrp_up
