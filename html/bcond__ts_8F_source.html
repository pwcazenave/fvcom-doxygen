30 ! Set Boundary Conditions on Temperature and Salinity |
31 ! NCON2 = 1: SET CONDITIONS SPECIFIC TO TEMPERATURE |
32 ! NCON2 = 2: SET CONDITIONS SPECIFIC TO SALINITY |
33 !==============================================================================|

35 SUBROUTINE bcond_ts(NCON2)
36 
37  !------------------------------------------------------------------------------|
38  USE all_vars
39  USE bcs
40  USE mod_utils
41  USE mod_obcs
42  USE mod_force
43 
44  IMPLICIT NONE
45  REAL(SP) :: S2D,S2D_NEXT,S2D_OBC,T2D,T2D_NEXT,T2D_OBC,XFLUX2D,TMP,RAMP_TS
46 
47 
48  INTEGER :: I,J,K,J1,J11,J22,NCON2
49  REAL(SP), ALLOCATABLE :: TTMP(:,:),STMP(:,:)
50 
51  REAL(SP) ::TMAX,TMIN,SMAX,SMIN
52 
53  !------------------------------------------------------------------------------|
54 
55  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: bcond_ts"
56 
57  !
58  !--SET CONDITIONS FOR FRESH WATER INFLOW---------------------------------------|
59  !
60  IF(river_ts_setting == 'specified') THEN
61  IF(numqbc > 0) THEN
62  IF(river_inflow_location == 'node') THEN
63  DO i=1,numqbc
64  j11=inodeq(i)
65  DO k=1,kbm1
66  tf1(j11,k)=tdis(i)
67  sf1(j11,k)=sdis(i)
68  END DO
69  END DO
70  ELSE IF(river_inflow_location == 'edge') THEN
71  DO i=1,numqbc
72  j11=n_icellq(i,1)
73  j22=n_icellq(i,2)
74  DO k=1,kbm1
75  tf1(j11,k)=tdis(i)
76  sf1(j11,k)=sdis(i)
77  tf1(j22,k)=tdis(i)
78  sf1(j22,k)=sdis(i)
79  END DO
80  END DO
81  END IF
82  END IF
83  END IF
84 
85 
86  IF(.NOT. obc_on) RETURN
87 
88 
89  !
90  ! SET TEMPERATURE CONDITIONS ON OUTER BOUNDARY
91  !
92  IF(ncon2 == 1) THEN
93 
94  IF(obc_temp_nudging) CALL update_obc_temp(inttime,temp_obc)
95 
96  ALLOCATE(ttmp(iobcn,kbm1)); ttmp = 0.0_sp
97  DO i=1,iobcn
98  j=i_obc_n(i)
99  j1=next_obc(i)
100  t2d=0.0_sp
101  t2d_next=0.0_sp
102  xflux2d=0.0_sp
103  DO k=1,kbm1
104  t2d=t2d+t1(j,k)*dz(j,k)
105  t2d_next=t2d_next+tf1(j1,k)*dz(j1,k)
106  xflux2d=xflux2d+xflux_obc(i,k) !*DZ(J,K)
107  END DO
108 
109  ! IF THE FLOW IS OUT OF THE DOMAIN
110  IF(uard_obcn(i) > 0.0_sp) THEN
111 
112  tmp=xflux2d+t2d*uard_obcn(i)
113  t2d_obc=(t2d*dt(j)-tmp*dti/art1(j))/d(j)
114 
115  CALL bcond_t_perturbation(t2d_next,t2d,ttmp,i,j,j1)
116 
117  DO k=1,kbm1
118  tf1(j,k)=t2d_obc+ttmp(i,k)
119  ! TF1(J,K)=T2D_OBC+(TF1(J1,K)-T2D_NEXT)
120  END DO
121 
122  DO k=1,kbm1
123  tmax = maxval(t1(nbsn(j,1:ntsn(j)),k))
124  tmin = minval(t1(nbsn(j,1:ntsn(j)),k))
125 
126  IF(k == 1)THEN
127  tmax = max(tmax,(t1(j,k)*dz(j,k+1)+t1(j,k+1)*dz(j,k))/ &
128  (dz(j,k)+dz(j,k+1)))
129  tmin = min(tmin,(t1(j,k)*dz(j,k+1)+t1(j,k+1)*dz(j,k))/ &
130  (dz(j,k)+dz(j,k+1)))
131  ELSE IF(k == kbm1)THEN
132  tmax = max(tmax,(t1(j,k)*dz(j,k-1)+t1(j,k-1)*dz(j,k))/ &
133  (dz(j,k)+dz(j,k-1)))
134  tmin = min(tmin,(t1(j,k)*dz(j,k-1)+t1(j,k-1)*dz(j,k))/ &
135  (dz(j,k)+dz(j,k-1)))
136  ELSE
137  tmax = max(tmax,(t1(j,k)*dz(j,k-1)+t1(j,k-1)*dz(j,k))/ &
138  (dz(j,k)+dz(j,k-1)), &
139  (t1(j,k)*dz(j,k+1)+t1(j,k+1)*dz(j,k))/ &
140  (dz(j,k)+dz(j,k+1)))
141  tmin = min(tmin,(t1(j,k)*dz(j,k-1)+t1(j,k-1)*dz(j,k))/ &
142  (dz(j,k)+dz(j,k-1)), &
143  (t1(j,k)*dz(j,k+1)+t1(j,k+1)*dz(j,k))/ &
144  (dz(j,k)+dz(j,k+1)))
145  END IF
146 
147  IF(tmin-tf1(j,k) > 0.0_sp)tf1(j,k) = tmin
148  IF(tf1(j,k)-tmax > 0.0_sp)tf1(j,k) = tmax
149 
150  END DO
151 
152  else! IF THE FLOW IS INTO THE DOMAIN
153 
154  IF(obc_temp_nudging) THEN
155  DO k=1,kbm1
156  tf1(j,k) = t1(j,k) - obc_temp_nudging_timescale*ramp*(t1(j,k)&
157  &-temp_obc(i,k))
158  END DO
159  ELSE
160  DO k=1,kbm1
161  tf1(j,k) = t1(j,k)
162  END DO
163  END IF
164 
165  END IF
166 
167  END DO
168  DEALLOCATE(ttmp)
169 
170 
171  !
172  ! SET SALINITY CONDITIONS ON OUTER BOUNDARY
173  !
174  ELSE IF(ncon2 == 2) THEN
175 
176  IF (obc_salt_nudging) CALL update_obc_salt(inttime,salt_obc)
177 
178  ALLOCATE(stmp(iobcn,kbm1)); stmp = 0.0_sp
179  DO i=1,iobcn
180  j=i_obc_n(i)
181  j1=next_obc(i)
182  s2d=0.0_sp
183  s2d_next=0.0_sp
184  xflux2d=0.0_sp
185  DO k=1,kbm1
186  s2d=s2d+s1(j,k)*dz(j,k)
187  s2d_next=s2d_next+sf1(j1,k)*dz(j1,k)
188  xflux2d=xflux2d+xflux_obc(i,k) !*DZ(J,K)
189  END DO
190 
191  ! IF THE FLOW IS OUT OF THE DOMAIN
192  IF(uard_obcn(i) > 0.0_sp) THEN
193  tmp=xflux2d+s2d*uard_obcn(i)
194  s2d_obc=(s2d*dt(j)-tmp*dti/art1(j))/d(j)
195 
196  CALL bcond_s_perturbation(s2d_next,s2d,stmp,i,j,j1)
197 
198  DO k=1,kbm1
199  sf1(j,k)=s2d_obc+stmp(i,k)
200  ! SF1(J,K)=S2D_OBC+(SF1(J1,K)-S2D_NEXT)
201  END DO
202 
203  DO k=1,kbm1
204  smax = maxval(s1(nbsn(j,1:ntsn(j)),k))
205  smin = minval(s1(nbsn(j,1:ntsn(j)),k))
206 
207  IF(k == 1)THEN
208  smax = max(smax,(s1(j,k)*dz(j,k+1)+s1(j,k+1)*dz(j,k))/ &
209  (dz(j,k)+dz(j,k+1)))
210  smin = min(smin,(s1(j,k)*dz(j,k+1)+s1(j,k+1)*dz(j,k))/ &
211  (dz(j,k)+dz(j,k+1)))
212  ELSE IF(k == kbm1)THEN
213  smax = max(smax,(s1(j,k)*dz(j,k-1)+s1(j,k-1)*dz(j,k))/ &
214  (dz(j,k)+dz(j,k-1)))
215  smin = min(smin,(s1(j,k)*dz(j,k-1)+s1(j,k-1)*dz(j,k))/ &
216  (dz(j,k)+dz(j,k-1)))
217  ELSE
218  smax = max(smax,(s1(j,k)*dz(j,k-1)+s1(j,k-1)*dz(j,k))/ &
219  (dz(j,k)+dz(j,k-1)), &
220  (s1(j,k)*dz(j,k+1)+s1(j,k+1)*dz(j,k))/ &
221  (dz(j,k)+dz(j,k+1)))
222  smin = min(smin,(s1(j,k)*dz(j,k-1)+s1(j,k-1)*dz(j,k))/ &
223  (dz(j,k)+dz(j,k-1)), &
224  (s1(j,k)*dz(j,k+1)+s1(j,k+1)*dz(j,k))/ &
225  (dz(j,k)+dz(j,k+1)))
226  END IF
227 
228  IF(smin-sf1(j,k) > 0.0_sp) sf1(j,k) = smin
229  IF(sf1(j,k)-smax > 0.0_sp) sf1(j,k) = smax
230 
231  END DO
232  ELSE ! IF THE FLOW IS INTO THE DOMAIN
233 
234  IF (obc_salt_nudging) THEN
235  DO k=1,kbm1
236  sf1(j,k) = s1(j,k) - obc_salt_nudging_timescale*ramp*(s1(j,k)&
237  &-salt_obc(i,k))
238  END DO
239  ELSE
240  DO k=1,kbm1
241  sf1(j,k) = s1(j,k)
242  END DO
243  END IF
244 
245  END IF
246  END DO
247  DEALLOCATE(stmp)
248  ELSE
249  print*, 'NCON2 NOT IN THE LIST'
250  print*, 'MUST BE 1 OR 2'
251  CALL pstop
252  END IF
253 
254 
255 
256  !
257  !--SET BOUNDARY CONDITIONS-----------------------------------------------------|
258  !
259  DO k=1,kbm1
260  t(0,k)=0.0_sp
261  s(0,k)=0.0_sp
262  END DO
263 
264  IF(dbg_set(dbg_sbr)) WRITE(ipt,*) "End: bcond_ts"
265 
266  RETURN
267 END SUBROUTINE bcond_ts
