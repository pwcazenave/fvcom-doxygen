47 ! !IROUTINE: scale_fluxes
48 !
49 ! !DESCRIPTION:
50 !
51 ! Divide ice fluxes by ice area before sending them to the
52 ! coupler, since the coupler multiplies by ice area. This
53 ! is the ice area at the beginning of the timestep, i.e.
54 ! the value sent to the coupler.
55 !
56 ! !INTERFACE:
57 !
58  subroutine scale_fluxes
59 !
60 ! !REVISION HISTORY:
61 !
62 ! author: C.M.Bitz ?
63 !
64 ! !USES:
65 !
66  use ice_albedo
67 !
68 ! !INPUT/OUTPUT PARAMETERS:
69 !
70 !EOP
71 !
72  real (kind=dbl_kind) :: ar ! 1/aice
73 
74  integer (kind=int_kind) :: i, j
75 
76  do j=jlo,jhi
77  do i=ilo,ihi
78  if (tmask(i,j) .and. aice(i,j) > c0i) then
79  ar = c1i/aice(i,j)
80 ! strairxT(i,j) = strairxT(i,j) * ar
81 ! strairyT(i,j) = strairyT(i,j) * ar
82 ! the dynamic is not considered
83 
84  fsens (i,j) = fsens (i,j) * ar
85  flat (i,j) = flat (i,j) * ar
86  fswabs(i,j) = fswabs(i,j) * ar
87  flwout(i,j) = flwout(i,j) * ar
88  evap (i,j) = evap (i,j) * ar
89  tref (i,j) = tref (i,j) * ar
90  qref (i,j) = qref (i,j) * ar
91  fresh (i,j) = fresh (i,j) * ar
92  fsalt (i,j) = fsalt (i,j) * ar
93  fhnet (i,j) = fhnet (i,j) * ar
94  fswthru(i,j) = fswthru(i,j) * ar
95  else ! zero out fluxes
96  strairxt(i,j) = c0i
97  strairyt(i,j) = c0i
98 ! dynamic part using fvcom code
99  fsens (i,j) = c0i
100  flat (i,j) = c0i
101  fswabs(i,j) = c0i
102  flwout(i,j) = -stefan_boltzmann*(tffresh - tf(i,j))**4
103  ! to make upward longwave over ocean
104  ! reasonable for history file
105  evap (i,j) = c0i
106  tref (i,j) = tair(i,j)
107  qref (i,j) = qa(i,j)
108  fresh (i,j) = c0i
109  fsalt (i,j) = c0i
110  fhnet (i,j) = c0i
111  fswthru(i,j) = c0i
112  alvdf (i,j) = c0i ! zero out albedo where ice is absent
113  alidf (i,j) = c0i
114  alvdr (i,j) = c0i
115  alidr (i,j) = c0i
116  endif
117  enddo
118  enddo
119 
120  end subroutine scale_fluxes
