257  SUBROUTINE algorithm_crank_nicolson(CAD,IG,DTW,IDCMIN,IDCMAX,DD)

266  REAL :: cad(:,:,:)
267  REAL :: dtw,dd
268  REAL :: n32m,n32p,idp1

270  REAL,DIMENSION(MDC) :: a,b,c,r,u

272 ! IF(ALLOCATED(N32)) DEALLOCATE(N32)
273 ! ALLOCATE(N32(MDC,MSC,MCGRD)); N32 = 0.0
274 
275  DO iss = 1,msc
276  DO iddum = idcmin(iss),idcmax(iss)
277  id = mod(iddum-1+mdc,mdc)+1
278  idp1 = mod(iddum+mdc,mdc)+1
279  idm1 = mod(iddum-2+mdc,mdc)+1
280 
281  b(id) = 1.0
282  IF(id == 1)THEN
283  a(id) = 0.0
284  ELSE
285  a(id) = -0.5*zeta*dtw*cad(idm1,iss,1)/dd
286  END IF
287  IF(id == mdc)THEN
288  c(id) = 0.0
289  ELSE
290  c(id) = 0.5*zeta*dtw*cad(idp1,iss,1)/dd
291  END IF
292 
293  IF(id == 1)THEN
294  r(id) = cad(idp1,iss,1)*n31(idp1,iss)
295  r(id) = (1.0-zeta)*0.5*dtw*r(id)/dd
296  r(id) = n31(id,iss)-r(id)
297  ELSE IF(id == mdc)THEN
298  r(id) = -cad(idm1,iss,1)*n31(idm1,iss)
299  r(id) = (1.0-zeta)*0.5*dtw*r(id)/dd
300  r(id) = n31(id,iss)-r(id)
301  ELSE
302  r(id) = cad(idp1,iss,1)*n31(idp1,iss)-cad(idm1,iss,1)*n31(idm1,iss)
303  r(id) = (1.0-zeta)*0.5*dtw*r(id)/dd
304  r(id) = n31(id,iss)-r(id)
305  END IF
306  END DO
307 
308  CALL tridag(a,b,c,r,u,mdc)
309 
310  DO iddum = idcmin(iss),idcmax(iss)
311  id = mod(iddum-1+mdc,mdc)+1
312  n32(id,iss,ig) = u(id)
313  END DO
314  END DO
315 
316  RETURN
317  END SUBROUTINE algorithm_crank_nicolson

321  SUBROUTINE tridag(A,B,C,R,U,N)
322  IMPLICIT NONE
323  INTEGER :: n,j
324  REAL,DIMENSION(N) :: a,b,c,r,u
325  INTEGER, PARAMETER :: nmax = 500
326  REAL bet,gam(nmax)
327 
328  IF(b(1) == 0.)pause 'TRIDAG: REWRITE EQUATIONS'
329  bet = b(1)
330  u(1) = r(1)/bet
331  DO j=2,n
332  gam(j) = c(j-1)/bet
333  bet = b(j)-a(j)*gam(j)
334  IF(bet == 0.)pause 'TRIDAG FAILED'
335  u(j) = (r(j)-a(j)*u(j-1))/bet
336  END DO
337  DO j=n-1,1,-1
338  u(j) = u(j)-gam(j+1)*u(j+1)
339  END DO
340 
341  RETURN
342  END SUBROUTINE tridag
