29 !==============================================================================|
30 ! this subroutine is used to calculate the q2 and q2l by solving !
31 ! the vertical diffusion equation implicitly. !
32 !==============================================================================|
33 
34  SUBROUTINE vdif_q
35 
36 !------------------------------------------------------------------------------|
37  USE all_vars
38  USE mod_utils
39  USE mod_wd
40  USE mod_par
41 # if defined (SEMI_IMPLICIT)
42  USE mod_semi_implicit, ONLY : stage, kstage_te
43 # endif
44 
45 # if defined (WAVE_CURRENT_INTERACTION)
46  USE mod_wave_current_interaction
47 # endif
48 
49 # if defined (THIN_DAM)
50  USE mod_dam
51 # endif
52 
53  IMPLICIT NONE
54  INTEGER :: I,J,K,KI
55  REAL(SP) :: CONST1,COEF1,COEF2,COEF3,COEF4,COEF5,LMAX
56  REAL(SP), DIMENSION(0:MT,KB) :: GH,SM,SH,PROD,DTEF,KN,BOYGR,A,C,VHP,VH,STF
57  REAL(SP), DIMENSION(0:MT) :: L0
58  REAL(SP) :: UTAU2
59  REAL(SP) :: WUSURF_NODE,WVSURF_NODE,WUBOT_NODE,WVBOT_NODE
60  REAL(SP) :: UU_NODE_K,VV_NODE_K,UU_NODE_KM1,VV_NODE_KM1
61 
62 !--Stability Function Coefficients-------------
63  REAL(SP), PARAMETER :: A1 = 0.92_sp
64  REAL(SP), PARAMETER :: B1 = 16.6_sp
65  REAL(SP), PARAMETER :: A2 = 0.74_sp
66  REAL(SP), PARAMETER :: B2 = 10.1_sp
67  REAL(SP), PARAMETER :: C1 = 0.08_sp
68 
69 !--Source/Sink Term Coefficients---------------
70  REAL(SP), PARAMETER :: E1 = 1.80_sp
71  REAL(SP), PARAMETER :: E2 = 1.33_sp
72  REAL(SP), PARAMETER :: SEF = 1.00_sp
73 
74 !--Von Karmans Constant------------------------
75  REAL(SP), PARAMETER :: KAPPA = 0.40_sp
76 
77 !--Gravitational Constant----------------------
78  REAL(SP), PARAMETER :: GEE = 9.806_sp
79 
80 !--Limiting Values of Q2/Q2L-------------------
81  REAL(SP), PARAMETER :: SMALL = 1.e-8_sp
82 
83 !--Coefficient of buoyancy length scale--------
84  REAL(SP), PARAMETER :: CB = 1.e-8_sp
85 
86 !--Denominator small number threshold----------
87  REAL(SP), PARAMETER :: DEN_S = 1.e-10_sp
88 
89 !--Upper bound of GH for unstable flow---------
90  REAL(SP), PARAMETER :: GH_MAX = .0233_sp
91 
92 !--Lower bound for GH for stable flow----------
93  REAL(SP), PARAMETER :: GH_MIN = -.281_sp
94 
95 # if defined (WAVE_CURRENT_INTERACTION)
96  REAL(SP) :: WIND10,CDD,USTAR,CPP,WAVE_AGE,ALFA_CB
97  REAL(SP) :: WUWIND_NODE,WVWIND_NODE
98 # endif
99 
100 # if defined (WAVE_CURRENT_INTERACTION)
101 !--??????????????????????????????????----------
102 ! REAL(SP), PARAMETER :: HLC = 2.4_SP
103 # endif
104 
105 !----------------------------------------------
106  REAL(SP), PARAMETER :: CBCNST = 100.0_sp
107  REAL(SP), PARAMETER :: SURFL = 2.e+5_sp
108  REAL(SP), PARAMETER :: SHIW = 0.0_sp
109  REAL(SP), PARAMETER :: GHC = -6.0_sp
110 
111 # if defined (WET_DRY)
112  REAL(SP) :: KQ_TMP,KM_TMP,KH_TMP
113 # endif
114 
115  IF(dbg_set(dbg_sbr)) write(ipt,*) "Start: vdif_q"
116 !------------------------------------------------------------------------------|
117 
118  ! INITIALIZE MEMORY
119  boygr = 0.0_sp
120  gh = 0.0_sp
121  sm = 0.0_sp
122  sh = 0.0_sp
123  prod = 0.0_sp
124  dtef = 0.0_sp
125  kn = 0.0_sp
126  a = 0.0_sp
127  c = 0.0_sp
128  vhp = 0.0_sp
129  vh = 0.0_sp
130  stf = 0.0_sp
131  l0 = 0.0_sp
132 
133 !------------------------------------------------------------------------------|
134 ! place a threshold on the turbulence quantities following advection |
135 !------------------------------------------------------------------------------|
136 
137  DO k = 2, kbm1
138  DO i = 1, m
139  IF (q2f(i,k) <= small .OR. q2lf(i,k) <= small) THEN
140 ! Q2F(I,K) = 0.0_SP
141 ! Q2LF(I,K) = 0.0_SP
142  q2f(i,k) = small
143  q2lf(i,k) = small
144  END IF
145  END DO
146  END DO
147 
148 !------------------------------------------------------------------------------!
149 ! set up coefficients for implicit calculation of vertical diffusion !
150 !------------------------------------------------------------------------------!
151 
152  DO i=1,m
153 # if defined (WET_DRY)
154  IF(iswetn(i) == 1)THEN
155 # endif
156  DO k=2,kbm1
157  a(i,k) = -0.5_sp* dti * (kq(i,k+1)+kq(i,k)+2.0_sp*umol)/ &
158  (dzz(i,k-1)*dz(i,k)*d(i)*d(i))
159  c(i,k) = -0.5_sp* dti*(kq(i,k-1)+kq(i,k)+2.0_sp*umol)/ &
160  (dzz(i,k-1)*dz(i,k-1)*d(i)*d(i))
161  END DO
162 # if defined (WET_DRY)
163  END IF
164 # endif
165  END DO
166 
167 !------------------------------------------------------------------------------!
168 ! the following section solves the equation !
169 ! dti*(kq*q2')' - q2*(2.*dti*dtef+1.) = -q2b !
170 !------------------------------------------------------------------------------!
171 
172  const1 = 16.6_sp ** .6666667_sp * sef
173  DO i = 1, m
174 ! VHP(I,1) = SQRT(WUSURF(I)**2+WVSURF(I)**2) * CONST1
175 ! VH(I,1) = 0.0_SP
176 
177  wusurf_node = 0.0_sp
178  wvsurf_node = 0.0_sp
179  DO j=1,ntve(i)
180  wusurf_node = wusurf_node + wusurf(nbve(i,j))
181  wvsurf_node = wvsurf_node + wvsurf(nbve(i,j))
182  END DO
183  wusurf_node = wusurf_node/float(ntve(i))
184  wvsurf_node = wvsurf_node/float(ntve(i))
185 
186 # if defined (WAVE_CURRENT_INTERACTION)
187 !!$ UTAU2 = SQRT(WUSURF_NODE**2+WVSURF_NODE**2)
188 !!$ VHP(I,1) = (15.8_SP*CBCNST)**0.6666667_SP*UTAU2
189 !!$ VH(I,1) = 0.0_SP
190 !!$! L0(I) = SURFL*UTAU2/GEE
191 !!$ L0(I) = SURFL*UTAU2/GRAV_N(I)
192  utau2 = sqrt(wusurf_node**2+wvsurf_node**2)
193 
194  wuwind_node = 0.0_sp
195  wvwind_node = 0.0_sp
196  DO j=1,ntve(i)
197  wuwind_node = wuwind_node + uwwind(nbve(i,j))
198  wvwind_node = wvwind_node + vwwind(nbve(i,j))
199  END DO
200  wuwind_node = wuwind_node/float(ntve(i))
201  wvwind_node = wvwind_node/float(ntve(i))
202 
203  wind10=wuwind_node*wuwind_node + wvwind_node*wvwind_node
204  wind10=sqrt(wind10)
205  IF(wind10 < 7.5)THEN
206  cdd = 1.2875e-3_sp
207  ELSE
208  cdd = (0.8+0.065*wind10)*1.0e-3_sp
209  END IF
210  ustar=cdd*wind10*wind10
211  ustar=sqrt(ustar)
212 
213  cpp = wlen(i)/(tpeak(i) + small)
214 
215  wave_age = cpp/(ustar + small)
216  alfa_cb = 0.04_sp*wave_age
217  alfa_cb = alfa_cb**4.0_sp
218  alfa_cb = exp(-alfa_cb)
219  alfa_cb = 15.0_sp*wave_age*alfa_cb
220 
221  vhp(i,1) = alfa_cb*utau2**1.5_sp !1.5=3.0/2.0
222  vh(i,1) = 0.0_sp
223  l0(i) = 0.85*hsc1(i)
224 
225 # else
226  utau2 = sqrt(wusurf_node**2+wvsurf_node**2)
227  vhp(i,1) = (15.8_sp*cbcnst)**0.6666667_sp*utau2
228  vh(i,1) = 0.0_sp
229 ! L0(I) = SURFL*UTAU2/GEE
230  l0(i) = surfl*utau2/grav_n(i)
231 ! HLC = 2.4_SP
232 ! L0(I) = HLC*HSC1(I)
233 # endif
234 
235 
236  wubot_node = 0.0_sp
237  wvbot_node = 0.0_sp
238  DO j=1,ntve(i)
239  wubot_node = wubot_node + wubot(nbve(i,j))
240  wvbot_node = wvbot_node + wvbot(nbve(i,j))
241  END DO
242  wubot_node = wubot_node/float(ntve(i))
243  wvbot_node = wvbot_node/float(ntve(i))
244 
245  q2f(i,kb) = sqrt(wubot_node**2+wvbot_node**2) * const1
246  END DO
247 
248  q2 = abs(q2+small)
249  q2l = abs(q2l)
250 
251 !------------------------------------------------------------------------------!
252 ! calculate boygr = -Brunt Vaisala Frequency^2 !
253 ! calculate internal wave shear energy contribution: prod = 200*(b.v.freq)**3 !
254 !------------------------------------------------------------------------------!
255 
256  DO k=2,kbm1
257  DO i=1,m
258 # if defined (WET_DRY)
259  IF(iswetn(i) == 1)THEN
260 # endif
261 ! BOYGR(I,K) = GEE * (RHO1(I,K-1)-RHO1(I,K))/(DZZ(I,K-1)*D(I))
262  boygr(i,k) = grav_n(i) * (rho1(i,k-1)-rho1(i,k))/(dzz(i,k-1)*d(i))
263  prod(i,k) = km(i,k) * 0.0_sp * (.5_sp*(-boygr(i,k)+abs(boygr(i,k)))) ** 1.5_sp
264 # if defined (WET_DRY)
265  END IF
266 # endif
267  END DO
268  END DO
269 
270 !------------------------------------------------------------------------------!
271 ! calculate shear production source term = prod !
272 ! calculate buoyancy production source term = kh*boygr !
273 !------------------------------------------------------------------------------!
274  DO k = 2, kbm1
275  DO i = 1, m
276 # if !defined (WET_DRY)
277  IF (d(i) > 0.0_sp) THEN
278 # else
279  IF(iswetn(i) == 1)THEN
280 # endif
281 
282  uu_node_k = 0.0_sp
283  vv_node_k = 0.0_sp
284  uu_node_km1 = 0.0_sp
285  vv_node_km1 = 0.0_sp
286  DO j=1,ntve(i)
287  uu_node_k = uu_node_k + u(nbve(i,j),k)
288  vv_node_k = vv_node_k + v(nbve(i,j),k)
289  uu_node_km1 = uu_node_km1 + u(nbve(i,j),k-1)
290  vv_node_km1 = vv_node_km1 + v(nbve(i,j),k-1)
291  END DO
292  uu_node_k = uu_node_k/float(ntve(i))
293  vv_node_k = vv_node_k/float(ntve(i))
294  uu_node_km1 = uu_node_km1/float(ntve(i))
295  vv_node_km1 = vv_node_km1/float(ntve(i))
296 
297  prod(i,k) = prod(i,k) + km(i,k) * sef * &
298  ((uu_node_k-uu_node_km1)**2+(vv_node_k-vv_node_km1)**2)/ &
299  (dzz(i,k-1)*d(i))**2
300  prod(i,k) = prod(i,k) + kh(i,k) * boygr(i,k)
301  END IF
302  END DO
303  END DO
304 !------------------------------------------------------------------------------!
305 ! solve for turbulent length scale l = q2l/q2 !
306 !------------------------------------------------------------------------------!
307 
308 ! L = Q2L/Q2
309  DO k=2,kbm1
310  DO i=1,m
311  l(i,k) = q2l(i,k)/q2(i,k)
312  END DO
313  END DO
314 
315 !------------------------------------------------------------------------------!
316 ! in stably stratified regions, length scale is limited by buoyancy length !
317 ! scale, l_b = c_b*(k^.5/N). length scale limitation also puts an effective !
318 ! lower bound of -.281 on gh in stably stratified regions !
319 ! length scale near surface is modified with wind mixed length scale !
320 !------------------------------------------------------------------------------!
321 
322  DO k=2,kbm1
323  DO i=1,m
324  IF(z(i,k) > -0.5_sp)l(i,k) = max(l(i,k),kappa*l0(i))
325  IF(boygr(i,k) < 0.0_sp)THEN
326  lmax = sqrt(abs(gh_min) * q2(i,k) / (-boygr(i,k) + small) )
327  l(i,k) = min(l(i,k),lmax)
328  q2l(i,k) = q2(i,k)*l(i,k)
329  END IF
330  END DO
331  END DO
332 
333 !------------------------------------------------------------------------------!
334 ! solve for gh = (-l^2*N^2/q^2) !
335 !------------------------------------------------------------------------------!
336 
337 !! GH = (L**2/Q2)*BOYGR
338  gh(1:mt,1:kbm1) = (l(1:mt,1:kbm1)**2/q2(1:mt,1:kbm1))*boygr(1:mt,1:kbm1)
339 
340 !------------------------------------------------------------------------------!
341 ! limit maximum of GH in unstable regions to reasonable physical value !
342 ! limiting GH also constrains stability functions SH/SM to finite values !
343 !------------------------------------------------------------------------------!
344 
345  gh = min(gh,gh_max)
346 
347  l(:,1) = kappa*l0(:) !0.0_SP
348  l(:,kb) = 0.0_sp
349  gh(:,1) = 0.0_sp
350  gh(:,kb) = 0.0_sp
351 
352 
353 !------------------------------------------------------------------------------!
354 ! calculate eddy kinetic energy dissipation rate: dtef !
355 !------------------------------------------------------------------------------!
356 
357  stf = 1.0_sp
358 
359  IF(.not.surface_wave_mixing)THEN
360  DO k = 1,kb
361  DO i = 1,m
362  IF(gh(i,k) < 0.0_sp) stf(i,k)=1.0-0.9*(gh(i,k)/ghc)**1.5
363  IF(gh(i,k) < ghc) stf(i,k) = 0.1
364  END DO
365  END DO
366  END IF
367 
368 ! DTEF = Q2*SQRT(Q2)/(B1*Q2L + small)*STF
369  dtef = sqrt(q2)/(b1*l + small)*stf
370 
371  DO k = 2, kbm1
372  DO i = 1, m
373  vhp(i,k) = 1. / (a(i,k)+c(i,k)*(1.-vh(i,k-1))-(2.*dti*dtef(i,k)+1.))
374  vh(i,k) = a(i,k) * vhp(i,k)
375  vhp(i,k) = (-2.*dti*prod(i,k)+c(i,k)*vhp(i,k-1)-q2f(i,k))*vhp(i,k)
376  END DO
377  END DO
378 
379  DO k=1,kbm1
380  ki = kb-k
381  DO i = 1, m
382 # if defined (WET_DRY)
383  IF(iswetn(i) == 1)THEN
384 # endif
385  q2f(i,ki) = vh(i,ki) * q2f(i,ki+1) + vhp(i,ki)
386 # if defined (WET_DRY)
387  END IF
388 # endif
389  END DO
390  END DO
391 
392 !------------------------------------------------------------------------------!
393 ! the following section solves the equation !
394 ! dti*(kq*q2l')' - q2l*(dti*dtef+1.) = -q2lb !
395 !------------------------------------------------------------------------------!
396 
397  vh(:,1) = 0.0_sp
398  vhp(:,1) = 0.0_sp
399 
400  DO k = 2, kbm1
401  DO i = 1, m
402 # if !defined (WET_DRY)
403  IF (d(i) > 0.0_sp) THEN
404 # else
405  IF(iswetn(i) == 1)THEN
406 # endif
407  dtef(i,k) = dtef(i,k) * (1.+e2*((1./abs(z(i,k)-z(i,1))+1./ &
408  abs(z(i,k)-z(i,kb)))*l(i,k)/(d(i)*kappa))**2)
409  vhp(i,k) = 1. / (a(i,k)+c(i,k)*(1.-vh(i,k-1))- &
410  (dti*dtef(i,k)+1.))
411  vh(i,k) = a(i,k) * vhp(i,k)
412  vhp(i,k) = (dti*(-prod(i,k)*l(i,k)*e1)+c(i,k)* &
413  vhp(i,k-1)-q2lf(i,k)) * vhp(i,k)
414  END IF
415  END DO
416  END DO
417 
418 
419  DO k=1,kbm1
420  ki = kb - k
421  DO i = 1, m
422 # if defined (WET_DRY)
423  IF(iswetn(i) == 1)THEN
424 # endif
425  q2lf(i,ki) = vh(i,ki) * q2lf(i,ki+1) + vhp(i,ki)
426 # if defined (WET_DRY)
427  END IF
428 # endif
429  END DO
430  END DO
431 
432 !------------------------------------------------------------------------------|
433 ! place a threshold on the turbulence quantities following vertical diffusion |
434 !------------------------------------------------------------------------------|
435 
436 # if defined (SEMI_IMPLICIT)
437  IF(stage==kstage_te) THEN
438 # endif
439 
440  DO k = 2, kbm1
441  DO i = 1, m
442  IF (q2f(i,k) <= small .OR. q2lf(i,k) <= small) THEN
443  q2f(i,k) = small
444  q2lf(i,k) = small
445  END IF
446  END DO
447  END DO
448 
449 
450 !===========THRESHOLD ON LENGTH SCALE ALT STYLE (Active in Barotropic Case)====
451 ! DO K=2,KBM1
452 ! DO I=1,N
453 ! LMAX = SQRT(ABS(GH_MIN) * Q2F(I,K) / MAX(0.0_SP,-BOYGR(I,K)+SMALL))
454 ! L(I,K) = MIN(L(I,K),LMAX)
455 ! Q2LF(I,K) = Q2F(I,K)*L(I,K)
456 ! END DO
457 ! END DO
458 !------------------------------------------------------------------------------!
459 
460 ! L(:,1) = 0.0_SP
461 ! L(:,KB) = 0.0_SP
462 ! GH(:,1) = 0.0_SP
463 ! GH(:,KB) = 0.0_SP
464 
465 
466 !------------------------------------------------------------------------------!
467 ! calculate stability functions sh + sm using Galperin 1988 formulation !
468 !------------------------------------------------------------------------------!
469  coef4 = 18.0_sp * a1 * a1 + 9.0_sp * a1 * a2
470  coef5 = 9.0_sp * a1 * a2
471 
472  DO k = 1,kb
473  DO i = 1,m
474  coef1 = a2 * (1.0_sp-6.0_sp*a1/b1*stf(i,k))
475  coef2 = 3.0_sp * a2 * b2/stf(i,k) + 18.0_sp * a1 * a2
476  coef3 = a1 * (1.0_sp-3.0_sp*c1-6.0_sp*a1/b1*stf(i,k))
477 
478  sh(i,k) = coef1/(1.0_sp-coef2*gh(i,k))
479  sm(i,k) = (coef3+sh(i,k)*coef4*gh(i,k))/(1.-coef5*gh(i,k))
480  END DO
481  END DO
482 
483 !------------------------------------------------------------------------------!
484 ! calculate turbulent eddy viscosities/diffusivities kq,km,kh !
485 !------------------------------------------------------------------------------!
486 
487  kn = l*sqrt(q2)
488  kq = 0.5_sp*(kn*kappa*sm+kq)
489  km = 0.5_sp*(kn*sm+km)
490  kh = 0.5_sp*(kn*sh*vprnu+kh)
491 
492 # if defined (WET_DRY)
493  DO i=1,m
494  IF(h(i) <= static_ssh_adj)THEN
495  kq_tmp = 0.0_sp
496  km_tmp = 0.0_sp
497  kh_tmp = 0.0_sp
498  DO k=2,kbm1
499  kq_tmp = kq_tmp + kq(i,k)
500  km_tmp = km_tmp + km(i,k)
501  kh_tmp = kh_tmp + kh(i,k)
502  END DO
503  kq_tmp = kq_tmp/(kbm1-1)
504  km_tmp = km_tmp/(kbm1-1)
505  kh_tmp = kh_tmp/(kbm1-1)
506  DO k=2,kbm1
507  kq(i,k) = kq_tmp
508  km(i,k) = km_tmp
509  kh(i,k) = kh_tmp
510  END DO
511  END IF
512  END DO
513 # endif
514 
515 # if defined (THIN_DAM)
516  DO k=1,kbm1
517  DO i=1,node_dam1_n
518  IF(k<=kdam(i_node_dam1_n(i,1)).AND.k<=kdam(i_node_dam1_n(i,2)))THEN
519  kh(i_node_dam1_n(i,1),k) = sum(kh(i_node_dam1_n(i,1:2),k))/2.0
520  kh(i_node_dam1_n(i,2),k) = kh(i_node_dam1_n(i,1),k)
521  END IF
522  END DO
523  DO i=1,node_dam2_n
524  IF(k<=kdam(i_node_dam2_n(i,1)).AND.k<=kdam(i_node_dam2_n(i&
525  &,2)).AND.k<=kdam(i_node_dam2_n(i,3)))THEN
526  kh(i_node_dam2_n(i,1),k) = sum(kh(i_node_dam2_n(i,1:3),k))/3.0
527  kh(i_node_dam2_n(i,2),k) = kh(i_node_dam2_n(i,1),k)
528  kh(i_node_dam2_n(i,3),k) = kh(i_node_dam2_n(i,1),k)
529  END IF
530  END DO
531  DO i=1,node_dam3_n
532  IF(k<=kdam(i_node_dam3_n(i,1)).AND.k<=kdam(i_node_dam3_n(i&
533  &,2)).AND.k<=kdam(i_node_dam3_n(i,3)).AND.k<=kdam(i_node_dam3_n(i,4)))THEN
