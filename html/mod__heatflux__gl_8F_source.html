172  SUBROUTINE coare26z(UR,ZU,TA,ZT,RH,ZQ,PA,TS,DLW,DSW,TAU,HSB,HLB,LAT,USR,DTER) !mdr include USR, DTER
173 ! A=coare26(u,zu,Ta,zt,rh,zq,Pa,Ts,dlw,dsw)
174 ! Simplified non-vectorized version of coare2.6 code
175 ! with cool skin option retained but warm layer and
176 ! surface wave options removed, and rain set to zero.
177 ! Assumes input are single scalars and that u is the
178 ! magitude of the difference between wind and surface
179 ! current vectors, if latter available. Output:
180 ! A = [tau qsen qlat Cd Ch Ce Cdn_10 Chn_10 Cen_10].
181 
182  REAL(SP) :: ur,zu,ta,zt,rh,zq,pa,ts,dlw,dsw,tau,hsb,hlb,lat,le,l,jcool,l10
183  REAL(SP) :: u,us,t,p,rl,rs,qsat26s,qs,q,zi,beta,von,fdg,tdk,grvs,grav,rgas
184  REAL(SP) :: cpa,cpv,rhoa,visa,al,be,cpw,rhow,visw,tcw,bigc,wetc,rns,rnl,du
185  REAL(SP) :: dt,dq,ug,dter,dqer,ut,u10,usr,zo10,cd10,ch10,ct10,zot10,cd,ct,cc
186  REAL(SP) :: ribcu,ribu,nits,zetu,psiu_26s,psit_26s,tsr,qsr,tkt,charn,zet
187  REAL(SP) :: zo,zoq,zot,bf,qout,dels,qcol,alq,xlamx,ch,ce,cdn_10,chn_10,cen_10,rr
188 
189  INTEGER :: i
190 ! Set jcool=0 if Ts is surface, =1 if Ts is bulk.
191 ! rcb checked 6/9/04
192 ! set jcool=1 if Ts is bulk, 0 if Ts is true skin jcool=1;
193  jcool=1.
194 ! rename variables from fairall et al coare3 code
195 
196 
197 ! wind speed (m/s) at height zr (m)
198  u=ur
199 ! surface current speed in the wind direction(m/s)
200  us=0.*ur
201 ! water temperature (deg C)
202  ts=ts
203 ! BULK AIR TEMPERATURE (C) AT HEIGHT ZT(m)
204  t=ta
205 ! RELATIVE HUMIDITY (%) AT HEIGHT zq(M)
206  rh=rh
207 ! SURFACE PRESSURE (mb)
208 ! P=PA
209  p=pa/100. !ejw 8/15/2011 from pa to mb in this subroutine only
210 ! DOWNWARD LONGWAVE RADIATION (W/m2)
211  rl=dlw
212 ! DOWNWARD SHORTWAVE RADIATION (W/m2)
213  rs=dsw
214 ! CONVERT RH TO SPECIFIC HUMIDITY (G/KG)
215  CALL qsat26(ts,p,qsat26s)
216 !MDR QS=0.98_SP*QSAT26S/1000._SP
217  qs=qsat26s/1000._sp !MDR, don't apply 0.98 for freshwater
218 ! SPECIFIC HUMIDITY OF AIR (G/KG)
219  CALL qsat26(t,p,qsat26s)
220  q=(0.01_sp*rh)*qsat26s/1000._sp
221 ! print *,'qq',Q
222 ! SET RAIN TO ZERO RAIN=0*U
223 ! SET RAIN RATE (MM/HR) - KEEP AS OPTION
224 
225 ! ***********SET LOCAL CONSTANTS *********
226 ! PBL HEIGHT (M)
227  zi=600._sp
228 ! LATITUDE (DEG,N=+)- GEORGES BANK
229 ! LAT=42.
230 ! ************SET CONSTANTS **************
231  beta=1.2_sp
232  von=0.4_sp
233  fdg=1.00_sp
234  tdk=273.16_sp
235  CALL grv(lat,grvs)
236  grav=grvs
237 ! ************AIR CONSTANTS **************
238  rgas=287.1_sp
239  le=(2.501_sp-0.00237_sp*ts)*1000000._sp
240  cpa=1004.67_sp
241  cpv=cpa*(1._sp+0.84_sp*q)
242  rhoa=p*100._sp/(rgas*(t+tdk)*(1+0.61_sp*q))
243  visa=1.326_sp*0.00001_sp*(1+6.542_sp*0.001_sp*t+ &
244  8.301_sp*0.000001_sp*t*t-4.84_sp*0.000000001_sp*t*t*t)
245 ! ***********COOL SKIN CONSTANTS******************
246  al=2.1_sp*0.00001_sp*((ts+3.2_sp)**0.79_sp)
247 !!MDR salinity expansion coefficient, BE, to zero for freshwater
248 ! confirmed by email with CW Fairall 4-24-2013
249 !MDR BE=0.026_SP
250  be=0.0_sp
251  cpw=4000._sp
252 !MDR RHOW=1022._SP
253  rhow=1000._sp !MDR, freshwater density
254  visw=0.000001_sp
255  tcw=0.6_sp
256  bigc=16._sp*grav*cpw*((rhow*visw)**3)/(tcw*tcw*rhoa*rhoa)
257  wetc=0.622_sp*le*qs/(rgas*((ts+tdk)**2))
258 ! ***************COMPUTE AUX STUFF *********
259 ! MDR RNS=RS*0.945_SP
260  rns=rs !Mark Rowe, remove albedo here because albedo is included in forcings
261  rnl=0.97_sp*(5.67_sp*0.00000001_sp*((ts-0.3_sp*jcool+tdk)**4)-rl)
262 
263 
264 ! **************BEGIN BULK LOOP ***********
265 
266 ! **************FIRST GUESS ***************
267  du=u-us
268  dt=ts-t-0.0098_sp*zt
269  dq=qs-q
270  ta=t+tdk
271  ug=0.5_sp
272  dter=0.3_sp
273  dqer=wetc*dter
274  ut=sqrt(du*du+ug*ug)
275  u10=ut*log(10._sp/1e-4)/log(zu/1e-4)
276  usr=0.035_sp*u10
277 
278  zo10=0.011_sp*usr*usr/grav+0.11_sp*visa/usr
279  cd10=(von/log(10._sp/zo10))**2
280  ch10=0.00115_sp
281  ct10=ch10/sqrt(cd10)
282  zot10=10._sp/exp(von/ct10)
283  cd=(von/log(zu/zo10))**2
284  ct=von/log(zt/zot10)
285  cc=von*ct/cd
286  ribcu=-zu/(zi*0.004_sp*(beta**3))
287  ribu=-grav*zu*((dt-dter*jcool)+.61_sp*ta*dq)/(ta*(ut**2))
288 ! same as edson
289 !MDR NITS=6._SP
290  nits=3._sp !MDR set NITS to 3, after F90 version of COARE 3.0
291  IF(ribu < 0)THEN
292  zetu=cc*ribu/(1._sp+ribu/ribcu)
293  ELSE
294 !MDR ZETU=CC*RIBU/(1._SP+27._SP/(9*RIBU*CC))
295  zetu=cc*ribu*(1.0+27.0/9.0*ribu/cc) !MDR, corrected from F90 version
296  END IF
297 
298  l10=zu/zetu
299  IF(zetu > 50)THEN
300  nits=1
301  END IF
302  CALL psiu_26(zu/l10,psiu_26s)
303  usr=ut*von/(log(zu/zo10)-psiu_26s)
304  CALL psit_26(zt/l10,psit_26s)
305  tsr=-(dt-dter*jcool)*von*fdg/(log(zt/zot10)-psit_26s)
306  CALL psit_26(zq/l10,psit_26s)
307 
308  qsr=-(dq-wetc*dter*jcool)*von*fdg/(log(zq/zot10)-psit_26s)
309  tkt=.001_sp
310  charn=0.011_sp
311 
312  IF(ut > 10)THEN
313  charn=0.011_sp+(ut-10)/(18-10)*(0.018_sp-0.011_sp)
314  END IF
315 
316  IF(ut > 18)THEN
317  charn=0.018_sp
318  END IF
319 
320 !*************** bulk loop ************
321  do i=1,nits !MDR uncomment to do iterations, was hardwired to 1 iteration
322 !MDR DO I=1,1
323  zet=von*grav*zu/ta*(tsr*(1._sp+0.61_sp*q)+0.61_sp*ta*qsr)/ &
324  (usr*usr)/(1._sp+0.61_sp*q)
325  zo=charn*usr*usr/grav+0.11_sp*visa/usr
326  rr=zo*usr/visa
327  l=zu/zet
328  zoq=amin1(1.15e-4_sp,5.5e-5_sp/(rr**0.6_sp))
329  zot=zoq
330  CALL psiu_26(zu/l,psiu_26s)
331  usr=ut*von/(log(zu/zo)-psiu_26s)
332  CALL psit_26(zt/l,psit_26s)
333  tsr=-(dt-dter*jcool)*von*fdg/(log(zt/zot)-psit_26s)
334  CALL psit_26(zq/l,psit_26s)
335  qsr=-(dq-wetc*dter*jcool)*von*fdg/(log(zq/zoq)-psit_26s)
336  bf=-grav/ta*usr*(tsr+.61_sp*ta*qsr)
337 ! print *,usr,tsr,qsr,Bf
338  IF(bf > 0)THEN
339  ug=beta*((bf*zi)**0.333_sp)
340  ELSE
341  ug=0.2_sp
342  END IF
343 
344  ut=sqrt(du*du+ug*ug)
345  rnl=0.97_sp*(5.67_sp*0.00000001_sp*((ts-dter*jcool+tdk)**4)-rl)
346  hsb=-rhoa*cpa*usr*tsr
347  hlb=-rhoa*le*usr*qsr
348  qout=rnl+hsb+hlb
349 
350 ! DELS=RNS*(.065_SP+11*TKT-6.6_SP*0.00001_SP/(TKT*(1-EXP(-TKT/8.0_SP*0.0001_SP))))
351 !Mark Rowe 4-24-2013 the line above has misplaced parentheses in two places, and was giving
352 !DELS = -Inf By comparison to COARE3.0 Matlab version, below is correct
353  dels=rns*(.065_sp+11.0_sp*tkt-6.6_sp*0.00001_sp/tkt*(1.0_sp-exp(-tkt/(8.0_sp*0.0001_sp))))
354  qcol=qout-dels
355  alq=al*qcol+be*hlb*cpw/le
356 
357  IF(alq > 0)THEN
358  xlamx=6._sp/(1._sp+(bigc*alq/usr**4)**0.75_sp)**0.333_sp
359  tkt=xlamx*visw/(sqrt(rhoa/rhow)*usr)
360  ELSE
361  xlamx=6.0_sp
362  tkt=amin1(.01_sp,xlamx*visw/(sqrt(rhoa/rhow)*usr))
363  END IF
364 ! cool skin
365 
366  dter=qcol*tkt/tcw
367  dqer=wetc*dter
368 
369  END DO
370 ! of end bulk iter loop
371 
372 !****** compute fluxes ******************************************
373 ! wind stress
374  tau=rhoa*usr*usr*du/ut
375 ! sensible heat flux
376  hsb=rhoa*cpa*usr*tsr
377 ! latent heat flux
378  hlb=rhoa*le*usr*qsr
379 
380 
381 !****** compute transfer coeffs relative to ut @ meas. ht ********
382  cd=tau/rhoa/ut/amax1(.1_sp,du)
383  ch=-usr*tsr/ut/(dt-dter*jcool)
384  ce=-usr*qsr/(dq-dqer*jcool)/ut
385 
386 !****** compute 10-m neutral coeff relative to ut ****************
387  cdn_10=von*von/log(10._sp/zo)/log(10._sp/zo)
388  chn_10=von*von*fdg/log(10._sp/zo)/log(10._sp/zot)
389  cen_10=von*von*fdg/log(10._sp/zo)/log(10._sp/zoq)
390 
391 !******** rain heat flux (save to use if desired) *************
392 ! dwat=2.11e-5*((t+tdk)/tdk)^1.94; %! water vapour diffusivity
393 ! dtmp=(1.+3.309e-3*t-1.44e-6*t*t)*0.02411/(rhoa*cpa); %!heat diffusivity
394 ! alfac= 1/(1+(wetc*Le*dwat)/(cpa*dtmp)); %! wet bulb factor
395 ! RF= rain*alfac*cpw*((ts-t-dter*jcool)+(Qs-Q-dqer*jcool)*Le/cpa)/3600;
396 !**************************************************************
397 !----------------------------------------------------------
398  RETURN
399  END SUBROUTINE coare26z

403  SUBROUTINE psit_26(ZET,PSI)
404 ! computes temperature structure function
405  IMPLICIT NONE
406  REAL(SP) :: zet,psi,x,psik,psic,f,c
407 
408  IF(zet < 0.0_sp)THEN
409  x=(1._sp-15._sp*zet)**.5_sp
410  psik=2._sp*log((1._sp+x)/2._sp)
411  x=(1._sp-34.15_sp*zet)**.3333_sp
412  psic=1.5_sp*log((1._sp+x+x*x)/3._sp)-sqrt(3._sp) &
413  *atan((1._sp+2._sp*x)/sqrt(3._sp))+4._sp*atan(1._sp)/sqrt(3._sp)
414  f=zet*zet/(1._sp+zet*zet)
415  psi=(1._sp-f)*psik+f*psic
416  ELSE
417  c=amin1(50._sp,.35_sp*zet)
418  psi=-((1._sp+2._sp/3._sp*zet)**1.5_sp+.6667_sp &
419  *(zet-14.28_sp)/exp(c)+8.525_sp)
420  END IF
421 
422  RETURN
423  END SUBROUTINE psit_26

427  SUBROUTINE psiu_26(ZET,PSI)
428 ! computes velocity structure function
429  IMPLICIT NONE
430  REAL(SP) :: zet,psi,x,psik,psic,f,c
431 
432  IF(zet < 0.0_sp)THEN
433  x=(1._sp-15._sp*zet)**.25_sp
434  psik=2._sp*log((1._sp+x)/2._sp)+log((1._sp+x*x)/2._sp) &
435  -2._sp*atan(x)+2._sp*atan(1._sp)
436  x=(1._sp-10.15_sp*zet)**.3333_sp
437  psic=1.5_sp*log((1._sp+x+x*x)/3._sp)-sqrt(3._sp) &
438  *atan((1._sp+2._sp*x)/sqrt(3._sp))+4._sp*atan(1._sp)/sqrt(3._sp)
439  f=zet*zet/(1._sp+zet*zet)
440  psi=(1._sp-f)*psik+f*psic
441  ELSE
442  c=amin1(50._sp,.35_sp*zet)
443  psi=-((1._sp+1.0_sp*zet)**1.0_sp+.667_sp*(zet-14.28_sp)/exp(c)+8.525_sp)
444  END IF
445 
446  RETURN
447  END SUBROUTINE psiu_26
