420  SUBROUTINE read_ssh
421 # if defined (SEDIMENT)
422  USE mod_sed, only : morpho_model,sed_hot_start
423 # endif
424  IMPLICIT NONE
425  TYPE(ncvar), POINTER :: VAR
426  TYPE(ncdim), POINTER :: DIM
427  LOGICAL :: FOUND
428  INTEGER :: STKCNT
429 
430  IF(dbg_set(dbg_sbr)) write(ipt,*) "Start: READ_SSH"
431 
432  stkcnt = nc_start%FTIME%PREV_STKCNT
433 
434  ! LOAD EL
435  var => find_var(nc_start,'zeta',found)
436  IF(.not. found) CALL fatal_error("COULD NOT FIND VARIABLE 'zeta'&
437  & IN THE HOTSTART FILE OBJECT")
438  CALL nc_connect_avar(var, el)
439  CALL nc_read_var(var,stkcnt)
440 
441  ! LOAD ET
442  var => find_var(nc_start,'et',found)
443  IF(.not. found) CALL fatal_error("COULD NOT FIND VARIABLE 'et'&
444  & IN THE HOTSTART FILE OBJECT")
445  CALL nc_connect_avar(var, et)
446  CALL nc_read_var(var,stkcnt)
447 
448  !----------------------------------------------------------------
449  ! Read the most recent bathymetry if Morphodynamics is Active
450  !----------------------------------------------------------------
451 # if defined(SEDIMENT)
452  IF(morpho_model .and. sed_hot_start)THEN
453  var => find_var(nc_start,'h',found)
454  IF(.not. found) CALL fatal_error("COULD NOT FIND VARIABLE 'h'&
455  & IN THE HOTSTART FILE OBJECT")
456  CALL nc_connect_avar(var, h)
457  CALL nc_read_var(var,stkcnt)
458  ENDIF
459 # endif
460 
461 
462  !----------------------------------------------------------------
463  ! Given SSH and Bathy, Update the Bathymetry
464  !----------------------------------------------------------------
465  d = h + el
466  dt = h + et
467 
468 
469  CALL n2e2d(h,h1)
470  CALL n2e2d(el,el1)
471  CALL n2e2d(d,d1)
472  CALL n2e2d(dt,dt1)
473 
474  IF(dbg_set(dbg_sbr)) write(ipt,*) "Start: READ_SSH"
475 
476  END SUBROUTINE read_ssh
