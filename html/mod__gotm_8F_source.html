87  SUBROUTINE advance_gotm
88 !==============================================================================|
89 ! A WRAPPER ROUTINE TO CALL THE GOTM LIBRARIES |
90 !==============================================================================|
91 
92 !--Variables from FVCOM Modules------------------------------
93  use lims, only : m,kbm1,kb
94  use all_vars, only : rho1,dz,dzz,d,t1,u,v,km,kh,tke,teps,wusurf,wvsurf,l, &
95  z,zz,nbve,ntve,grav_n,cc_z0b
96  use control, only: dti,umol,iint,iend
97 
98 !--Variables from GOTM Modules-------------------------------
99  use turbulence, only: do_turbulence,cde
100  use turbulence, only: tke1d => tke, eps1d => eps, l1d => l
101  use turbulence, only: num,nuh
102 
103 !--Local Variables (Temporary)-------------------------------
104  IMPLICIT NONE
105  INTEGER :: i,j,k
106  REAL(SP) :: rr,ztemp
107 
108 !--Variables for Interfacing with GOTM-----------------------
109  DOUBLE PRECISION :: depth,time_step
110  DOUBLE PRECISION :: u_taus,u_taub,z0s_gotm,z0b_gotm,umold
111  DOUBLE PRECISION :: h(0:kbm1)
112  REAL(SP) :: nn(m,1:kb),ss(m,1:kb)
113  DOUBLE PRECISION :: nn1d(0:kbm1),ss1d(0:kbm1)
114 !--Parameters------------------------------------------------
115  REAL(SP), PARAMETER :: kappa = .4 !!Von Karman's Constant
116  REAL(SP), PARAMETER :: charnok_val = 1400. !!Charnok Constant
117  REAL(SP), PARAMETER :: z0s_min = .00 !!Minimum Surface Roughness
118  INTEGER , PARAMETER :: maxitz0b = 10
119 
120 !------------------------------------------------------------
121  REAL(SP) :: uu_node_k,vv_node_k,uu_node_km1,vv_node_km1
122  REAL(SP) :: uu_node_kbm1,vv_node_kbm1
123  REAL(SP) :: wusurf_node,wvsurf_node
124 
125 !==============================================================================|
126 
127 
128 !------------------Calculate Buoyancy Frequency Squared (NN)-------------------!
129 
130  DO k=2,kbm1
131  DO i=1,m
132  nn(i,k) = -grav_n(i) * (rho1(i,k-1)-rho1(i,k))/(dzz(i,k-1)*d(i))
133  END DO
134  END DO
135  !!Set BC's as GOTM Does
136  nn(:,1) = nn(:,2)
137  nn(:,kb) = nn(:,kbm1)
138 
139 !------------------Calculate Shear Frequency Squared (SS)----------------------!
140 
141  DO k=2,kbm1
142  DO i=1,m
143  uu_node_k = 0.0_sp
144  vv_node_k = 0.0_sp
145  uu_node_km1 = 0.0_sp
146  vv_node_km1 = 0.0_sp
147  DO j=1,ntve(i)
148  uu_node_k = uu_node_k + u(nbve(i,j),k)
149  vv_node_k = vv_node_k + v(nbve(i,j),k)
150  uu_node_km1 = uu_node_km1 + u(nbve(i,j),k-1)
151  vv_node_km1 = vv_node_km1 + v(nbve(i,j),k-1)
152  END DO
153  uu_node_k = uu_node_k/float(ntve(i))
154  vv_node_k = vv_node_k/float(ntve(i))
155  uu_node_km1 = uu_node_km1/float(ntve(i))
156  vv_node_km1 = vv_node_km1/float(ntve(i))
157 
158 ! SS(I,K) = ((U(I,K)-U(I,K-1))**2+(V(I,K)-V(I,K-1))**2)/(DZZ(I,K-1)*D(I))**2
159  ss(i,k) = ((uu_node_k-uu_node_km1)**2+(vv_node_k-vv_node_km1)**2)/(dzz(i,k-1)*d(i))**2
160  END DO
161  END DO
162 
163  !!Set BC's as GOTM Does
164  ss(:,1) = ss(:,2)
165  ss(:,kb) = ss(:,kbm1)
166 
167 !------------------Main Loop Over Elements-------------------------------------!
168 
169  DO i=1,m
170  !Surface Friction Velocity [m/s]
171  wusurf_node = 0.0_sp
172  wvsurf_node = 0.0_sp
173  DO j=1,ntve(i)
174  wusurf_node = wusurf_node + wusurf(nbve(i,j))
175  wvsurf_node = wvsurf_node + wvsurf(nbve(i,j))
176  END DO
177  wusurf_node = wusurf_node/float(ntve(i))
178  wvsurf_node = wvsurf_node/float(ntve(i))
179 
180 ! u_taus=(wusurf(i)**2+wvsurf(i)**2)**(1./4.)
181  u_taus=(wusurf_node**2+wvsurf_node**2)**(1./4.)
182 
183  !Set Surface Roughness Height Using Charnok Formula[m]
184 ! z0s_gotm = charnok_val*u_taus**2/grav
185 ! if(z0s_gotm < z0s_min) z0s_gotm = z0s_min
186  z0s_gotm = z0s_min
187 
188  !Set Bottom Roughness Height [m] and Friction Velocity [m/s] using GOTM Method
189  ztemp=(zz(i,kbm1)-z(i,kb))*d(i)
190  umold = umol
191 
192  uu_node_kbm1 = 0.0_sp
193  vv_node_kbm1 = 0.0_sp
194  DO j=1,ntve(i)
195  uu_node_kbm1 = uu_node_kbm1 + u(nbve(i,j),kbm1)
196  vv_node_kbm1 = vv_node_kbm1 + v(nbve(i,j),kbm1)
197  END DO
198  uu_node_kbm1 = uu_node_kbm1/float(ntve(i))
199  vv_node_kbm1 = vv_node_kbm1/float(ntve(i))
200 
201  DO j=1,maxitz0b
202  z0b_gotm = 0.1*umol/max(umold,u_taub)+0.03*(cc_z0b(i)/.03)
203  rr = kappa/(log((z0b_gotm+ztemp)/z0b_gotm))
204 ! u_taub = rr*sqrt( u(i,kbm1)*u(i,kbm1) + v(i,kbm1)*v(i,kbm1) )
205  u_taub = rr*sqrt(uu_node_kbm1*uu_node_kbm1+vv_node_kbm1*vv_node_kbm1)
206  END DO
207 
208  !Set up Depth [m]
209  depth = d(i)
210 
211  !Set up Time Step [s]
212  time_step = dti
213 
214  !Set up Layer Thicknesses [m]
215  h(1:kbm1) = dz(i,kbm1:1:-1)*depth
216 
217  !Set Up 1-D Arrays
218  num(0:kbm1) = km(i,kb:1:-1) !Vertical Kinematic Viscosity [m^2/s]
219  nuh(0:kbm1) = kh(i,kb:1:-1) !Vertical Kinematic Viscosity [m^2/s]
220  ss1d(0:kbm1) = ss(i,kb:1:-1) !Shear Frequency Squared [1/s^2]
221  nn1d(0:kbm1) = nn(i,kb:1:-1) !Buoyancy Frequency Squared [1/s^2]
222  tke1d(0:kbm1) = tke(i,kb:1:-1) !Turbulent Kinetic Energy [m^2/s^2]
223  eps1d(0:kbm1) = teps(i,kb:1:-1) !Turbulence Dissipation Rate [m^2/s^3]
224  l1d(0:kbm1) = l(i,kb:1:-1)
225 
226  !Update Turbulence Model
227  call do_turbulence(kbm1,time_step,depth,u_taus,u_taub,z0s_gotm,z0b_gotm,h,nn1d,ss1d)
228 
229  !Update 3D Fields of TKE,EPS,KM,KH
230  tke(i,1:kb) = tke1d(kbm1:0:-1)
231  teps(i,1:kb) = eps1d(kbm1:0:-1)
232  teps(i,1) = 0.0 !required, as gotm can produce "Inf" which will crash NetCDF on output
233  km(i,1:kb) = num(kbm1:0:-1)
234  kh(i,1:kb) = nuh(kbm1:0:-1)
235  l(i,1:kb) = l1d(kbm1:0:-1)
236  END DO
237 
238 
239  RETURN
240  END SUBROUTINE advance_gotm
241 !==============================================================================|
242 # endif
243 END MODULE mod_gotm
