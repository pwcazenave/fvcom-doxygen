29  SUBROUTINE adjust2d3d(ADJUST_TYPE)
30 # if !defined (SEMI_IMPLICIT)
31 !==============================================================================|
32 ! ADJUST 3D VELOCITY USING DEFECT BETWEEN UPDATED AND CURRENT VERTICALLY !
33 ! AVERAGED VELOCITIES !
34 ! !
35 ! FORMULA IS: !
36 ! !
37 ! U_adjusted = U_orig + eps*(U_avg_new - U_avg_current) !
38 ! eps = 0 : no adjustment !
39 ! eps = 1 : full adjustment !
40 !==============================================================================|
41 # if defined (WET_DRY)
42  USE mod_wd
43 # endif
44  USE all_vars
45  USE mod_utils
46  IMPLICIT NONE
47  INTEGER, INTENT(IN) :: ADJUST_TYPE
48  INTEGER :: I,K
49  REAL(SP), PARAMETER :: EPS = 1.0_sp
50  REAL(SP) :: UAC,VAC,UTMP,VTMP
51 !==============================================================================!
52 
53  if(dbg_set(dbg_sbr)) write(ipt,*) "Start: adjust2d3d"
54 
55  SELECT CASE(adjust_type)
56 
57  CASE(1)
58  DO i=1,nt
59  uac = sum(u(i,1:kbm1)*dz1(i,1:kbm1))
60  vac = sum(v(i,1:kbm1)*dz1(i,1:kbm1))
61  u(i,1:kbm1) = u(i,1:kbm1) + eps*(ua(i) - uac)
62  v(i,1:kbm1) = v(i,1:kbm1) + eps*(va(i) - vac)
63  END DO
64 
65  CASE(2)
66  uard = uard/float(isplit)
67  vard = vard/float(isplit)
68 !!# if defined (WET_DRY)
69 !! UARDS = UARDS/FLOAT(ISPLIT)
70 !! VARDS = VARDS/FLOAT(ISPLIT)
71 !!# endif
72 
73 
74  DO i=1,nt
75 # if defined (WET_DRY)
76  IF(iswetct(i)*iswetc(i) == 1)THEN
77 # endif
78  utmp = 0.0_sp ; vtmp = 0.0_sp
79  DO k=1,kbm1
80  utmp = utmp + u(i,k)*dz1(i,k)
81  vtmp = vtmp + v(i,k)*dz1(i,k)
82  END DO
83  utmp = utmp*dt1(i)
84  vtmp = vtmp*dt1(i)
85  DO k=1,kbm1
86  u(i,k) = u(i,k) - (utmp-uard(i))/dt1(i)
87  v(i,k) = v(i,k) - (vtmp-vard(i))/dt1(i)
88  END DO
89 # if defined (WET_DRY)
90  END IF
91 # endif
92  END DO
93 
94 !!# if defined (WET_DRY)
95 !! DO I=1,NT
96 !! UTMP = 0.0_SP ; VTMP = 0.0_SP
97 !! DO K=1,KBM1
98 !! UTMP = UTMP + U(I,K)*DZ1(I,K)
99 !! VTMP = VTMP + V(I,K)*DZ1(I,K)
100 !! END DO
101 !! UTMP = UTMP*DT1(I)
102 !! VTMP = VTMP*DT1(I)
103 !! DO K=1,KBM1
104 !! US(I,K) = U(I,K) - (UTMP-UARDS(I))/DT1(I)
105 !! VS(I,K) = V(I,K) - (VTMP-VARDS(I))/DT1(I)
106 !! END DO
107 !! END DO
108 !!# endif
109 
110  END SELECT
111 
112  if(dbg_set(dbg_sbr)) write(ipt,*) "End: adjust2d3d"
113 
114 # endif
115  END SUBROUTINE adjust2d3d
