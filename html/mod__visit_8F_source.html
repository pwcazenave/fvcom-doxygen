206  SUBROUTINE update_mesh(vmp,error)
207  USE all_vars
208  implicit none
209  include "visitfortransiminterface.inc"
210  integer, intent(out) :: error
211  integer :: ind, vncp, lind, uind
212  TYPE(visitmeshtype), POINTER :: vmp
213  REAL*4, pointer, Dimension(:) :: px,py,pz
214 
215  error =-1 ! BAD RESULT!
216 
217  if (.NOT. ASSOCIATED(vmp)) then
218  if(msr) WRITE(ipt,*) "Mesh data pointer passed to UPDATE MESH was not associa&
219  &ted"
220  return
221  end if
222 
223  if(dbg_set(dbg_scl)) then
224  WRITE(ipt,*)"Starting update_mesh"
225  CALL print_mesh(vmp)
226  end if
227 
228 
229  IF (.NOT.vmp%MESHALLOCATED) then
230 
231  SELECT CASE(vmp%MESH)
232 
233  CASE ( visit_twodmesh )
234  vncp=4 ! The number of integers to define one cell
235  vmp%NDIMS=2 ! Spatial dimension of the mesh
236  vmp%LYRS=1 ! The number of layers
237  vmp%LVLS=1 ! The number of levels
238  vmp%Nodes=vmp%LVLS * mt ! Total number of nodes
239  vmp%Zones=vmp%LYRS * nt ! Total number of zones or cells
240  vmp%LCONN=vncp*vmp%Zones ! The number of integer in the
241  ! connectivity array passed to visit
242  vmp%DATAOWNER = visit_owner_sim
243  ! SINCE THE TWOD MESH IS STATIC SET THE TIME TO ZERO AND LEAVE IT
244  vmp%Updated_Time=zerotime
245  case ( visit_bathymesh )
246  vncp=4 ! The number of integers to define one cell
247  vmp%NDIMS=3
248  vmp%LYRS=1
249  vmp%LVLS=1
250  vmp%Nodes= vmp%LVLS * mt
251  vmp%Zones= vmp%LYRS * nt
252  vmp%LCONN=vncp*vmp%Zones
253  vmp%DATAOWNER = visit_owner_sim
254  ! SINCE THE BATHYMETRY MESH IS STATIC SET THE TIME TO ZERO AND LEAVE IT
255  vmp%Updated_Time=zerotime
256  CASE ( visit_sshmesh )
257  vncp=4 ! The number of integers to define one cell
258  vmp%NDIMS=3
259  vmp%LYRS=1
260  vmp%LVLS=1
261  vmp%Nodes= vmp%LVLS * mt
262  vmp%Zones= vmp%LYRS * nt
263  vmp%LCONN=vncp*vmp%Zones
264  vmp%Nodes=mt
265  vmp%Zones=nt
266  vmp%LCONN=vncp*nt
267  vmp%DATAOWNER = visit_owner_sim
268  ! SINCE THE SSH MESH IS NOT STATIC SET THE TIME TO -1
269  vmp%Updated_Time=days2time(-1000000)
270  CASE ( visit_layermesh )
271 
272  vncp=7 ! The number of integers to define one cell
273  vmp%NDIMS=3
274  vmp%LYRS=kbm2
275  vmp%LVLS=kbm1
276  vmp%Nodes= vmp%LVLS * mt
277  vmp%Zones= vmp%LYRS * nt
278  vmp%LCONN=vncp*vmp%Zones
279  vmp%Nodes= kbm1 * mt
280  vmp%Zones= kbm2 * nt
281  vmp%LCONN=vncp*nt*kbm2
282  vmp%DATAOWNER = visit_owner_sim
283  ! SINCE THE LAYER MESH IS NOT STATIC SET THE TIME TO -1
284  vmp%Updated_Time=days2time(-1000000)
285  CASE ( visit_levelmesh )
286  vncp=7 ! The number of integers to define one cell
287  vmp%NDIMS=3
288  vmp%LYRS=kbm1
289  vmp%LVLS=kb
290  vmp%Nodes= vmp%LVLS * mt
291  vmp%Zones= vmp%LYRS * nt
292  vmp%LCONN=vncp*vmp%Zones
293  vmp%DATAOWNER = visit_owner_sim
294  ! SINCE THE LEVEL MESH IS NOT STATIC SET THE TIME TO -1
295  vmp%Updated_Time=days2time(-1000000)
296 
297  CASE DEFAULT
298  CALL fatal_error( "VISIT UPDATE MESH",&
299  & "CASE(DEFAULT): bad mesh!")
300 
301  END SELECT
302 
303  vmp%MESHALLOCATED=.true.
304 
305 
306  ! NV must be allocated by the total number of zones(NT * layers)
307  ! by VnCP
308 
309  ! All other variables will be zones by layer or nodes by layers
310  ALLOCATE(vmp%NV(vncp,vmp%Zones),stat=error)
311  if ( error .ne. 0 ) then
312  if(msr) WRITE(ipt,*)"Allocation error in UPDATE_VMP: CAN NOT ALLOCA&
313  &TE NV, error= ",error,"; MYID= ",myid
314  return
315  end if
316 
317  if (vncp == 4) then
318  vmp%NV(1,:)=visit_cell_tri
319  vmp%NV(2:4,:)=transpose(nv(1:nt,1:3)) -1
320 
321  else if (vncp == 7) then
322  vmp%NV(1,:)=visit_cell_wedge
323  DO ind =0,(vmp%LYRS-1)
324 
325  lind=ind*nt+1
326  uind=(ind+1)*nt
327  ! 1 NT
328  ! NT+1 2*NT
329  !
330  ! (LYRS-1)*NT+1 LYRS*NT
331  vmp%NV(2:4,lind:uind)=ind*mt + transpose(nv(1:nt,1:3)) -1
332 
333  vmp%NV(5:7,lind:uind)=(ind+1)*mt + transpose(nv(1:nt,1:3)) -1
334  END DO
335 
336  else
337  call fatal_error("VISIT UPDATE MESH",&
338  &"Unknown Cell type: VnCP")
339  end if
340 
341  ALLOCATE(vmp%GHOSTZONES(nt,vmp%LYRS),stat=error)
342  if ( error .ne. 0 ) then
343  if(msr)WRITE(ipt,*) "Allocation error in UPDATE_VMP: CAN NOT ALLOCA&
344  &TE NV, error= ",error,"; MYID= ",myid
345  return
346  end if
347 
348  DO ind = 1,vmp%LYRS
349  vmp%GHOSTZONES(1:n,ind)=0
350  if(nt .GT. n) vmp%GHOSTZONES((n+1):nt,ind)=1
351  END DO
352 
353 
354  ALLOCATE(vmp%VX(mt,vmp%LVLS),stat=error)
355  if (error .ne. 0 ) then
356  if(msr)WRITE(ipt,*) "Allocation error in UPDATE_VMP: CAN NOT ALLOCA&
357  &TE VX, MYID=",myid
358  return
359  end if
360 
361 
362 
363  ALLOCATE(vmp%VY(mt,vmp%LVLS),stat=error)
364  if (error .ne. 0 ) then
365  if(msr)WRITE(ipt,*) "Allocation error in UPDATE_VMP: CAN NOT ALLOCA&
366  &TE VY, MYID=",myid
367  return
368  end if
369 
370 
371 
372  ALLOCATE(vmp%VZ(mt,vmp%LVLS),stat=error)
373  if (error .ne. 0 ) then
374  if(msr)WRITE(ipt,*) "Allocation error in UPDATE_VMP: CAN NOT ALLOCA&
375  &TE VZ, MYID=",myid
376  return
377  end if
378 
379  ! Add static data
380  DO ind=1,vmp%LVLS
381  vmp%VX(1:mt,ind)=vx(1:mt)
382  vmp%VY(1:mt,ind)=vy(1:mt)
383  END DO
384 
385  if (vmp%MESH == visit_twodmesh) vmp%VZ(1:mt,1)=0
386 
387  if (vmp%MESH == visit_bathymesh) vmp%VZ(1:mt,1)=-h(1:mt)
388 
389 # if defined (SPHERICAL)
390  if (vmp%MESH == visit_twodmesh .OR. &
391  &vmp%MESH == visit_bathymesh) then
392 
393  px=>vmp%VX(:,1)
394  py=>vmp%VY(:,1)
395  pz=>vmp%VZ(:,1)
396  Call sphere2cart(px,py,pz,mt)
397 
398 
399  end if
400 # endif
401 
402  END IF !
403 
404 
405  SELECT CASE(vmp%MESH)
406 
407  CASE ( visit_twodmesh )
408  ! Do NOTHING
409  CASE ( visit_bathymesh )
410  ! Do NOTHING
411  CASE ( visit_sshmesh )
412 
413  if (vmp%Updated_Time .NE. visit_time_ext) then
414 
415 # if defined (SPHERICAL)
416  vmp%VX(1:mt,1)=vx(1:mt)
417  vmp%VY(1:mt,1)=vy(1:mt)
418  vmp%VZ(1:mt,1)=el(1:mt)
419 
420  px=>vmp%VX(:,1)
421  py=>vmp%VY(:,1)
422  pz=>vmp%VZ(:,1)
423  Call sphere2cart(px,py,pz,mt)
424 # else
425  vmp%VZ(1:mt,1)=el(1:mt)
426  vmp%Updated_Time=visit_time_ext
427 # endif
428 ! if (MSR) WRITE(IPT,*) "TIME CHANGED: Updated SSH"
429  end if
430 
431  CASE ( visit_layermesh )
432 
433  if (vmp%Updated_Time .NE. visit_time_int) then
434 # if defined (SPHERICAL)
435  Do ind = 1,vmp%LVLS
436  vmp%VX(1:mt,ind)=vx(1:mt)
437  vmp%VY(1:mt,ind)=vy(1:mt)
438  vmp%VZ(1:mt,ind)=el(1:mt)+ zz(1:mt,ind)*d(1:mt)
439 
440  px=>vmp%VX(:,ind)
441  py=>vmp%VY(:,ind)
442  pz=>vmp%VZ(:,ind)
443  Call sphere2cart(px,py,pz,mt)
444  End Do
445 # else
446  Do ind = 1,vmp%LVLS
447  vmp%VZ(1:mt,ind)=el(1:mt)+ zz(1:mt,ind)*d(1:mt)
448  End Do
449 # endif
450 ! if (MSR) WRITE(IPT,*) "TIME CHANGED: Updated SigmaLayers"
451  vmp%Updated_Time=visit_time_int
452  end if
453 
454  CASE ( visit_levelmesh )
455 
456  if (vmp%Updated_Time .NE. visit_time_int) then
457 # if defined (SPHERICAL)
458  Do ind = 1,vmp%LVLS
459  vmp%VX(1:mt,ind)=vx(1:mt)
460  vmp%VY(1:mt,ind)=vy(1:mt)
461  vmp%VZ(1:mt,ind)=el(1:mt)+ z(1:mt,ind)*d(1:mt)
462 
463  px=>vmp%VX(:,ind)
464  py=>vmp%VY(:,ind)
465  pz=>vmp%VZ(:,ind)
466  Call sphere2cart(px,py,pz,mt)
467  End Do
468 
469 # else
470  Do ind = 1,vmp%LVLS
471  vmp%VZ(1:mt,ind)=el(1:mt)+ z(1:mt,ind)*d(1:mt)
472  End Do
473 # endif
474 ! if (MSR) WRITE(IPT,*) "TIME CHANGED: Updated SigmaLevels"
475  vmp%Updated_Time=visit_time_int
476  end if
477 
478  CASE DEFAULT
479  CALL fatal_error("CASE(DEFAULT) in UPDATE_MESH",&
480  &"ERROR while setting values: bad mesh!")
481 
482  END SELECT
483 
484 
485 
486 ! if (MSR) WRITE(IPT,*) "FINISHED MESH UPDATE"
487 ! if(MSR) CALL PRINT_MESH(VMP)
488 
489 
490  error = 0
491  END SUBROUTINE update_mesh

495  SUBROUTINE sphere2cart(PX,PY,PZ,n)
496  IMPLICIT NONE
497  integer, intent(in) :: n
498  REAL*4, pointer, Dimension(:), intent(inout) :: px,py,pz
499 
500  real*4, parameter :: rad=100000.0;
501  real*4, parameter :: d2r=3.14159/180.0;
502 
503  real*4, dimension(n) :: tx, ty, tz
504 
505  tx = (rad+pz)*cos(px*d2r)*cos(py*d2r);
506  ty = (rad+pz)*sin(px*d2r)*cos(py*d2r);
507  tz = (rad+pz)*sin(py*d2r);
508 
509  px=tx
510  py=ty
511  pz=tz
512 
513  END SUBROUTINE sphere2cart

630  SUBROUTINE spherevel2cart(PU,PV,PW)
631  USE all_vars, only : xc,yc, nt
632  IMPLICIT NONE
633  REAL*4, pointer, Dimension(:), intent(inout) :: pu,pv,pw
634  real*4, parameter :: d2r=3.14159/180.0;
635  real*4, dimension(NT) :: tu, tv, tw
636 
637 
638  tu=-sin(xc(1:nt)*d2r)*pu &
639  & -sin(yc(1:nt)*d2r)*cos(xc(1:nt)*d2r)*pv &
640  & + cos(xc(1:nt)*d2r)*cos(yc(1:nt)*d2r)*pw
641 
642  tv=cos(xc(1:nt)*d2r)*pu &
643  & -sin(yc(1:nt)*d2r)*sin(xc(1:nt)*d2r)*pv &
644  & + sin(xc(1:nt)*d2r)*cos(yc(1:nt)*d2r)*pw
645 
646  tw= 0.0 &
647  & + cos(yc(1:nt)*d2r)*pv &
648  & + sin(yc(1:nt)*d2r)*pw
