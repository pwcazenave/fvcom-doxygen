109 !

111 !--Corrects for Wind-Driven Barotropic Response (See Schwing 1989)

141  ELSE

143  END IF

153  END DO

161 
162  rhointcur= 0.0_sp
163  rhointnxt = 0.0_sp
164 
165  DO i=nobclsf,1,-1 ! COUNT BACKWARDS
166  ndx = nbclsf(i)
167  cdx = ibclsf(i)
168 
169  !INTEGRATE RHO IN DEPTH, CONVERT TO MKS UNITS DAMIT
170  rhointcur=rhointnxt
171  rhointnxt=0.0_sp
172  DO k=1,kbm1
173  rhointnxt=rhointnxt+(1.0_sp+rho1(cdx,k))*1.0e3_sp*dz(cdx,k)
174  END DO
175 
176 !ESTIMATE DENSITY GRADIENT, AND MODIFY BOUNDARY ELEVATION
177 !NOTE THE FACTOR OF 1000 AND 2 TO COMPENSATE FOR THE
178 !FACT THAT THE MODEL STORES RHO1 AS SIGMA, AND IN CGS.
179 
180 ! ADJUST THE SEA SURFACE HEIGHT FOR A MEAN FLOW DUE TO DENSITY GRADIENT
181 ! CALCULATE THE THERMAL WIND AND ADJUST THE SEA SURFACE HEIGHT SO THAT THE BOTTOM
182 ! BOUNDARY IS A LEVEL OF NO MOTION:
183 !
184 ! ETATAN=[ -1/(Rho_bar)*(del(h*rho)/del(s) - Rho_bot*(del(h)/del(s) ] *del(s)
185 !
186  IF (i /= nobclsf) THEN
187  eta_lcl(i)=-(1.0_sp/(0.5_sp*(rhointnxt+rhointcur))) &
188  *((h(cdx)*rhointnxt-h(ndx)*rhointcur) &
189  -0.5_sp*1.0e3_sp*(2.0_sp+rho1(cdx,kbm1)+rho1(ndx,kbm1)) &
190  *(h(cdx)-h(ndx)))
191  END IF
192 
193  END DO
