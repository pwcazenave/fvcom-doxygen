30 !==============================================================================!
31 ! PROBES MOD: CONTROLS TIME SERIES OUTPUT OF QUANTITIES AT SELECT LOCATIONS !
32 ! !
33 ! CONTROL: IF PROBE_ON = T IN RUNTIME PARAMETER CONTROL FILE TIME SERIES DUMP!
34 ! IS ACTIVATED !
35 ! !
36 ! SETUP: TIME SERIES NML OBJECTS ARE PUT IN THE RUN FILE FOR EACH PROBE !
37 ! !
38 ! !
39 ! SETUP FILE NAMING CONVENTION: !
40 ! ***_timeseriesXX.dat !
41 ! *** = case id tag (ex: gom/sat/mhb, etc) !
42 ! XX = digit from 01 to 99, not sequential !
43 ! !
44 ! SETUP VARIABLE DESCRIPTION: !
45 ! D_LOC = element/node location of time-series data !
46 ! D_TYP = type of location ("element" or "node") !
47 ! D_TIT = title of location, used for time series output filename !
48 ! D_DES = description of data (placed in time series output header) !
49 ! O_INT = printing interval for data !
50 ! K_ONE = initial sigma level
51 ! K_TWO = final sigma level
52 ! VAR = variable to output in time series !
53 ! VNAME = variable description (placed in time series output header) !
54 ! !
55 ! CURRENT VARS SET UP: !
56 ! u,v,w,ww,t1,s1,ua,va,rho1 !
57 ! !
58 ! !
59 ! NOTE 1: !
60 ! IF FILE ALREADY EXIST, A NEW FILE WILL BE CREATED NAMED FILENAME-01 !
61 ! IF FILENAME-01 EXISTS, A NEW FILE WILL BE CREATED NAMED FILENAME-02 !
62 ! ETC !
63 !==============================================================================!

488  SUBROUTINE set_probes(P_ON,NP,FNM)
489  !------------------------------------------------------------------------------|
490  ! READ IN TIME SERIES OBJECTS FROM INPUT |
491  !------------------------------------------------------------------------------|
492  USE mod_par
493  USE lims
494  USE mod_set_time
495 
496  IMPLICIT NONE
497  LOGICAL, INTENT(IN) :: P_ON
498  INTEGER, INTENT(IN) :: NP
499  CHARACTER(LEN=*), INTENT(IN):: FNM
500 
501  CHARACTER(LEN=80):: cstr1,cstr2,cstr3
502  LOGICAL FEXIST,ISLOCAL
503  INTEGER :: I,J,IERR,IOS,STATUS, N_PROBE
504  INTEGER :: PROCMAX, IBND, charnum
505  CHARACTER(LEN=120) :: pathnfile
506  CHARACTER(LEN=4) :: OFLAG
507  TYPE(time) :: OTIME
508  INTEGER(ITIME) :: OSTEP
509  if(dbg_set(dbg_sbr)) &
510  & write(ipt,*) "START: SET_PROBES;"
511 
512 
513  probe_on=p_on
514  global_probes=np
515  local_probes=0
516  input_fname=fnm
517 
518  IF (.not. probe_on) THEN
519  if(dbg_set(dbg_log)) &
520  & write(ipt,*) "! Time Series Probes are off"
521  return
522  ELSE
523  if(dbg_set(dbg_log)) then
524  write(ipt,*) "! Time Series Probes are on"
525  write(ipt,*) "! Setting up Probes:"
526  end if
527  END IF
528 
529 
530 
531  charnum = index(probes_file,".nml")
532  if (charnum /= len_trim(probes_file)-3)&
533  & CALL warning("PROBES FILE does not end in .nml", &
534  & trim(probes_file))
535  ! OPEN FILE - try both with appending input dir and without!
536  pathnfile = trim(input_dir)//trim(probes_file)
537  INQUIRE(file=pathnfile,exist=fexist)
538  IF(fexist) THEN
539  Call fopen(probeunit,trim(pathnfile),'cfr')
540  ELSE
541  pathnfile = trim(probes_file)
542  Call fopen(probeunit,trim(pathnfile),'cfr')
543  ! LET FOPEN CALL ERROR IF FILES DOES NOT EXIST
544  END IF
545 
546 
547  CALL alloc_probe(glb_probe,global_probes)
548 
549  !------------------------------------------------------------------------------|
550  ! READ TIME SERIES NAME LIST |
551  !------------------------------------------------------------------------------|
552  n_probe = 0
553  DO
554 
555  CALL init_nml_probe
556 
557  islocal = .false.
558 
559  READ(unit=probeunit, nml=nml_probe,iostat=ios)
560  if (ios /= 0 ) exit
561 
562  n_probe = n_probe +1
563  if (n_probe > global_probes) exit ! To prevent sigsev...
564 
565 
566  !------------------------------------------------------------------------------|
567  ! SET FUNDEMENTAL DATA
568  !------------------------------------------------------------------------------|
569  glb_probe(n_probe)%K_ONE = probe_levels(1)
570  glb_probe(n_probe)%K_TWO = probe_levels(2)
571  glb_probe(n_probe)%D_LOC_GL = probe_location
572 
573  glb_probe(n_probe)%D_TIT =probe_title
574  glb_probe(n_probe)%D_DES =probe_description
575  glb_probe(n_probe)%VAR =probe_variable
576  glb_probe(n_probe)%VNAME =probe_var_name
577 
578  !------------------------------------------------------------------------------|
579  ! SET TIME FOR PROBES
580  !------------------------------------------------------------------------------|
581  glb_probe(n_probe)%O_NEXT = starttime
582  CALL ideal_time_string2time(probe_interval,oflag,otime,ostep)
583  IF (oflag == 'time') THEN ! IF OUTPUT TIME INTERVAL WAS SPECIFIED
584  glb_probe(n_probe)%O_INT = otime
585 
586  ELSE IF(oflag == 'step') THEN ! IF OUTPUT CYCLE INTERVAL WAS SPECIFIED
587  glb_probe(n_probe)%O_INT = imdti * ostep
588 
589  END IF
590 
591  IF (glb_probe(n_probe)%O_INT <= zerotime) CALL fatal_error&
592  &('ERROR IN PROBE SETUP: Time series output interval is less than or equal to zero!')
593 
594  !------------------------------------------------------------------------------|
595  ! POINT STORAGE DATA AND LOCATION TYPE |
596  !------------------------------------------------------------------------------|
597  CALL probe_store(glb_probe(n_probe))
598 
599  IF (glb_probe(n_probe)%MINE) THEN
600  local_probes=local_probes+1
601 
602  glb_probe(n_probe)%O_NUM =n_probe
603 
604 
605  END IF
606 
607  END DO
608 
609  IF(global_probes .ne. n_probe) THEN
610 
611  if(dbg_set(dbg_log)) then
