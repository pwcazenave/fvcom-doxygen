283  implicit none
284  character(len=* ), intent(in ) :: fname
285  character(len=* ), intent(in ) :: argname
286  character(len=80), intent(out) :: argval
287  integer, intent(out) :: ierr

291  !------------------------------------------
292  character(len=80) :: temp,temp2,text_line,inpline
293  integer :: locex,eqloc,length,i,linecnt
294 
295  ierr = 0
296  open(10,file=trim(fname)) ; rewind(10)
297 
298  linecnt = 0
299  !move forward to start line if present
300  if(present(start))then
301  do i=1,start-1
302  read(10,*)
303  linecnt = linecnt + 1
304  end do
305  endif
306 
307  do while(.true.)
308  read(10,'(A)',end=20) inpline
309  linecnt = linecnt + 1
310 
311  ! determine location of comment character and remove blanks
312  locex = index(inpline,com_char)
313  if(locex == 0)locex = len_trim(inpline)+1
314  temp = inpline(1:locex-1)
315  text_line = temp
316 
317  ! set line length
318  length = len_trim(text_line)
319 
320  ! ensure "=" exist and determine location
321  eqloc = index(text_line,"=")
322 
323 
324  ! split off variable name and value strings
325  temp2 = text_line(1:eqloc-1)
326  if(adjustl(trim(temp2)) == trim(argname))then
327  argval = adjustl(text_line(eqloc+1:length))
328  line = linecnt
329  close(10)
330  return
331  endif
332 
333  ! exit if finish is present
334  if(present(finish))then
335  if(linecnt==finish)exit
336  endif
337  end do
338 
339  ! not found, return error = 1
340  20 continue
341  ierr = 1
342  close(10)

349  Subroutine check_arg_type(argval,argtype)
350  implicit none
351  character(len=80), intent(out) :: argtype
352  character(len=80), intent(in ) :: argval
353 
354  character(len=16) :: numchars
355  integer :: largval
356  integer :: dotloc

358  numchars = "0123456789+-Ee. "
359 
360  ! argument length
361  largval = len_trim(adjustl(argval))
362 
363  ! check if logical
364  if(largval == 1)then
365  if(argval(1:1) == "T" .or. argval(1:1) == "F")then
366  argtype = 'logical'
367  return
368  endif
369  endif
370 
371  ! check if it is a string (contains characters other than 0-9,+,-,E,e,.)
372  do i=1,largval

374  argtype = "string"
375  return
376  endif
377  end do
378

381  if(dotloc /= 0) then

383  else

385  end if
