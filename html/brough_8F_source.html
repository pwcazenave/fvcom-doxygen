30 ! Calculate Bottom Drag Coefficient based on Bottom Roughness !
31 ! note: !
32 ! when the log function derived from the constant stress log-viscous !
33 ! layer is applied to an estuary, if the value of z0 is close to !
34 ! (zz(kbm1)-z(kb)*dt1, drag coefficient "cbc" could become a huge !
35 ! number due to near-zero value of alog function. In our application !
36 ! we simply cutoff at cbc=0.005. One could adjust this cutoff value !
37 ! based on observations or his or her experiences. !
38 ! CALCULATES: WUBOT(N), WVBOT(N) : BOTTOM SHEAR STRESSES !
39 !==============================================================================|
40 
41 SUBROUTINE bottom_roughness
42 
43  !==============================================================================!
44  USE all_vars
45  USE mod_utils
46  USE mod_wd
47  USE mod_par
48 # if defined (SEDIMENT) && (WAVE_CURRENT_INTERACTION)
49  USE mod_sed
50  USE mod_bbl
51 # endif
52 
53  IMPLICIT NONE
54  INTEGER :: I,II
55  REAL(SP), PARAMETER :: KAPPA = .40_sp !!VON KARMAN LENGTH SCALE
56  REAL(SP), PARAMETER :: VK2 = .160_sp !!KAPPA SQUARED
57  REAL(SP) :: ZTEMP,BTPS,RR,U_TAUB,Z0B_GOTM,CFF
58 
59 ! USED IN 2D MODEL ONLY
60 ! REAL(SP), PARAMETER :: CONST_CD=.0015_SP !! CD SET CONSTANT TO THIS VALUE
61  REAL(SP), PARAMETER :: ALFA = .166667_sp, & !! POWER OF WATER DEPTH
62  nn = 0.02_sp !! FACTOR TO DIVIDE
63 ! REAL(SP), PARAMETER :: CFMIN = .0025_SP, & !! DEEP WATER VALUE
64 ! H_BREAK = 1.0_SP, & !!
65 ! THETA = 10._SP, & !!
66 ! LAMB = 0.3333333333_SP
67  !==============================================================================!
68 
69  if(dbg_set(dbg_sbr)) write(ipt,*) "Start: BOTTOM_ROUGHNESS"
70 
71 # if !defined (TWO_D_MODEL)
72  !
73  ! SET CONSTANTS
74  !
75 
76  SELECT CASE(bottom_roughness_type)
77  !==============================================================================|
78  CASE(br_orig) !USE ORIGINAL FVCOM FORM FOR BOTTOM FRICTION |
79  !==============================================================================|
80 
81  ! SET A EFFECTIVE MAXIMUM FOR CBC USING THE DEPTH
82 !! WHERE (DT1 > 3.0_SP)
83 !! CBC = VK2/(LOG((ZZ1(:,KBM1)-Z1(:,KB))*DT1/CC_Z0B))**2
84 !! ELSEWHERE
85 !! CBC = VK2/(LOG((ZZ1(:,KBM1)-Z1(:,KB))*3.0/CC_Z0B))**2
86 !! END WHERE
87  WHERE (dt1(1:nt) > 3.0_sp)
88  cbc(1:nt) = vk2/(log((zz1(1:nt,kbm1)-z1(1:nt,kb))*dt1(1:nt)/cc_z0b(1:nt)))**2
89  ELSEWHERE
90  cbc(1:nt) = vk2/(log((zz1(1:nt,kbm1)-z1(1:nt,kb))*3.0/cc_z0b(1:nt)))**2
91  END WHERE
92 
93  ! SET A MINIMUM VALUE FOR CBC
94  WHERE (cbc < cbcmin)
95  cbc=cbcmin
96  END WHERE
97 
98 
99  !==============================================================================|
100  CASE(br_gotm) !GOTM FORMULATION FOR BOTTOM FRICTION |
101  !==============================================================================|
102 
103  !----Convert Input Z0B to GOTMS H0B
104  ! H0B = Z0B/.03
105  ! DAS fixed bug to match gotm's friction.f90
106  DO i=1,n
107  u_taub = 0.0_sp
108  DO ii=1,40
109  IF (umol <= 0.0_sp) THEN
110  z0b_gotm=cc_z0b(i) !0.03*H0B
111  ELSE
112  z0b_gotm=0.1_sp*umol/max(umol,u_taub)+cc_z0b(i) !0.03*H0B
113  END IF
114  ztemp=(zz1(i,kbm1)-z1(i,kb))*dt1(i)
115  rr=kappa/(log((z0b_gotm+ztemp)/z0b_gotm))
116  u_taub = rr*sqrt( u(i,kbm1)*u(i,kbm1) + v(i,kbm1)*v(i,kbm1) )
117  END DO
118  cbc(i) = rr*rr
119  END DO
120 
121 
122  CASE DEFAULT
123  CALL fatal_error ("BROUGH: UNKNOWN BOTTOM_ROUGHNESS_TYPE:"&
124  & ,trim(bottom_roughness_type) )
125  END SELECT
126 
127 # else
128 ! 1.CONSTANT CD
129 ! CBC = CONST_CD
130 ! 2. formula 1
131  WHERE (dt1 < 3.0_sp)
132  cbc = 0.0027_sp
133  ELSEWHERE
134  cbc = grav*(dt1**alfa/nn)**(-2)
135  END WHERE
136 ! 3. formula 2
137 ! CBC = CFMIN*(1+(H_BREAK/DT1)**THETA)**(LAMB/THETA)
138 # endif
139 
140  !==============================================================================|
141  ! CALCULATE SHEAR STRESS ON BOTTOM --> WUBOT/WVBOT |
142  !==============================================================================|
143  DO i = 1, n
144 # if !defined (TWO_D_MODEL)
145  btps= cbc(i)*sqrt(u(i,kbm1)**2+v(i,kbm1)**2)
146  wubot(i) = -btps * u(i,kbm1)
147  wvbot(i) = -btps * v(i,kbm1)
148 # if defined (WET_DRY)
149  cff=0.75_sp*dz1(i,kbm1)*d1(i)
150  wubot(i)=sign(1.0_sp,wubot(i))*min(abs(wubot(i)),abs(u(i,kbm1)*cff/dti))
151  wvbot(i)=sign(1.0_sp,wvbot(i))*min(abs(wvbot(i)),abs(v(i,kbm1)*cff/dti))
152 # endif
153  !for plb case only
154 # if defined (PLBC)
155  btps= 0.0015*sqrt(ua(i)**2+va(i)**2)
156  wubot(i) = -btps * ua(i)
157  wvbot(i) = -btps * va(i)
158 # endif
159 # else
160  btps= cbc(i)*sqrt(ua(i)**2+va(i)**2)
161  wubot(i) = -btps * ua(i)
162  wvbot(i) = -btps * va(i)
163 # endif
164 
165  END DO
166 
167 # if defined (SEDIMENT) && (WAVE_CURRENT_INTERACTION)
168  if(mb_bbl_use)then
169  call mb_bbl
170  elseif(sg_bbl_use)then
171  call sg_bbl
172  elseif(ssw_bbl_use)then
173  call ssw_bbl
174 ! print*,'WUBOT_OLD=', WUBOT(5551)
175 ! print*,'WVBOT_OLD=', WVBOT(5551)
176 ! CALL N2E2D(bustr,WUBOT)
177 ! CALL N2E2D(bvstr,WVBOT)
178 ! WUBOT = -WUBOT
179 ! WVBOT = -WVBOT
180 ! print*,'WUBOT_NEW=', WUBOT(5551)
181 ! print*,'WVBOT_NEW=', WVBOT(5551)
182  end if
183 
184 # endif
185 
186  !==============================================================================|
187  ! Calculate shear stress on nodes (x-component, y-component, magnitude)
188  !==============================================================================|
189 # if defined (MULTIPROCESSOR)
190  IF(par)CALL aexchange(ec,myid,nprocs,wubot,wvbot)
191 # endif
192  taubm = sqrt(wubot**2 + wvbot**2)
193 
194 # if defined (SEDIMENT) && (WAVE_CURRENT_INTERACTION)
195  if(mb_bbl_use.or.sg_bbl_use.or.ssw_bbl_use)then
196  wubot_n = -bustr
197  wvbot_n = -bvstr
198  else
199 # endif
200  CALL e2n2d(wubot,wubot_n)
201  CALL e2n2d(wvbot,wvbot_n)
202 # if defined (SEDIMENT) && (WAVE_CURRENT_INTERACTION)
203  endif
204 # endif
205  taubm_n = sqrt(wubot_n**2 + wvbot_n**2)
206 # if defined (MULTIPROCESSOR)
207  IF(par)CALL aexchange(nc,myid,nprocs,taubm_n)
208 # endif
209 
210  if(dbg_set(dbg_sbr)) write(ipt,*) "End: BOTTOM_ROUGHNESS"
211 
212  RETURN
213 END SUBROUTINE bottom_roughness
