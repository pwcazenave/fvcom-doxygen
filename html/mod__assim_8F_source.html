5620 !==============================================================================|
5621 ! determine if point (x0,y0) is in triangle defined by nodes (xt(3),yt(3)) |
5622 ! using algorithm used for scene rendering in computer graphics |
5623 ! algorithm works well unless particle happens to lie in a line parallel |
5624 ! to the edge of a triangle. |
5625 ! This can cause problems if you use a regular grid, say for idealized |
5626 ! modelling and you happen to see particles right on edges or parallel to |
5627 ! edges. |
5628 !==============================================================================|
5629 
5630  USE mod_prec
5631  IMPLICIT NONE
5632  REAL(SP), INTENT(IN) :: x0,y0
5633  REAL(SP), INTENT(IN) :: xt(3),yt(3)
5634  REAL(SP) :: f1,f2,f3
5635  REAL(SP) :: x1(2)
5636  REAL(SP) :: x2(2)
5637  REAL(SP) :: x3(2)
5638  REAL(SP) :: p(2)
5639 
5640 !------------------------------------------------------------------------------|
5641  IF (dbg_set(dbg_sbr)) WRITE(ipt,*) "START: ISINTRIANGLE1"
5642 
5643  isintriangle1 = .false.
5644 
5645  IF(y0 < minval(yt) .OR. y0 > maxval(yt))THEN
5646  isintriangle1 = .false.
5647  RETURN
5648  END IF
5649  IF(x0 < minval(xt) .OR. x0 > maxval(xt))THEN
5650  isintriangle1 = .false.
5651  RETURN
5652  END IF
5653 
5654  f1 = (y0-yt(1))*(xt(2)-xt(1)) - (x0-xt(1))*(yt(2)-yt(1))

5657  IF(f1*f3 >= 0.0_sp .AND. f3*f2 >= 0.0_sp) isintriangle1 = .true.
5658
