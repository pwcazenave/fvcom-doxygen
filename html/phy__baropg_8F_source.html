30 ! This subroutine is used to calculate the baroclinic pressure !
31 ! gradient in the standarded z-levels. The water column is divided !
32 ! into 600 standard levels, and pressure gradient is then determined !
33 ! at each level. The resulting pressure gradients are converted back !
34 ! to sigma-levels through vertical interpolation approach. !
35 !==============================================================================|
36 
37  SUBROUTINE phy_baropg
38 
39 !------------------------------------------------------------------------------|
40 
41  USE all_vars
42  USE sinter
43  IMPLICIT NONE
44  INTEGER, PARAMETER :: KBB=601
45  INTEGER, PARAMETER :: KBBM1=kbb-1
46  REAL(SP) RHOZ(m,kbbm1),RHOZTMP(kbbm1),PHY_Z(kbbm1)
47  REAL(SP) RHOS(kbm1),SIGZTMP(kbm1),SIG_Z(m,kbm1)
48  REAL(SP) PB1(0:kbbm1),PB2(0:kbbm1),PB3(0:kbbm1)
49  REAL(SP) PBXZ(kbbm1),PBYZ(kbbm1)
50  REAL(SP) PBXS(kbm1) ,PBYS(kbm1)
51  REAL(SP) AREAX1,AREAX2,AREAX3,AREAY1,AREAY2,AREAY3
52  REAL(SP) RHOZI1,RHOZI2,RHOZI3,TMP,TEMP,DELTZ
53  REAL(SP) GRAV1,GRAV2,GRAV3
54  INTEGER I,K,J1,J2,J3,NTMP
55 
56  ! USE RAMP CALCULATED IN 'internal_step.F'
57 
58 !--CALCULATE Z-LEVELS TO MAX DEPTH---------------------------------------------|
59 
60  deltz=hmax/float(kbbm1)
61 
62  DO k=1,kbbm1
63  phy_z(k)=(0.5_sp-float(k))*deltz
64  END DO
65 
66 !--LINEARLY INTERPOLATE TO OBTAIN DENSITY VALUES AT Z LEVELS-------------------|
67 
68  DO i=1,m
69  DO k=1,kbm1
70  sig_z(i,k)=zz(i,k)*dt(i)+et(i)
71  sigztmp(k)=sig_z(i,k)
72  rhos(k)=rho1(i,k)
73  END DO
74 
75  CALL sinter_extrp_down(sigztmp,rhos,phy_z,rhoztmp,kbm1,kbbm1)
76 
77  DO k=1,kbbm1
78  rhoz(i,k)=rhoztmp(k)
79  END DO
80  END DO
81 
82  DO i=1,n
83  j1=nv(i,1)
84  j2=nv(i,2)
85  j3=nv(i,3)
86  ntmp=0
87  pb1(0)=0.0_sp
88  pb2(0)=0.0_sp
89  pb3(0)=0.0_sp
90  DO k=1,kbbm1
91  tmp=float(k)*deltz
92  IF((h(j1) < tmp.OR.h(j2) < tmp.OR.h(j3) < tmp)) THEN
93  pb1(k)=0.0_sp
94  pb2(k)=0.0_sp
95  pb3(k)=0.0_sp
96  ELSE
97  rhozi1=0.5_sp*(rhoz(j2,k)+rhoz(j3,k))
98  rhozi2=0.5_sp*(rhoz(j3,k)+rhoz(j1,k))
99  rhozi3=0.5_sp*(rhoz(j1,k)+rhoz(j2,k))
100  grav1 =0.5_sp*(grav_n(j2)+grav_n(j3))
101  grav2 =0.5_sp*(grav_n(j3)+grav_n(j1))
102  grav3 =0.5_sp*(grav_n(j1)+grav_n(j2))
103  pb1(k)=pb1(k-1)+grav1*rhozi1*deltz
104  pb2(k)=pb2(k-1)+grav2*rhozi2*deltz
105  pb3(k)=pb3(k-1)+grav3*rhozi3*deltz
106  ntmp=ntmp+1
107  END IF
108  END DO
109  areax1=(vy(j3)-vy(j2))*deltz
110  areay1=(vx(j2)-vx(j3))*deltz
111  areax2=(vy(j1)-vy(j3))*deltz
112  areay2=(vx(j3)-vx(j1))*deltz
113  areax3=(vy(j2)-vy(j1))*deltz
114  areay3=(vx(j1)-vx(j2))*deltz
115  DO k=1,kbbm1
116  pbxz(k)=areax1*pb1(k)+areax2*pb2(k)+areax3*pb3(k)
117  pbyz(k)=areay1*pb1(k)+areay2*pb2(k)+areay3*pb3(k)
118  pbxz(k)=pbxz(k)/art(i)/deltz
119  pbyz(k)=pbyz(k)/art(i)/deltz
120  END DO
121 
122  DO k=1,kbm1
123  j1=nv(i,1)
124  j2=nv(i,2)
125  j3=nv(i,3)
126  sigztmp(k)=(sig_z(j1,k)+sig_z(j2,k)+sig_z(j3,k))/3.
127  END DO
128 
129  IF(ntmp == 0) THEN
130  DO k=1,kbm1
131  pbxs(k)=0.0_sp
132  pbys(k)=0.0_sp
133  END DO
134  ELSE IF(ntmp == 1) THEN
135  DO k=1,kbm1
136  pbxs(k)=pbxz(1)
137  pbys(k)=pbyz(1)
138  END DO
139  ELSE
140  CALL sinter_extrp_down(phy_z,pbxz,sigztmp,pbxs,ntmp,kbm1)
141  CALL sinter_extrp_down(phy_z,pbyz,sigztmp,pbys,ntmp,kbm1)
142  END IF
143 
144  DO k=1,kbm1
145  drhox(i,k)=-pbxs(k)*dt1(i)*dz1(i,k)*art(i)*ramp
146  drhoy(i,k)=-pbys(k)*dt1(i)*dz1(i,k)*art(i)*ramp
147  END DO
148  END DO
149 
150  RETURN
151  END SUBROUTINE phy_baropg
