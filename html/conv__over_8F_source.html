30 ! ADJUST THE VERTICAL WATER COLUMN WHEN UNSTABLE |
31 !==============================================================================|
32 
33  SUBROUTINE conv_over
34 
35 !==============================================================================|
36  USE all_vars
37  USE mod_utils
38  IMPLICIT NONE
39  REAL(SP):: AVE_T,AVE_S,AVE_R
40  INTEGER :: I,K,KK,J1,J2,J3
41 !==============================================================================|
42 
43  IF (dbg_set(dbg_sbr)) WRITE(ipt,*) "Start: conv_over"
44 
45 
46 !--APPROXIMATE CONVECTIVE OVERTURNING------------------------------------------!
47 
48  DO i=1,m
49  DO k=kbm1,2,-1
50  DO kk=k-1,1,-1
51  IF(rho1(i,k) < rho1(i,kk)) THEN
52  ave_t = sum( t1(i,kk:k))/float(k-kk+1)
53  ave_s = sum( s1(i,kk:k))/float(k-kk+1)
54  ave_r = sum(rho1(i,kk:k))/float(k-kk+1)
55  t1(i,kk:k) = ave_t
56  s1(i,kk:k) = ave_s
57  rho1(i,kk:k) = ave_r
58  END IF
59  END DO
60  END DO
61  END DO
62 
63 !-----RECALCULATE ELEMENT-BASED VALUES OF SALINITY/TEMP/DENSITY----------------!
64 
65  DO i=1,n
66  j1=nv(i,1) ; j2 = nv(i,2) ; j3 = nv(i,3)
67  DO k=1,kbm1
68  t(i,k) = one_third*( t1(j1,k)+ t1(j2,k)+ t1(j3,k))
69  s(i,k) = one_third*( s1(j1,k)+ s1(j2,k)+ s1(j3,k))
70  END DO
71  END DO
72 
73  IF (dbg_set(dbg_sbr)) WRITE(ipt,*) "End: conv_over"
74 
75  RETURN
76  END SUBROUTINE conv_over
