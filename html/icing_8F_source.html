29 SUBROUTINE icing(NOW)
30  USE mod_utils
31  USE mod_force
32  USE all_vars
33  USE mod_time
34  IMPLICIT NONE
35  TYPE(time), INTENT(IN) :: NOW
36  REAL(SP), PARAMETER :: Tfreeze = -1.75_sp
37  REAL(SP), DIMENSION(0:MT) :: ICING_WNDSPD
38 
39  IF(dbg_set(dbg_log)) write(ipt,*) "Start Icing Update"
40 
41  ! GET THE FORCING DATA
42  CALL update_icing(now,icing_satmp,icing_wndx,icing_wndy)
43 
44  icing_wndspd = sqrt(icing_wndy**2 + icing_wndx**2)
45  icing_wndspd(0) = 0.0_sp
46 
47 
48  IF(dbg_set(dbg_io))THEN
49  WRITE(ipt,*) "min/max(SAT)",minval(icing_satmp(1:mt)),maxval(icing_satmp(1:mt))
50  WRITE(ipt,*) "min/max(WNDSPD)",minval(icing_wndspd(1:mt)),maxval(icing_wndspd(1:mt))
51  WRITE(ipt,*) "min/max(T1(:,1))",minval(t1(1:mt,1)),maxval(t1(1:mt,1))
52  END IF
53 
54  icing_0kts = (tfreeze - icing_satmp) /&
55  (1.0_sp + 0.4_sp *(t1(:,1) -tfreeze))
56 
57  WHERE (icing_0kts < 0.0_sp) icing_0kts = 0.0_sp
58 
59  icing_10kts = (icing_wndspd + 5.0_sp) * icing_0kts
60 
61  icing_0kts = icing_0kts * icing_wndspd
62 
63 
64 END SUBROUTINE icing
